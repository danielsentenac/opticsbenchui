75 QLOG_ERROR () << "Error setting mode for device " << _cur_dev;

141int DriverStanda_uSMC::Move(string actuatorSetting,

157 QLOG_DEBUG () << "DriverStanda_uSMC::Move> to position " << dest_pos;

164

172 }
173
