123 void AcquisitionThread::run() {
124  if (suspend) {
125  suspend = false;
126 
127  /* create a new data File */
128 #ifdef NO_HDF5
129  Utils::EmitWarning(this, __FUNCTION__,
130  "HDF5 support is disabled. Acquisition data will not be saved.");
131 #else
132  file_id = H5Fcreate(filename.toStdString().c_str(), H5F_ACC_TRUNC,
133  H5P_DEFAULT, H5P_DEFAULT);
134  if (file_id < 0) {
135  const QString message =
136  "Unable to open file - " + filename + " - Aborting scan";
137  Utils::EmitWarning(this, __FUNCTION__, message);
138  return;
139  }
140 #endif
141  record = 0;
142  int lastrecord = sequenceList.size();
143  QLOG_DEBUG() << "AcquisitionThread::run> Number of sequences " << lastrecord;
144  // run acquisition sequence
145  QLOG_INFO() << "AcquisitionThread::run> start Acquisition : "
146  << Utils::CurrentTimestampString();
147  bool stoppedEarly = false;
148  while (record < lastrecord) {
149  if (shouldStop()) {
150  stoppedEarly = true;
151  break;
152  }
153  AcquisitionSequence *sequence = sequenceList.at(record);
154  emit getAcquiring(sequence->record);
155  sequence->etime = Utils::GetTimeMicroseconds();
156  int cur_record = record;
157  this->execute(sequence);
158  this->nextRecord(sequence, cur_record);
159  QLOG_DEBUG() << "AcquisitionThread::run> Start sequence " << sequence->seq_record;
160  this->saveData(sequence, cur_record);
161  if (sequence->settings.contains("ANALYSIS=START")) {
162  QLOG_INFO() << "AcquisitionThread::run> Analysis requested by settings";
163  emit requestAnalysis();
164  }
165  if (shouldStop()) {
166  stoppedEarly = true;
167  break;
168  }
169  if (sequence->sleep > 0) {
170  usleep(sequence->sleep);
171  }
172  sequence->etime = Utils::GetTimeMicroseconds() - sequence->etime;
173  if (shouldStop()) {
174  stoppedEarly = true;
175  break;
176  }
177 #ifndef NO_HDF5
178  if (H5Fflush(file_id, H5F_SCOPE_GLOBAL) < 0) {
179  QLOG_WARN() << " File flushing failed !";
180  }
181  H5garbage_collect();
182 #endif
183  QLOG_INFO() << "AcquisitionThread::run> sequence " << sequence->seq_record
184  << ": etime = " << (int)sequence->etime;
185  }
186  QLOG_INFO() << "AcquisitionThread::run> stop Acquisition : "
187  << Utils::CurrentTimestampString();
188  emit getAcquiring(record);
189  filenumber++;
190  emit getFilenumber(filenumber);
191 #ifndef NO_HDF5
192  // close all groups
193  for (int i = ids.size(); i >= 0; i--) {
194  H5Gclose(ids.at(i));
195  }
196  ids.clear();
197  status = H5Fclose(file_id);
198 #endif
199  // Close eventually open camera
200  for (int i = 0; i < cameraList.size(); i++) {
201  Camera *camera = cameraList.at(i);
202  if (isopencamerawindow.at(i) == false) {
203  camera->stop();
204  }
205  }
206  for (int i = 0; i < sequenceList.size(); i++) {
207  AcquisitionSequence *sequence = sequenceList.at(i);
208  emit getAcquiring(sequence->record);
209  if (sequence->instrumentType == "DAC") {
210  QLOG_DEBUG() << "AcquisitionThread::execute> Update Db DAC values...";
211  // Update DB Dac values
212  if (dac != NULL) {
213  dac->updateDBValues(sequence->instrumentName);
214  }
215  }
216  if (sequence->instrumentType == "COMEDICOUNTER") {
217  QLOG_DEBUG() << "AcquisitionThread::execute> Update Db COMEDI COUNTER values...";
218  // Update DB Counter values
219  if (comedicounter != NULL) {
220  comedicounter->updateDBValues(sequence->instrumentName);
221  }
222  }
223  if (sequence->instrumentType == "COMEDIDAC") {
224  QLOG_DEBUG() << "AcquisitionThread::execute> Update Db COMEDI DAC values...";
225  // Update DB Comedi Dac values
226  if (comedidac != NULL) {
227  comedidac->updateDBValues(sequence->instrumentName);
228  }
229  }
230  if (sequence->instrumentType == "RASPIDAC") {
231  QLOG_DEBUG() << "AcquisitionThread::execute> Update Db RASPI DAC values...";
232  // Update DB Raspi Dac values
233  if (raspidac != NULL) {
234  raspidac->updateDBValues(sequence->instrumentName);
235  }
236  }
237  if (sequence->instrumentType == "MOTOR") {
238  QLOG_DEBUG() << "AcquisitionThread::execute> Update Db MOTOR values...";
239  // Emit last position
240  QString positionQString;
241  positionQString.setNum (sequence->position, 'f',3);
242  emit getPosition(positionQString);
243  }
244  if (sequence->instrumentType == "SUPERK") {
245  QLOG_DEBUG() << "AcquisitionThread::execute> Update Db SUPERK values...";
246  // Emit last data
247  QString superKData = QString::number(sequence->superkPowerValue) + " " +
248  QString::number(sequence->superkNdValue) + " " +
249  QString::number(sequence->superkSwpValue) + " " +
250  QString::number(sequence->superkLwpValue);
251  emit getSuperKData(superKData);
252  }
253  delete sequence;
254  }
255  sequenceList.clear();
256  if (stoppedEarly) {
257  emit getAcquiring(lastrecord);
258  }
259  }
260  suspend = true;
261  QLOG_DEBUG() << "AcquisitionThread::run> End of thread";
262 }
263 void AcquisitionThread::execute(AcquisitionSequence *sequence) {
264  if (shouldStop()) {
265  return;
266  }
267  // New acquisition
268  QLOG_INFO() << "AcquisitionThread::execute> instrument type " << sequence->instrumentType;
269  //emit getAcquiring(record);
270 
271  QLOG_INFO() << "AcquisitionThread::execute> Execute record " << record
272  << " instrument " << sequence->instrumentType << ":"
273  << sequence->instrumentName << " remaining loops "
274  << sequence->remainingLoops << ":" << sequence->loopends << ":"
275  << sequence->loopends;
276 
277  if (sequence->instrumentType == "MOTOR") {
278  motor->connectMotor(sequence->instrumentName);
279  activeMotorName = sequence->instrumentName;
280 
281  float motorValue = sequence->motorValue;
282  if (sequence->motorAction == "MOVEREL" && isSnakeReverseForRecord(record)) {
283  motorValue = -motorValue;
284  }
285 
286  if (sequence->motorAction == "MOVEREL") {
287  if (motorValue < 0) {
288  motor->moveBackward(sequence->instrumentName, (-motorValue));
289  } else {
290  motor->moveForward(sequence->instrumentName, motorValue);
291  }
292  }
293  else if (sequence->motorAction == "MOVEABS") {
294  motor->moveAbsolute(sequence->instrumentName, sequence->motorValue);
295  }
296  // Wait for motor movement to be completed
297  while (motor->getOperationComplete(sequence->instrumentName) <= 0) {
298  if (shouldStop()) {
299  motor->stopMotor(sequence->instrumentName);
300  break;
301  }
302  usleep(100);
303  motor->operationComplete();
304  }
305  if (shouldStop()) {
306  sequence->status = false;
307  activeMotorName.clear();
308  return;
309  }
310  // Update motor position
311  QString positionQString;
312  // Save motor position data
313  sequence->position = motor->getPosition(sequence->instrumentName);
314  positionQString.setNum (sequence->position, 'f',3);
315  emit getPosition(positionQString);
316  QLOG_INFO() << "AcquisitionThread::execute> " << sequence->instrumentName
317  << " new position:" << positionQString;
318  sequence->status = true;
319  activeMotorName.clear();
320  }
321  else if (sequence->instrumentType == "SUPERK") {
322  superk->connectSuperK(sequence->instrumentName);
323  if (sequence->superkPowerValue > -88888) {
324  superk->setPower(sequence->instrumentName,
325  sequence->superkPowerValue * 10);
326  }
327  if (sequence->superkNdValue > -88888) {
328  superk->setNd(sequence->instrumentName, sequence->superkNdValue * 10);
329  }
330  if (sequence->superkSwpValue > -88888) {
331  superk->setSwp(sequence->instrumentName, sequence->superkSwpValue * 10);
332  }
333  if (sequence->superkLwpValue > -88888) {
334  superk->setLwp(sequence->instrumentName, sequence->superkLwpValue * 10);
335  }
336  if (sequence->superkBwValue > -88888 && sequence->superkCwValue > -88888) {
337  superk->setLwp(sequence->instrumentName,
338  (sequence->superkCwValue - sequence->superkBwValue / 2) *
339  10);
340  superk->setSwp(sequence->instrumentName,
341  (sequence->superkCwValue + sequence->superkBwValue / 2) *
342  10);
343  }
344  if (sequence->superkRPowerValue > -88888) {
345  sequence->superkPowerValue =
346  (float)superk->getPower(sequence->instrumentName) / 10 +
347  sequence->superkRPowerValue;
348  QLOG_INFO() << "sequence->superkPowerValue=" << sequence->superkPowerValue
349  << " " << superk->getPower(sequence->instrumentName) << " + "
350  << sequence->superkRPowerValue;
351  superk->setPower(sequence->instrumentName,
352  sequence->superkPowerValue * 10);
353  }
354  if (sequence->superkRNdValue > -88888) {
355  sequence->superkNdValue =
356  (float)superk->getNd(sequence->instrumentName) / 10 +
357  sequence->superkRNdValue;
358  superk->setNd(sequence->instrumentName, sequence->superkNdValue * 10);
359  }
360  if (sequence->superkRSwpValue > -88888) {
361  sequence->superkSwpValue =
362  (float)superk->getSwp(sequence->instrumentName) / 10 +
363  sequence->superkRSwpValue;
364  superk->setSwp(sequence->instrumentName, sequence->superkSwpValue * 10);
365  }
366  if (sequence->superkRLwpValue > -88888) {
367  sequence->superkLwpValue =
368  (float)superk->getLwp(sequence->instrumentName) / 10 +
369  sequence->superkRLwpValue;
370  superk->setLwp(sequence->instrumentName, sequence->superkLwpValue * 10);
371  }
372  if (sequence->superkRBwValue > -88888 && sequence->superkCwValue > -88888) {
373  QLOG_INFO() << "sequence->superkRBwValue " << sequence->superkRBwValue;
374  QLOG_INFO() << "sequence->superkCwValue " << sequence->superkCwValue;
375 
376  sequence->superkBwValue =
377  ((float)superk->getSwp(sequence->instrumentName) -
378  (float)superk->getLwp(sequence->instrumentName)) /
379  10 +
380  sequence->superkRBwValue;
381  superk->setLwp(sequence->instrumentName,
382  (sequence->superkCwValue - sequence->superkBwValue / 2) *
383  10);
384  superk->setSwp(sequence->instrumentName,
385  (sequence->superkCwValue + sequence->superkBwValue / 2) *
386  10);
387  }
388  if (sequence->superkBwValue > -88888 && sequence->superkRCwValue > -88888) {
389  QLOG_INFO() << "sequence->superkBwValue " << sequence->superkBwValue;
390  QLOG_INFO() << "sequence->superkRCwValue " << sequence->superkRCwValue;
391 
392  sequence->superkCwValue =
393  ((float)superk->getLwp(sequence->instrumentName) +
394  (float)superk->getSwp(sequence->instrumentName)) /
395  10 / 2 +
396  sequence->superkRCwValue;
397  superk->setLwp(sequence->instrumentName,
398  (sequence->superkCwValue - sequence->superkBwValue / 2) *
399  10);
400  superk->setSwp(sequence->instrumentName,
401  (sequence->superkCwValue + sequence->superkBwValue / 2) *
402  10);
403  }
404  if (sequence->superkRBwValue > -88888 && sequence->superkRCwValue > -88888) {
405  QLOG_INFO() << "sequence->superkRBwValue " << sequence->superkRBwValue;
406  QLOG_INFO() << "sequence->superkRCwValue " << sequence->superkRCwValue;
407  sequence->superkCwValue =
408  ((float)superk->getLwp(sequence->instrumentName) +
409  (float)superk->getSwp(sequence->instrumentName)) /
410  10 / 2 +
411  sequence->superkRCwValue;
412  sequence->superkBwValue =
413  ((float)superk->getSwp(sequence->instrumentName) -
414  (float)superk->getLwp(sequence->instrumentName)) /
415  10 +
416  sequence->superkRBwValue;
417  superk->setLwp(sequence->instrumentName,
418  (sequence->superkCwValue - sequence->superkBwValue / 2) *
419  10);
420  superk->setSwp(sequence->instrumentName,
421  (sequence->superkCwValue + sequence->superkBwValue / 2) *
422  10);
423  }
424 
425  // Wait for superk to be completed
426  while (superk->getOperationComplete(sequence->instrumentName) <= 0) {
427  usleep(100);
428  superk->operationComplete();
429  }
430  sequence->status = true;
431  }
432  else if (sequence->instrumentType == "DAC") {
433  QLOG_INFO() << "AcquisitionThread::execute> connecting DAC ..." << sequence->instrumentName;
434  dacsuccess = dac->connectDac(sequence->instrumentName);
435  QLOG_INFO() << "AcquisitionThread::execute> DAC " << sequence->instrumentName
436  << " connected " << dacsuccess;
437 
438  if (dacsuccess == true) {
439  if (sequence->dacRValue != 0) {
440  dacsuccess = dac->setDacRValue(sequence->instrumentName,
441  sequence->dacOutput,
442  sequence->dacRValue);
443  sequence->dacValue =
444  dac->getDacValue(sequence->instrumentName, sequence->dacOutput);
445  }
446  else {
447  dacsuccess = dac->setDacValue(sequence->instrumentName,
448  sequence->dacOutput,
449  sequence->dacValue);
450  }
451  }
452  sequence->status = dacsuccess;
453  //emit getDacStatus(dacsuccess);
454  }
455  else if (sequence->instrumentType == "COMEDICOUNTER") {
456  if (!comedicounter) {
457  Utils::EmitWarning(this, __FUNCTION__,
458  "COMEDICOUNTER is not configured");
459  sequence->status = false;
460  return;
461  }
462  QLOG_INFO() << "AcquisitionThread::execute> connecting COMEDI COUNTER ..." << sequence->instrumentName
463  << " ComediCounter pointer " << comedicounter;
464  comedicountersuccess = comedicounter->connectComedi(sequence->instrumentName);
465  QLOG_INFO() << "AcquisitionThread::execute> COMEDI COUNTER " << sequence->instrumentName
466  << " connected " << comedicountersuccess;
467 
468  if (comedicountersuccess == true) {
469  comedicountersuccess =
470  comedicounter->setComediValue(sequence->instrumentName,
471  sequence->comediOutput,
472  (void*)&sequence->comediValue);
473  QLOG_INFO() << "AcquisitionThread::execute> COMEDI COUNTER "
474  << sequence->instrumentName << " setComediValue "
475  << comedicountersuccess;
476  // Copy comedi data in sequence for saving later
477  comedicountersuccess =
478  comedicounter->getComediValue(sequence->instrumentName,
479  sequence->comediOutput,
480  sequence->comediData);
481  QLOG_INFO() << "AcquisitionThread::execute> COMEDI COUNTER "
482  << sequence->instrumentName << " getComediValue "
483  << comedicountersuccess;
484  }
485  sequence->status = comedicountersuccess;
486  //emit getComediStatus(comedicountersuccess);
487  }
488  else if (sequence->instrumentType == "COMEDIDAC") {
489  if (!comedidac) {
490  Utils::EmitWarning(this, __FUNCTION__,
491  "COMEDIDAC is not configured");
492  sequence->status = false;
493  return;
494  }
495  QLOG_INFO() << "AcquisitionThread::execute> connecting COMEDI DAC ..." << sequence->instrumentName;
496  comedidacsuccess = comedidac->connectComedi(sequence->instrumentName);
497  QLOG_INFO() << "AcquisitionThread::execute> COMEDI DAC " << sequence->instrumentName
498  << " connected " << comedidacsuccess;
499 
500  if (comedidacsuccess == true) {
501  if (sequence->dacRValue != 0) {
502  double dacValue;
503  comedidacsuccess |=
504  comedidac->getComediValue(sequence->instrumentName,
505  sequence->dacOutput, dacValue);
506  dacValue += sequence->dacRValue;
507  comedidacsuccess |=
508  comedidac->setComediValue(sequence->instrumentName,
509  sequence->dacOutput, (void*)&dacValue);
510  }
511  else {
512  comedidacsuccess |=
513  comedidac->setComediValue(sequence->instrumentName,
514  sequence->comediOutput,
515  (void*)&sequence->dacValue);
516  }
517  comedidacsuccess |=
518  comedidac->getComediValue(sequence->instrumentName,
519  sequence->dacOutput, sequence->dacValue);
520  }
521  sequence->status = comedidacsuccess;
522  //emit getComediStatus(comedidacsuccess);
523  }
524  else if (sequence->instrumentType == "RASPIDAC") {
525  if (!raspidac) {
526  Utils::EmitWarning(this, __FUNCTION__,
527  "RASPIDAC is not configured");
528  sequence->status = false;
529  return;
530  }
531  QLOG_INFO() << "AcquisitionThread::execute> connecting RASPI DAC ..." << sequence->instrumentName;
532  raspidacsuccess = raspidac->connectRaspi(sequence->instrumentName);
533  QLOG_INFO() << "AcquisitionThread::execute> RASPI DAC " << sequence->instrumentName
534  << " connected " << raspidacsuccess;
535 
536  if (raspidacsuccess == true) {
537  if (sequence->dacRValue != 0) {
538  double dacValue;
539  raspidacsuccess |=
540  raspidac->getRaspiValue(sequence->instrumentName,
541  sequence->dacOutput, dacValue);
542  dacValue += sequence->dacRValue;
543  raspidacsuccess |=
544  raspidac->setRaspiValue(sequence->instrumentName,
545  sequence->dacOutput, (void*)&dacValue);
546  }
547  else {
548  raspidacsuccess |=
549  raspidac->setRaspiValue(sequence->instrumentName,
550  sequence->comediOutput,
551  (void*)&sequence->dacValue);
552  }
553  raspidacsuccess |=
554  raspidac->getRaspiValue(sequence->instrumentName,
555  sequence->dacOutput, sequence->dacValue);
556  }
557  sequence->status = raspidacsuccess;
558  //emit getRaspiStatus(raspidacsuccess);
559  }
560 
561  else if (sequence->instrumentType == "CAMERA") {
562  imagesuccess = false;
563  bool cameraExists = false;
564  Camera *camera = nullptr;
565  for (int i = 0; i < cameraList.size(); i++) {
566  camera = cameraList.at(i);
567  if (camera->model == sequence->instrumentName) {
568  cameraExists = true;
569  }
570  break;
571  }
572  if (cameraExists == true) {
573  if (camera->suspend == true) {
574  QLOG_INFO() << "AcquisitionThread::execute> open CAMERA " << camera->model;
575  camera->start();
576  while (camera->has_started == false) {
577  usleep(100);
578  }
579  }
580  QLOG_INFO() << "AcquisitionThread::execute> wait for Image Acquisition ";
581  camera->mutex->lock();
582  while (!shouldStop()) {
583  if (camera->acqstart->wait(camera->mutex, 100)) {
584  break;
585  }
586  }
587  camera->mutex->unlock();
588  if (shouldStop()) {
589  return;
590  }
591  camera->mutex->lock();
592  while (!shouldStop()) {
593  if (camera->acqend->wait(camera->mutex, 100)) {
594  break;
595  }
596  }
597  camera->mutex->unlock();
598  if (shouldStop()) {
599  return;
600  }
601  if (sequence->settings.contains("SNAPSHOT") &&
602  !sequence->settings.contains("32") &&
603  !sequence->settings.contains("16")) {
604  sequence->setImage(camera->getSnapshot(),
605  camera->width,
606  camera->height);
607  }
608  else if (sequence->settings.contains("SNAPSHOT") &&
609  sequence->settings.contains("16")) {
610  sequence->setImage16(camera->getSnapshot16(),
611  camera->width,
612  camera->height);
613  }
614  else if (sequence->settings.contains("SNAPSHOT") &&
615  sequence->settings.contains("32")) {
616  sequence->setImage32(camera->getSnapshot32(),
617  camera->width,
618  camera->height);
619  }
620  sequence->setImageMin(camera->snapShotMin);
621  sequence->setImageMax(camera->snapShotMax);
622  QLOG_INFO() << "AcquisitionThread::execute> Save Image buffer in sequence from " << camera->vendor
623  << " imageMin " << camera->width << " " << sequence->imageMin
624  << " imageMax " << camera->height << " " << sequence->imageMax;
625  imagesuccess = true;
626  }
627  else {
628  QLOG_WARN() << "CAMERA does not exist...";
629  }
630 
631  //emit getCameraStatus(imagesuccess);
632  sequence->status = imagesuccess;
633  }
634  else if (sequence->instrumentType == "SLM") {
635  sequence->imgnum = sequence->imgnum + sequence->stepnum;
636  sequence->fullpath = sequence->imagepath + //sequence->keepzero +
637  QString::number(sequence->imgnum) + "." +
638  sequence->imagetype;
639  QLOG_INFO() << sequence->fullpath;
