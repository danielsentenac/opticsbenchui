112void AcquisitionThread::run() {
113 if (suspend) {
114 suspend = false;
115
116 /* create a new data File */
117 file_id = H5Fcreate(filename.toStdString().c_str(), H5F_ACC_TRUNC,
118 H5P_DEFAULT, H5P_DEFAULT);
119 if (file_id < 0) {
120 const QString message =
121 "Unable to open file - " + filename + " - Aborting scan";
122 Utils::EmitWarning(this, __FUNCTION__, message);
123 return;
124 }
125 record = 0;
126 int lastrecord = sequenceList.size();
127 QLOG_DEBUG() << "AcquisitionThread::run> Number of sequences " << lastrecord;
128 // run acquisition sequence
129 QLOG_INFO() << "AcquisitionThread::run> start Acquisition : "
130 << Utils::CurrentTimestampString();
131 while (record < lastrecord) {
132 AcquisitionSequence *sequence = sequenceList.at(record);
133 emit getAcquiring(sequence->record);
134 sequence->etime = Utils::GetTimeMicroseconds();
135 int cur_record = record;
136 this->execute(sequence);
137 this->nextRecord(sequence, cur_record);
138 QLOG_DEBUG() << "AcquisitionThread::run> Start sequence " << sequence->seq_record;
139 this->saveData(sequence, cur_record);
140 if (sequence->sleep > 0) {
141 usleep(sequence->sleep);
142 }
143 sequence->etime = Utils::GetTimeMicroseconds() - sequence->etime;
144 if (suspend) {
145 break;
146 }
147 if (H5Fflush(file_id, H5F_SCOPE_GLOBAL) < 0) {
148 QLOG_WARN() << " File flushing failed !";
149 }
150 H5garbage_collect();
151 QLOG_INFO() << "AcquisitionThread::run> sequence " << sequence->seq_record
152 << ": etime = " << (int)sequence->etime;
153 }
154 QLOG_INFO() << "AcquisitionThread::run> stop Acquisition : "
155 << Utils::CurrentTimestampString();
156 emit getAcquiring(record);
157 filenumber++;
158 emit getFilenumber(filenumber);
159 // close all groups
160 for (int i = ids.size(); i >= 0; i--) {
161 H5Gclose(ids.at(i));
162 }
163 ids.clear();
164 status = H5Fclose(file_id);
165 // Close eventually open camera
166 for (int i = 0; i < cameraList.size(); i++) {
167 Camera *camera = cameraList.at(i);
168 if (isopencamerawindow.at(i) == false) {
169 camera->stop();
170 }
171 }
172 for (int i = 0; i < sequenceList.size(); i++) {
173 AcquisitionSequence *sequence = sequenceList.at(i);
174 emit getAcquiring(sequence->record);
175 if (sequence->instrumentType == "DAC") {
176 QLOG_DEBUG() << "AcquisitionThread::execute> Update Db DAC values...";
177 // Update DB Dac values
178 if (dac != NULL) {
179 dac->updateDBValues(sequence->instrumentName);
180 }
181 }
182 if (sequence->instrumentType == "COMEDICOUNTER") {
183 QLOG_DEBUG() << "AcquisitionThread::execute> Update Db COMEDI COUNTER values...";
184 // Update DB Counter values
185 if (comedicounter != NULL) {
186 comedicounter->updateDBValues(sequence->instrumentName);
187 }
188 }
189 if (sequence->instrumentType == "COMEDIDAC") {
190 QLOG_DEBUG() << "AcquisitionThread::execute> Update Db COMEDI DAC values...";
191 // Update DB Comedi Dac values
192 if (comedidac != NULL) {
193 comedidac->updateDBValues(sequence->instrumentName);
194 }
195 }
196 if (sequence->instrumentType == "RASPIDAC") {
197 QLOG_DEBUG() << "AcquisitionThread::execute> Update Db RASPI DAC values...";
198 // Update DB Raspi Dac values
199 if (raspidac != NULL) {
200 raspidac->updateDBValues(sequence->instrumentName);
201 }
202 }
203 if (sequence->instrumentType == "MOTOR") {
204 QLOG_DEBUG() << "AcquisitionThread::execute> Update Db MOTOR values...";
205 // Emit last position
206 QString positionQString;
207 positionQString.setNum (sequence->position, 'f',3);
208 emit getPosition(positionQString);
209 }
210 if (sequence->instrumentType == "SUPERK") {
211 QLOG_DEBUG() << "AcquisitionThread::execute> Update Db SUPERK values...";
212 // Emit last data
213 QString superKData = QString::number(sequence->superkPowerValue) + " " +
214 QString::number(sequence->superkNdValue) + " " +
215 QString::number(sequence->superkSwpValue) + " " +
216 QString::number(sequence->superkLwpValue);
217 emit getSuperKData(superKData);
218 }
219 delete sequence;
220 }
221 sequenceList.clear();
222 }
223 suspend = true;
224 QLOG_DEBUG() << "AcquisitionThread::run> End of thread";
225}
226void AcquisitionThread::execute(AcquisitionSequence *sequence) {
227 // New acquisition
228 QLOG_INFO() << "AcquisitionThread::execute> instrument type " << sequence->instrumentType;
229 //emit getAcquiring(record);
230
231 QLOG_INFO() << "AcquisitionThread::execute> Execute record " << record
232 << " instrument " << sequence->instrumentType << ":"
233 << sequence->instrumentName << " remaining loops "
234 << sequence->remainingLoops << ":" << sequence->loopends << ":"
235 << sequence->loopends;
236
237 if (sequence->instrumentType == "MOTOR") {
238 motor->connectMotor(sequence->instrumentName);
239
240 float motorValue = sequence->motorValue;
241 if (sequence->motorAction == "MOVEREL" && isSnakeReverseForRecord(record)) {
242 motorValue = -motorValue;
243 }
244
245 if (sequence->motorAction == "MOVEREL") {
246 if (motorValue < 0) {
247 motor->moveBackward(sequence->instrumentName, (-motorValue));
248 } else {
249 motor->moveForward(sequence->instrumentName, motorValue);
250 }
251 }
252 else if (sequence->motorAction == "MOVEABS") {
253 motor->moveAbsolute(sequence->instrumentName, sequence->motorValue);
254 }
255 // Wait for motor movement to be completed
256 while (motor->getOperationComplete(sequence->instrumentName) <= 0) {
257 if (suspend) {
258 motor->stopMotor(sequence->instrumentName);
259 }
260 usleep(100);
261 motor->operationComplete();
262 }
263 // Update motor position
264 QString positionQString;
265 // Save motor position data
266 sequence->position = motor->getPosition(sequence->instrumentName);
267 positionQString.setNum (sequence->position, 'f',3);
268 emit getPosition(positionQString);
269 QLOG_INFO() << "AcquisitionThread::execute> " << sequence->instrumentName
270 << " new position:" << positionQString;
271 sequence->status = true;
272 }
273 else if (sequence->instrumentType == "SUPERK") {
274 superk->connectSuperK(sequence->instrumentName);
275 if (sequence->superkPowerValue > -88888) {
276 superk->setPower(sequence->instrumentName,
277 sequence->superkPowerValue * 10);
278 }
279 if (sequence->superkNdValue > -88888) {
280 superk->setNd(sequence->instrumentName, sequence->superkNdValue * 10);
281 }
282 if (sequence->superkSwpValue > -88888) {
283 superk->setSwp(sequence->instrumentName, sequence->superkSwpValue * 10);
284 }
285 if (sequence->superkLwpValue > -88888) {
286 superk->setLwp(sequence->instrumentName, sequence->superkLwpValue * 10);
287 }
288 if (sequence->superkBwValue > -88888 && sequence->superkCwValue > -88888) {
289 superk->setLwp(sequence->instrumentName,
290 (sequence->superkCwValue - sequence->superkBwValue / 2) *
291 10);
292 superk->setSwp(sequence->instrumentName,
293 (sequence->superkCwValue + sequence->superkBwValue / 2) *
294 10);
295 }
296 if (sequence->superkRPowerValue > -88888) {
297 sequence->superkPowerValue =
298 (float)superk->getPower(sequence->instrumentName) / 10 +
299 sequence->superkRPowerValue;
300 QLOG_INFO() << "sequence->superkPowerValue=" << sequence->superkPowerValue
301 << " " << superk->getPower(sequence->instrumentName) << " + "
302 << sequence->superkRPowerValue;
303 superk->setPower(sequence->instrumentName,
304 sequence->superkPowerValue * 10);
305 }
306 if (sequence->superkRNdValue > -88888) {
307 sequence->superkNdValue =
308 (float)superk->getNd(sequence->instrumentName) / 10 +
309 sequence->superkRNdValue;
310 superk->setNd(sequence->instrumentName, sequence->superkNdValue * 10);
311 }
312 if (sequence->superkRSwpValue > -88888) {
313 sequence->superkSwpValue =
314 (float)superk->getSwp(sequence->instrumentName) / 10 +
315 sequence->superkRSwpValue;
316 superk->setSwp(sequence->instrumentName, sequence->superkSwpValue * 10);
317 }
318 if (sequence->superkRLwpValue > -88888) {
319 sequence->superkLwpValue =
320 (float)superk->getLwp(sequence->instrumentName) / 10 +
321 sequence->superkRLwpValue;
322 superk->setLwp(sequence->instrumentName, sequence->superkLwpValue * 10);
323 }
324 if (sequence->superkRBwValue > -88888 && sequence->superkCwValue > -88888) {
325 QLOG_INFO() << "sequence->superkRBwValue " << sequence->superkRBwValue;
326 QLOG_INFO() << "sequence->superkCwValue " << sequence->superkCwValue;
327
328 sequence->superkBwValue =
329 ((float)superk->getSwp(sequence->instrumentName) -
330 (float)superk->getLwp(sequence->instrumentName)) /
331 10 +
332 sequence->superkRBwValue;
333 superk->setLwp(sequence->instrumentName,
334 (sequence->superkCwValue - sequence->superkBwValue / 2) *
335 10);
336 superk->setSwp(sequence->instrumentName,
337 (sequence->superkCwValue + sequence->superkBwValue / 2) *
338 10);
339 }
340 if (sequence->superkBwValue > -88888 && sequence->superkRCwValue > -88888) {
341 QLOG_INFO() << "sequence->superkBwValue " << sequence->superkBwValue;
342 QLOG_INFO() << "sequence->superkRCwValue " << sequence->superkRCwValue;
343
344 sequence->superkCwValue =
345 ((float)superk->getLwp(sequence->instrumentName) +
346 (float)superk->getSwp(sequence->instrumentName)) /
347 10 / 2 +
348 sequence->superkRCwValue;
349 superk->setLwp(sequence->instrumentName,
350 (sequence->superkCwValue - sequence->superkBwValue / 2) *
351 10);
352 superk->setSwp(sequence->instrumentName,
353 (sequence->superkCwValue + sequence->superkBwValue / 2) *
354 10);
355 }
356 if (sequence->superkRBwValue > -88888 && sequence->superkRCwValue > -88888) {
357 QLOG_INFO() << "sequence->superkRBwValue " << sequence->superkRBwValue;
358 QLOG_INFO() << "sequence->superkRCwValue " << sequence->superkRCwValue;
359 sequence->superkCwValue =
360 ((float)superk->getLwp(sequence->instrumentName) +
361 (float)superk->getSwp(sequence->instrumentName)) /
362 10 / 2 +
363 sequence->superkRCwValue;
364 sequence->superkBwValue =
365 ((float)superk->getSwp(sequence->instrumentName) -
366 (float)superk->getLwp(sequence->instrumentName)) /
367 10 +
368 sequence->superkRBwValue;
369 superk->setLwp(sequence->instrumentName,
370 (sequence->superkCwValue - sequence->superkBwValue / 2) *
371 10);
372 superk->setSwp(sequence->instrumentName,
373 (sequence->superkCwValue + sequence->superkBwValue / 2) *
374 10);
375 }
376
377 // Wait for superk to be completed
378 while (superk->getOperationComplete(sequence->instrumentName) <= 0) {
379 usleep(100);
380 superk->operationComplete();
381 }
382 sequence->status = true;
383 }
384 else if (sequence->instrumentType == "DAC") {
385 QLOG_INFO() << "AcquisitionThread::execute> connecting DAC ..." << sequence->instrumentName;
386 dacsuccess = dac->connectDac(sequence->instrumentName);
387 QLOG_INFO() << "AcquisitionThread::execute> DAC " << sequence->instrumentName
388 << " connected " << dacsuccess;
389
390 if (dacsuccess == true) {
391 if (sequence->dacRValue != 0) {
392 dacsuccess = dac->setDacRValue(sequence->instrumentName,
393 sequence->dacOutput,
394 sequence->dacRValue);
395 sequence->dacValue =
396 dac->getDacValue(sequence->instrumentName, sequence->dacOutput);
397 }
398 else {
399 dacsuccess = dac->setDacValue(sequence->instrumentName,
400 sequence->dacOutput,
401 sequence->dacValue);
402 }
403 }
404 sequence->status = dacsuccess;
405 //emit getDacStatus(dacsuccess);
406 }
407 else if (sequence->instrumentType == "COMEDICOUNTER") {
408 if (!comedicounter) {
409 Utils::EmitWarning(this, __FUNCTION__,
410 "COMEDICOUNTER is not configured");
411 sequence->status = false;

415 << " ComediCounter pointer " << comedicounter;

421 comedicountersuccess =
422 comedicounter->setComediValue(sequence->instrumentName,

451

636bool AcquisitionThread::isSnakeReverseForRecord(int cur_record) const {
637 int loopDepth = 0;
638 for (int i = cur_record; i >= 0; --i) {
639 AcquisitionSequence *tmpSequence = sequenceList.at(i);
640 if (tmpSequence->loop != AcquisitionSequence::IS_LOOP) {
641 continue;
642 }
643 if (tmpSequence->loopends == AcquisitionSequence::LOOP_END) {
644 loopDepth++;
645 continue;
646 }
647 if (tmpSequence->loopends == AcquisitionSequence::LOOP_START) {
648 if (loopDepth > 0) {
649 loopDepth--;
650 continue;
651 }
652 if (tmpSequence->loopSnake) {
653 return tmpSequence->loopReverse;
654 }
655 }
656 }
657 return false;
658}

660void AcquisitionThread::updateChildSnakeReverse(int parent_start,
661 int parent_end,
662 bool reset) {
663 int depth = 0;
664 for (int i = parent_start + 1; i < parent_end; ++i) {
665 AcquisitionSequence *tmpSequence = sequenceList.at(i);
666 if (tmpSequence->loop != AcquisitionSequence::IS_LOOP) {
667 continue;
668 }
669 if (tmpSequence->loopends == AcquisitionSequence::LOOP_START) {
670 depth++;
671 if (depth == 1 && tmpSequence->loopSnake) {
672 tmpSequence->loopReverse = reset ? false : !tmpSequence->loopReverse;
673 }
674 continue;
675 }
676 if (tmpSequence->loopends == AcquisitionSequence::LOOP_END) {
677 if (depth > 0) {
678 depth--;
679 }
680 continue;
681 }
682 }

684void AcquisitionThread::nextRecord(AcquisitionSequence *sequence, int cur_record) {
685 if (sequence->loop == AcquisitionSequence::IS_LOOP &&
686 sequence->loopends == AcquisitionSequence::LOOP_END) {
687 QLOG_DEBUG() << "AcquisitionThread::nextRecord> End of loop at record "
688 << cur_record << " loop " << sequence->loop << " loopNumber "
689 << sequence->loopNumber << " instrument "
690 << sequence->instrumentType << ":" << sequence->instrumentName;
691
692 // go back to starting loop record
693 int loopnumber = 0;
694 for (int i = cur_record - 1; i >= 0; i--) {
695 AcquisitionSequence *tmpSequence = sequenceList.at(i);
696 QLOG_DEBUG() << "AcquisitionThread::nextRecord> Search start of loop at record "
697 << cur_record << " loop " << tmpSequence->loop << " loopends "
698 << tmpSequence->loopends << " loopNumber "
699 << tmpSequence->loopNumber << " instrument "
700 << tmpSequence->instrumentType << ":"
701 << tmpSequence->instrumentName;
702
703 if (tmpSequence->loop == AcquisitionSequence::IS_LOOP &&
704 tmpSequence->loopends == AcquisitionSequence::LOOP_END) {
705 loopnumber++;
706 continue;
707 }
708 if (tmpSequence->loop == AcquisitionSequence::IS_LOOP &&
709 tmpSequence->loopends == AcquisitionSequence::LOOP_START) {
710 if (loopnumber > 0) {
711 loopnumber--;
712 continue;
713 }
714 else if (loopnumber == 0 && tmpSequence->loopNumber > 0) {
715 QLOG_DEBUG() << "AcquisitionThread::nextRecord> Start of loop at record "
716 << tmpSequence->record << " loop " << tmpSequence->loop
717 << " loopNumber " << tmpSequence->loopNumber
718 << " instrument " << tmpSequence->instrumentType << ":"
719 << tmpSequence->instrumentName;
720
721 if (tmpSequence->remainingLoops > 1) {
722 record = tmpSequence->record;
723 tmpSequence->remainingLoops--;
724 updateChildSnakeReverse(i, cur_record, false);
725 // Set new group name
726 if (sequence->inc_group >= 10) {
727 sequence->group =
728 sequence->datagroup + "_" + QString::number(sequence->inc_group);
729 }
730 else {
731 sequence->group =
732 sequence->datagroup + "_0" + QString::number(sequence->inc_group);
733 }
734 sequence->inc_group++;
735 return;
736 }
737 else {
738 // reset remainScans and go to next record
739 QLOG_DEBUG() << "AcquisitionThread::nextRecord> RemainingLoops = 1";
740 tmpSequence->remainingLoops = tmpSequence->loopNumber;
741 updateChildSnakeReverse(i, cur_record, true);
742 // reset inc_group number for the entire loop sequence
743 for (int j = cur_record; j >= i; j--) {
744 AcquisitionSequence *tmpSequencebis = sequenceList.at(j);
745 tmpSequencebis->inc_group = 0;
746 }
747 record++;
748 return;
749 }
750 }
751 }
752 }
753 }
754 else {
755 // go to next record
756 record++;
757 // Set new group name
758 if (sequence->inc_group >= 10) {
759 sequence->group =
760 sequence->datagroup + "_" + QString::number(sequence->inc_group);
761 }
762 else {
763 sequence->group =
764 sequence->datagroup + "_0" + QString::number(sequence->inc_group);
765 }
766 sequence->inc_group++;
767 }
