67void
68MotorControlWidget::setMotor(Motor *_motor) {
69 motor = _motor;
70 connect(motor,SIGNAL(getPosition(float)),this,SLOT(getPosition(float)));
71 connect(motor,SIGNAL(getDescription(QString)),this,SLOT(getDescription(QString)));
72 connect(motor,SIGNAL(stopTimer()),this,SLOT(stopTimer()));
73 //
74 // Launch a timer to get the current position of the actuator every second
75 //
76 timer = new QTimer(this);
77 connect(timer, SIGNAL(timeout()), motor, SLOT(operationComplete()));
78}
