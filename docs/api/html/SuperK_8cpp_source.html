86 void
87 SuperK::connectSuperK(QString newdriver)
88 {
89  QSqlDatabase db = connectDb(path);
90  QSqlQuery query(db);
91  if (newdriver == "") return;
92  //
93  // Get driver data from Db
94  //
95  QString comtype, comaddress,comsettings;
96  QString drvsettings, drvdescription;
97  QString drvdata;
98  int drvpower,drvnd,drvswp,drvlwp;
99  query.prepare("select comtype,comaddress,comsettings,drvsettings,description,data from superk_driver where name = ?");
100  query.addBindValue(newdriver);
101  query.exec();
102  while (query.next()) {
103  comtype = query.value(0).toString();
104  comaddress = query.value(1).toString();
105  comsettings = query.value(2).toString();
106  drvsettings = query.value(3).toString();
107  drvdescription = query.value(4).toString();
108  drvdata = query.value(5).toString();
109 #if QT_VERSION >= QT_VERSION_CHECK(6, 0, 0)
110  QStringList dataSplit = drvdata.split(" ", Qt::SkipEmptyParts);
111 #else
112  QStringList dataSplit = drvdata.split(" ", QString::SkipEmptyParts);
113 #endif
114  if (dataSplit.size() == 4) {
115  drvpower = dataSplit.at(0).toInt();
116  drvnd = dataSplit.at(1).toInt();
117  drvswp = dataSplit.at(2).toInt();
118  drvlwp = dataSplit.at(3).toInt();
119  }
120  }
121  //
122  // Check if driver already configured
123  //
124  for (int i = 0 ; i < driver.size(); i++) {
125  if ( driver.at(i) == newdriver ) {
126  QLOG_DEBUG () << newdriver << " already connected";
127  if ( connectSuccess.at(i) == true) {
128  emit getPower(drvpower);
129  emit getNd(drvnd);
130  emit getSwp(drvswp);
131  emit getLwp(drvlwp);
132  emit getDescription(drvdescription);
133  return;
134  }
135  else {
136  QLOG_DEBUG () << newdriver << " try to reconnect...";
137  //
138  // Free Driver related structures
139  //
140  driverSuperK.remove(i);
141  driverSettings.remove(i);
142  description.remove(i);
143  driver.remove(i);
144  connectSuccess.remove(i);
145  operationcomplete.remove(i);
146  data.remove(i);
147  break;
148  }
149  }
150  }
151  //
152  // New driver initialization
153  //
154  //
155  // Create new com
156  //
157  QLOG_INFO() << " Create new Communication channel at " << comaddress;
158 
159  ACCom *accom = ACCom::Create(comtype.toStdString(),
160  comaddress.toStdString(),
161  comsettings.toStdString());
162  if (!accom) {
163  Utils::EmitWarning(this, __FUNCTION__, "Could not create Communication channel");
164  return;
165  }
166  else if (accom->Open()) {
167  Utils::EmitWarning(this, __FUNCTION__, "Could not open communication with device");
168  delete accom;
169  return;
170  }
171  //
172  // Create new driver
173  //
174  QLOG_INFO() << " Create new Driver with settings " << drvsettings;
175  DriverSuperK *drv = new DriverSuperK(accom);
176  if (!drv) {
177  Utils::EmitWarning(this, __FUNCTION__, "Could not create Driver");
178  return;
179  }
180  else {
181  if (drv->Init()) {
182  Utils::EmitWarning(this, __FUNCTION__, "Could not init driver");
183  delete drv;
184  return;
185  }
186  drv->getPower(drvpower);
187  drv->getNd(drvnd);
188  drv->getSwp(drvswp);
189  drv->getLwp(drvlwp);
190  drvdata = QString::number(drvpower) + " " + QString::number(drvnd) + " " +
191  QString::number(drvswp) + " " + QString::number(drvlwp);
192  driverCom.push_back(accom);
193  driverSuperK.push_back(drv);
194  driver.push_back(newdriver);
195  connectSuccess.push_back(true);
196  driverSettings.push_back(drvsettings);
197  data.push_back(drvdata);
198  description.push_back(drvdescription);
199  operationcomplete.push_back(0);
200 
201  // Save new data in Db
202  QSqlQuery query(db);
203  query.prepare("update superk_driver set data = ? "
204  "where name = ?");
205  query.addBindValue(drvdata);
206  query.addBindValue(newdriver);
207  query.exec();
208 
209  emit getPower(drvpower);
210  emit getNd(drvnd);
211  emit getSwp(drvswp);
212  emit getLwp(drvlwp);
213 
214  emit getDescription(drvdescription);
215  return;
216  }

333 void
334 SuperK::operationComplete()
335 {
336  QLOG_DEBUG() << "SuperK::operationComplete runningDriver:" << runningDriver;
337  for (int i = 0 ; i < driver.size(); i++) {
338  if (driver.at(i) == runningDriver ) {
339  iscompleting = false;
340  if (connectSuccess.at(i) == true && iscompleting == false) {
341  int success = 0;
342  int power,nd,swp,lwp;
343  iscompleting = true;
344  success = driverSuperK.at(i)->operationComplete();
345  QLOG_DEBUG() << "SuperK::operationComplete success:" << success;
346  driverSuperK.at(i)->getPower(power);
347  driverSuperK.at(i)->getNd(nd);
348  driverSuperK.at(i)->getSwp(swp);
349  driverSuperK.at(i)->getLwp(lwp);
350  QString dataStr = QString::number(power) + " " + QString::number(nd) + " " +
351  QString::number(swp) + " " + QString::number(lwp);
352  data.replace(i, dataStr);
353  emit getPower(power);
354  emit getNd(nd);
355  emit getSwp(swp);
356  emit getLwp(lwp);
357  if (success > 0 ) {
358  // Save last data in Db
359  QSqlDatabase db = connectDb(path);
360  QSqlQuery query(db);
361 
362  query.prepare("update superk_driver set data = ? "
363  "where name = ?");
364  query.addBindValue(dataStr);
365  query.addBindValue(driver.at(i));
366  query.exec();
367  operationcomplete.replace(i, success);
368  emit stopTimer();
369  }
370  iscompleting = false;
371  }
372  break;
373  }
374  }
