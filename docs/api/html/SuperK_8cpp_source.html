81void
82SuperK::connectSuperK(QString newdriver)
83{
84 QSqlDatabase db = connectDb(path);
85 QSqlQuery query(db);
86 if (newdriver == "") return;
87 //
88 // Get driver data from Db
89 //
90 QString comtype, comaddress,comsettings;
91 QString drvsettings, drvdescription;
92 QString drvdata;
93 int drvpower,drvnd,drvswp,drvlwp;
94 query.prepare("select comtype,comaddress,comsettings,drvsettings,description,data from superk_driver where name = ?");
95 query.addBindValue(newdriver);
96 query.exec();
97 while (query.next()) {
98 comtype = query.value(0).toString();
99 comaddress = query.value(1).toString();
100 comsettings = query.value(2).toString();
101 drvsettings = query.value(3).toString();
102 drvdescription = query.value(4).toString();
103 drvdata = query.value(5).toString();
104#if QT_VERSION >= QT_VERSION_CHECK(6, 0, 0)
105 QStringList dataSplit = drvdata.split(" ", Qt::SkipEmptyParts);
106#else
107 QStringList dataSplit = drvdata.split(" ", QString::SkipEmptyParts);
108#endif
109 if (dataSplit.size() == 4) {
110 drvpower = dataSplit.at(0).toInt();
111 drvnd = dataSplit.at(1).toInt();
112 drvswp = dataSplit.at(2).toInt();
113 drvlwp = dataSplit.at(3).toInt();
114 }
115 }
116 //
117 // Check if driver already configured
118 //
119 for (int i = 0 ; i < driver.size(); i++) {
120 if ( driver.at(i) == newdriver ) {
121 QLOG_DEBUG () << newdriver << " already connected";
122 if ( connectSuccess.at(i) == true) {
123 emit getPower(drvpower);
124 emit getNd(drvnd);
125 emit getSwp(drvswp);
126 emit getLwp(drvlwp);
127 emit getDescription(drvdescription);
128 return;
129 }
130 else {
131 QLOG_DEBUG () << newdriver << " try to reconnect...";
132 //
133 // Free Driver related structures
134 //
135 driverSuperK.remove(i);
136 driverSettings.remove(i);
137 description.remove(i);
138 driver.remove(i);
139 connectSuccess.remove(i);
140 operationcomplete.remove(i);
141 data.remove(i);
142 break;
143 }
144 }
145 }
146 //
147 // New driver initialization
148 //
149 //
150 // Create new com
151 //
152 QLOG_INFO() << " Create new Communication channel at " << comaddress;
153
154 ACCom *accom = ACCom::Create(comtype.toStdString(),
155 comaddress.toStdString(),
156 comsettings.toStdString());
157 if (!accom) {
158 Utils::EmitWarning(this, __FUNCTION__, "Could not create Communication channel");
159 return;
160 }
161 else if (accom->Open()) {
162 Utils::EmitWarning(this, __FUNCTION__, "Could not open communication with device");
163 delete accom;
164 return;
165 }
166 //
167 // Create new driver
168 //
169 QLOG_INFO() << " Create new Driver with settings " << drvsettings;
170 DriverSuperK *drv = new DriverSuperK(accom);
171 if (!drv) {
172 Utils::EmitWarning(this, __FUNCTION__, "Could not create Driver");
173 return;
174 }
175 else {
176 if (drv->Init()) {
177 Utils::EmitWarning(this, __FUNCTION__, "Could not init driver");
178 delete drv;
179 return;
180 }
181 drv->getPower(drvpower);
182 drv->getNd(drvnd);
183 drv->getSwp(drvswp);
184 drv->getLwp(drvlwp);
185 drvdata = QString::number(drvpower) + " " + QString::number(drvnd) + " " +
186 QString::number(drvswp) + " " + QString::number(drvlwp);
187 driverCom.push_back(accom);
188 driverSuperK.push_back(drv);
189 driver.push_back(newdriver);
190 connectSuccess.push_back(true);
191 driverSettings.push_back(drvsettings);
192 data.push_back(drvdata);
193 description.push_back(drvdescription);
194 operationcomplete.push_back(0);
195
196 // Save new data in Db
197 QSqlQuery query(db);
198 query.prepare("update superk_driver set data = ? "
199 "where name = ?");
200 query.addBindValue(drvdata);
201 query.addBindValue(newdriver);
202 query.exec();
203
204 emit getPower(drvpower);
205 emit getNd(drvnd);
206 emit getSwp(drvswp);
207 emit getLwp(drvlwp);
208
209 emit getDescription(drvdescription);
210 return;
211 }

328void
329SuperK::operationComplete()
330{
331 QLOG_DEBUG() << "SuperK::operationComplete runningDriver:" << runningDriver;
332 for (int i = 0 ; i < driver.size(); i++) {
333 if (driver.at(i) == runningDriver ) {
334 iscompleting = false;
335 if (connectSuccess.at(i) == true && iscompleting == false) {
336 int success = 0;
337 int power,nd,swp,lwp;
338 iscompleting = true;
339 success = driverSuperK.at(i)->operationComplete();
340 QLOG_DEBUG() << "SuperK::operationComplete success:" << success;
341 driverSuperK.at(i)->getPower(power);
342 driverSuperK.at(i)->getNd(nd);
343 driverSuperK.at(i)->getSwp(swp);
344 driverSuperK.at(i)->getLwp(lwp);
345 QString dataStr = QString::number(power) + " " + QString::number(nd) + " " +
346 QString::number(swp) + " " + QString::number(lwp);
347 data.replace(i, dataStr);
348 emit getPower(power);
349 emit getNd(nd);
350 emit getSwp(swp);
351 emit getLwp(lwp);
352 if (success > 0 ) {
353 // Save last data in Db
354 QSqlDatabase db = connectDb(path);
355 QSqlQuery query(db);
356
357 query.prepare("update superk_driver set data = ? "
358 "where name = ?");
359 query.addBindValue(dataStr);
360 query.addBindValue(driver.at(i));
361 query.exec();
362 operationcomplete.replace(i, success);
363 emit stopTimer();
364 }
365 iscompleting = false;
366 }
367 break;
368 }
369 }
