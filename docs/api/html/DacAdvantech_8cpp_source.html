144 DacAdvantech::connectDac(QString newdac) {
145 
146  int index;
147  if (newdac == "") return false;
148 
149  char err_msg[100];
150  QSqlDatabase db = connectDb(path);
151  QSqlQuery query(db);
152  int status;
153  QString description;
154  QString dacvaluesString = "";
155  QString dacSettings = "";
156  //
157  // Configure DacAdvantech
158  //
159  query.prepare("select settings,description,dacvalues from dac_settings where name = ?");
160  query.addBindValue(newdac);
161  query.exec();
162 
163  while (query.next()) {
164  dacSettings = query.value(0).toString();
165  description = query.value(1).toString();
166  dacvaluesString = query.value(2).toString();
167  }
168  if (dacSettings == "") {
169  Utils::EmitWarning(this, __FUNCTION__, tr("%1 does not exist").arg(newdac));
170  return false;
171  }
172 
173  //
174  // Check if dac already configured
175  //
176  for (int i = 0 ; i < dac.size(); i++) {
177  if ( dac.at(i) == newdac ) {
178  QLOG_DEBUG () << newdac << " already connected";
179  if ( connectSuccess.at(i) == true )
180  return true;
181  else {
182  QLOG_DEBUG () << newdac << " try to reconnect...";
183  //
184  // Free ADVANTECHDAC related structures
185  //
186  if (fd.at(i)) {
187  free(fd.at(i));
188  fd.remove(i);
189  }
190  if (config.at(i)) {
191  free(config.at(i));
192  config.remove(i);
193  }
194  if (chOutV.at(i)) {
195  free(chOutV.at(i));
196  chOutV.remove(i);
197  }
198  if (chOutC.at(i)) {
199  free(chOutC.at(i));
200  chOutC.remove(i);
201  }
202  if (ao_chan_range.at(i)) {
203  free(ao_chan_range.at(i));
204  ao_chan_range.remove(i);
205  }
206  if (dacvalues.at(i)) delete dacvalues.at(i);
207  dacvalues.remove(i);
208  dacSettings.remove(i);
209  dac.remove(i);
210  mode.remove(i);
211  min.remove(i);
212  max.remove(i);
213  outputs.remove(i);
214  fname.remove(i);
215  connectSuccess.remove(i);
216  break;
217  }
218  }
219  }
220  // First time DacAdvantech connection
221  dac.push_back(newdac);
222  // Assign index
223  index = dac.size() - 1;
224  QStringList settings;
225  QStringList subsettings;
226  settings = dacSettings.split(" ");
227  subsettings = settings.at(0).split("=");
228  fname.push_back(subsettings.at(1));
229  subsettings = settings.at(1).split("=");
230  mode.push_back(subsettings.at(1));
231  subsettings = settings.at(2).split("=");
232  outputs.push_back(subsettings.at(1).toInt());
233  subsettings = settings.at(3).split("=");
234  max.push_back(subsettings.at(1).toFloat());
235  subsettings = settings.at(4).split("=");
236  min.push_back(subsettings.at(1).toFloat());
237  connectSuccess.push_back(false);
238  dacvalues.push_back(new QVector<float>);
239  PTR_T* fdPtr = (PTR_T*)malloc(sizeof(PTR_T));
240  fd.push_back(fdPtr);
241  PT_AOConfig* configPtr = (PT_AOConfig*)malloc(sizeof(PT_AOConfig));
242  config.push_back(configPtr);
243  PT_AOVoltageOut* chOutVPtr = (PT_AOVoltageOut*)malloc(sizeof(PT_AOVoltageOut));
244  chOutV.push_back(chOutVPtr);
245  PT_AOCurrentOut* chOutCPtr = (PT_AOCurrentOut*)malloc(sizeof(PT_AOCurrentOut));
246  chOutC.push_back(chOutCPtr);
247  AORANGESET* ao_chan_rangePtr = (AORANGESET*)malloc(sizeof(AORANGESET));
248  ao_chan_range.push_back(ao_chan_rangePtr);
249 
250  QLOG_DEBUG ( ) << "fname " << fname.at(index);
251  QLOG_DEBUG ( ) << "mode = " << mode.at(index);
252  QLOG_DEBUG ( ) << "Outputs = " << outputs.at(index);
253  QLOG_DEBUG ( ) << "Max = " << max.at(index);
254  QLOG_DEBUG ( ) << "Min = " << min.at(index);
255 
256  // Open device
257  QLOG_INFO() << " Open device " << fname.at(index);
258  status = DRV_DeviceOpen((char*)fname.at(index).toStdString().c_str(), fd.at(index));
259  if (status) {
260  DRV_GetErrorMessage(status, err_msg);
261  Utils::EmitWarning(this, __FUNCTION__, tr("DRV_DeviceOpen:%1").arg(err_msg));
262  connectSuccess.replace(index, false);
263  return false;
264  }
265  if (mode.at(index) == "VOLT") {
266  for (int i = 0; i < outputs.at(index); i++) {
267  QLOG_INFO ( ) << "Configuring output " << i
268  << " with min=" << min.at(index)
269  << " with max=" << max.at(index);
270  memset(config.at(index), 0, sizeof(PT_AOConfig));
271  config.at(index)->chan = i;
272  config.at(index)->RefSrc = 0;
273  config.at(index)->MaxValue = max.at(index);
274  config.at(index)->MinValue = min.at(index);
275  status = DRV_AOConfig(*(fd.at(index)), config.at(index));
276  if (status) {
277  DRV_GetErrorMessage(status, err_msg);
278  Utils::EmitWarning(this, __FUNCTION__, tr("DRV_AOConfig:%1").arg(err_msg));
279  connectSuccess.replace(index, false);
280  return false;
281  }
282  }
283  }
284  int cur_volt;
285  if (mode.at(index) == "CURRENT")
286  cur_volt = 1;
287  else
288  cur_volt = 0;
289  AORANGESET aorange;
290  for (int i = 0; i < outputs.at(index); i++) {
291  memset(ao_chan_range.at(index), 0, sizeof(aorange));
292  ao_chan_range.at(index)->usGainCount = 6;
293  ao_chan_range.at(index)->usAOSource = 0; /* =0 internal, =1 external */
294  ao_chan_range.at(index)->usAOType = cur_volt; /* =0 voltage, =1 current */
295  ao_chan_range.at(index)->usChan = i;
296  ao_chan_range.at(index)->fAOMax = max.at(index);
297  ao_chan_range.at(index)->fAOMin = min.at(index);
298  status = DRV_DeviceSetProperty(*(fd.at(index)),
299  CFG_AoChanRange,ao_chan_range.at(index),
300  sizeof(aorange));
301  if (status) {
302  DRV_GetErrorMessage(status, err_msg);
303  Utils::EmitWarning(this, __FUNCTION__, tr("DRV_DeviceSetProperty:%1").arg(err_msg));
304  connectSuccess.replace(index, false);
305  return false;
306  }
307  INT32U size;
308  status = DRV_DeviceGetProperty(*(fd.at(index)),
309  CFG_AoChanRange,
310  ao_chan_range.at(index),&size);
311  if (status) {
312  DRV_GetErrorMessage(status, err_msg);
313  Utils::EmitWarning(this, __FUNCTION__, tr("DRV_DeviceGetProperty:%1").arg(err_msg));
314  connectSuccess.replace(index, false);
315  return false;
316  }
317  QLOG_INFO ( ) << dac.at(index) << " Channel " << i
318  << " GainCount: " << ao_chan_range.at(index)->usGainCount;
319  QLOG_INFO ( ) << "AOSource: " << ao_chan_range.at(index)->usAOSource;
320  QLOG_INFO ( ) << "AOType: " << ao_chan_range.at(index)->usAOType;
321  QLOG_INFO ( ) << "Chan: " << ao_chan_range.at(index)->usChan;
322  QLOG_INFO ( ) << "AOMax: " << ao_chan_range.at(index)->fAOMax;
323  QLOG_INFO ( ) << "AOMin: " << ao_chan_range.at(index)->fAOMin;
324  QLOG_INFO ( ) << "Configuring " << dac.at(index) << " Channel " << i << "...Done";
325  }
326  QLOG_DEBUG ( ) << "DacAdvantech configuration success";
327  connectSuccess.replace(index, true);
328 
329  // Update control widget
330  emit getOutputs(outputs.at(index),mode.at(index));
331  emit getDescription(description);
332  // Get registered values from Db
333  QStringList dacvaluesStringList;
334  dacvaluesStringList = dacvaluesString.split(" ",QString::SkipEmptyParts);
335  for (int i = 0 ; i < dacvaluesStringList.size(); i++ )
336  dacvalues.at(index)->push_back(dacvaluesStringList.at(i).toFloat());
337 
338  emit getOutputValues(dacvalues.at(index));
339  return connectSuccess.at(index);
340 }
341 bool
342 DacAdvantech::resetDac(QString newdac) {
343  QSqlDatabase db = connectDb(path);
344  QSqlQuery query(db);
345  int status;
346  char err_msg[100];
347  for (int index = 0 ; index < dac.size(); index++) {
348  if ( dac.at(index) == newdac && connectSuccess.at(index) == true) {
349  /* Init device channels to 0 */
350  /* enable sync*/
351  status = DRV_EnableSyncAO(*(fd.at(index)), 1);
352  if (status) {
353  DRV_GetErrorMessage(status, err_msg);
354  Utils::EmitWarning(this, __FUNCTION__, tr("DRV_EnableSyncAO:%1").arg(err_msg));
355  return false;
356  }
357  dacvalues.at(index)->clear();
358  float value = 0;
359  for (int i = 0; i < outputs.at(index); i++) {
360  if (mode.at(index) == "VOLT") {
361  memset(chOutV.at(index), 0, sizeof(PT_AOVoltageOut));
362  chOutV.at(index)->chan = i;
363  chOutV.at(index)->OutputValue = value;
364  }
365  else if (mode.at(index) == "CURRENT") {
366  memset(chOutC.at(index), 0, sizeof(PT_AOCurrentOut));
367  chOutC.at(index)->chan = i;
368  chOutC.at(index)->OutputValue = value;
369 
370  }
371  if (mode.at(index) == "VOLT")
372  status = DRV_AOVoltageOut(*(fd.at(index)), chOutV.at(index));
373  else if (mode.at(index) == "CURRENT")
374  status = DRV_AOCurrentOut(*(fd.at(index)), chOutC.at(index));
375 
376  if (status) {
377  DRV_GetErrorMessage(status, err_msg);
378  Utils::EmitWarning(this, __FUNCTION__, tr("DRV_AO/Current/Voltage/Out%1").arg(err_msg));
379  return false;
380  }
381  dacvalues.at(index)->push_back(value);
382  }
383  /* write sync*/
384  status = DRV_WriteSyncAO(*(fd.at(index)));
385  if (status) {
386  DRV_GetErrorMessage(status, err_msg);
387  Utils::EmitWarning(this, __FUNCTION__, tr("DRV_WriteSyncAO:%1").arg(err_msg));
388  return false;
389  }
390  /* disable sync*/
391  status = DRV_EnableSyncAO(*(fd.at(index)), 0);
392  if (status) {
393  DRV_GetErrorMessage(status, err_msg);
394  Utils::EmitWarning(this, __FUNCTION__, tr("DRV_EnableSyncAO:%1").arg(err_msg));
395  return false;
396  }
397  // Set new values in Db
398  QString dacvaluesString = "";
399  QString valueString;
400  for (int i = 0 ; i < outputs.at(index); i++) {
401  valueString.setNum (dacvalues.at(index)->at(i), 'f',3);
402  dacvaluesString = dacvaluesString + valueString + " ";
403  }
404  query.prepare("update dac_settings set dacvalues = ? where name = ?");
405  query.addBindValue(dacvaluesString);
406  query.addBindValue(dac.at(index));
407  query.exec();
408  // Update control widget
409  emit getOutputValues(dacvalues.at(index));
410  return true;
411  }
412  }
413  Utils::EmitWarning(this, __FUNCTION__, tr("Check connection to %1").arg(newdac));
414  return false;

417 DacAdvantech::setDacRValue(QString newdac, int output, double rvalue) {

459  else

462  /* write sync*/
