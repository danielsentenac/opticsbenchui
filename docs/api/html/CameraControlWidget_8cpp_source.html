47static bool IsAutoEnumFeature(const Camera *camera, int index) {
48 if (camera == nullptr) {
49 return false;
50 }
51 const QString name = camera->featureNameList.at(index);
52 return name == "ExposureAuto" || name == "GainAuto";
53}

55static QString AutoEnumToText(int value) {
56 switch (value) {
57 case 1:
58 return "Once";
59 case 2:
60 return "Continuous";
61 default:
62 return "Off";
63 }
64}

66static int AutoEnumFromText(const QString &text, bool *ok) {
67 const QString trimmed = text.trimmed();
68 const QString lower = trimmed.toLower();
69 if (lower == "off") {
70 if (ok) {
71 *ok = true;
72 }
73 return 0;
74 }
75 if (lower == "once") {
76 if (ok) {
77 *ok = true;
78 }
79 return 1;
80 }
81 if (lower == "continuous" || lower == "on") {
82 if (ok) {
83 *ok = true;
84 }
85 return 2;
86 }
87 bool localOk = false;
88 const int numeric = trimmed.toInt(&localOk);
89 if (ok) {
90 *ok = localOk;
91 }
92 return numeric;
93}

250void
251CameraControlWidget::snapShot() {
252#ifdef NO_HDF5
253 Utils::EmitWarning(this, __FUNCTION__,
254 "HDF5 support is disabled. Snapshot in HDF5 format is unavailable.");
255 return;
256#else
257
258 uchar *img8 = nullptr;
259 int *img32 = nullptr;
260
261 // Take snapshot
262 if ( camera->pixel_encoding == B8 ) {
263 QLOG_INFO() << " Take SNAPSHOT...";
264 img8 = camera->getSnapshot();
265 }
266 else
267 img32 = camera->getSnapshot32();
268 // Save File
269 QString filename = QFileDialog::getSaveFileName(
270 this, tr("Take snapshot in HDF5 format"),
271 Utils::DefaultHdf5Path("untitled"),
272 Utils::Hdf5FileDialogFilter());
273 if (filename == "") return;
274 filename = Utils::EnsureHdf5Extension(filename);
275 hid_t file_id = H5Fcreate(filename.toStdString().c_str(),
276 H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT);
277 if ( file_id < 0 ) {
278 QLOG_WARN () << "Unable to open file - " << filename ;
279 Utils::EmitWarning(this, __FUNCTION__, "Unable to open file - " + filename );
280 return;
281 }
282 // Save data to File
283 if ( camera->pixel_encoding == B8 ) {
284 QLOG_INFO () << "Saving image to file - " << filename ;
285 QLOG_INFO () << "width - " << camera->width ;
286 QLOG_INFO () << "height - " << camera->height ;
287 QLOG_INFO () << "min - " << camera->snapShotMin ;
288 QLOG_INFO () << "max - " << camera->snapShotMax ;
289
290 H5IMmake_image_8bit(file_id,"SNAPSHOT",
291 camera->width,
292 camera->height,
293 img8);
294
295 H5LTset_attribute_int(file_id, "SNAPSHOT", "min", &camera->snapShotMin,1);
296 H5LTset_attribute_int(file_id, "SNAPSHOT", "max", &camera->snapShotMax,1);
297 }
298 else {
299 QLOG_INFO() << "CameraControlWidget::snapShot> Take snapshot pixel encoding "
300 << camera->pixel_encoding
301 << " width " << camera->width
302 << " height " << camera->height;
303 hsize_t dset_dims[2];
304 dset_dims[0] = camera->height;
305 dset_dims[1] = camera->width;
306 H5LTmake_dataset_int(file_id,"SNAPSHOT32", 2,dset_dims,img32);
307 H5LTset_attribute_int(file_id, "SNAPSHOT32", "min", &camera->snapShotMin,1);
308 H5LTset_attribute_int(file_id, "SNAPSHOT32", "max", &camera->snapShotMax,1);
309 }
310 // Close file
311 H5Fclose(file_id);
312#endif
313}

324 QSlider *featureSlider = featureSliderList.at(position);
325 QLOG_INFO() << "CameraControlWidget::setFeatureValue> Request "

353 featureSlider->setValue(value * SLIDER_FACTOR);
354 else {
355 if (IsAutoEnumFeature(camera, position)) {

358 if (ok) {

361 } else {
362 featureSlider->setValue(value);
