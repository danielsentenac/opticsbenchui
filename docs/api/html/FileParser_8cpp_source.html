67 QString list = "";
68 list = _type.mid(_type.indexOf("[") + 1, _type.indexOf("]") - _type.indexOf("[") - 1);
69 QLOG_DEBUG() << "FileParser::setType> list : " << list;
70 if (!list.isNull()) {

73 setNameList(list);
74 type = _type.left(_type.indexOf("["));
75 QLOG_DEBUG() << "FileParser::setType> type : " << type;

78 type = _type;

168bool
169FileParser::getFileData()
170{
171 bool success = false;
172 // Create QDir object to navigate into directory path
173 QDir directory(dir);
174 QDir subdirectory(dir + "/Results");
175 QStringList fileList = directory.entryList(QDir::Files, QDir::Time | QDir::Reversed);
176 QStringList subfileList = subdirectory.entryList(QDir::Files, QDir::Time | QDir::Reversed);
177 // Check directory and Results subdirectory content deletion
178 if ( del == "BEFORE" ) {
179 int cnt = 0;
180 foreach (const QString &filename, fileList) {
181 if ( cnt == fileList.count() )
182 break;
183 QLOG_DEBUG() << "FileParser::getFileData> deleting file" << filename;
184 directory.remove( filename );
185 cnt++;
186 }
187 cnt = 0;
188 foreach (const QString &filename, subfileList) {
189 if ( cnt == subfileList.count() )
190 break;
191 QLOG_DEBUG() << "FileParser::getFileData> deleting file" << filename;
192 subdirectory.remove( filename );
193 cnt++;
194 }
195 }
196
197 // Apply a delay before new dir scan
198 usleep( (int) (delay.toFloat() * 1000000) );
199
200 // Look for the requested file from new dir scan
201 QDir freshdir(dir + "/Results");
202 QFileInfoList freshFileInfoList = freshdir.entryInfoList(QDir::NoFilter, QDir::Time | QDir::Reversed);
203 QFile file ("");
204 uint filetime = 0;
205 parsedFileName = "";
206 foreach (const QFileInfo &fileinfo, freshFileInfoList) {
207 QString filename = fileinfo.fileName();
208 QLOG_DEBUG() << "FileParser::getFileData> treating file" << filename;
209 QDateTime date (fileinfo.lastModified());
210 uint filetimeTmp = static_cast<uint>(date.toSecsSinceEpoch());
211 if ( filename.left(prefix.length()) == prefix &&
212 filename.right(suffix.length()) == suffix ) {
213 if ( order == "FIRST" ) {
214 file.setFileName(freshdir.filePath(filename));
215 break;
216 }
217 else if ( order == "LAST" && filetime < filetimeTmp ) {
218 file.setFileName(freshdir.filePath(filename));
219 filetime = filetimeTmp;
220 }
221 }
222 }
223 // parse the file data
224 if ( file.fileName() == "" ) {
225 QLOG_WARN () << "FileParser::getFileData> No available file "
226 << " with prefix " << prefix
227 << " suffix " << suffix
228 << " in directory " << dir + "/Results";
229 return false;
230 }
231 else {
232 QLOG_INFO () << "FileParser::getFileData> parsing file " << file.fileName();
233 parsedFileName = file.fileName();
234 // Open File
235 if ( format == "ASCII" )
236 file.open(QIODevice::ReadOnly | QIODevice::Text);
237 else
238 file.open(QIODevice::ReadOnly);
239 QTextStream in(&file);
240 if ( type == "ARRAY" ) {
241 // Get array size in File
242 arrayHeight = 1;
243 arrayWidth = 0;
244 QString line = "";
245 QStringList dataItems;
246 QLOG_DEBUG() << "FileParser::getFileData> reading..." << file.fileName();
247 QLOG_INFO() << "FileParser::getFileData> separator [" << sep << "]";
248 line = in.readLine();
249 while (!in.atEnd()) {
250 QLOG_DEBUG() << "FileParser::getFileData> reading..." << line ;
251 dataItems = line.split(sep);
252 arrayHeight++;
253 arrayWidth = dataItems.count();
254 line = in.readLine();
255 }
256 QLOG_INFO() << "FileParser::getFileData> arrayHeight = " << arrayHeight
257 << " arrayWidth = " << arrayWidth;
258 if ( array ) { free(array); array = NULL;}
259 array = (double*) malloc(sizeof(double) * arrayHeight * arrayWidth);
260 // Parse the file to array
261 in.seek(0);
262 line = "";
263 int cnt = 0;
264 line = in.readLine();
265 while (!in.atEnd()) {
266 QLOG_DEBUG() << "FileParser::getFileData> grabbing..." << line ;
267 dataItems = line.split(sep);
268 foreach (const QString &data, dataItems) {
269 QLOG_DEBUG() << "FileParser::getFileData> grabbing [" << data << "]";
270 array[cnt++] = data.toDouble();
271 }
272 success = true;
273 line = in.readLine();
274 }
275 }
276 else if ( type == "LIST" ) {
277 // Extract list of data
278 QString nameline = "";
279 QString dataline = "";
280 QStringList nameItems;
281 QStringList dataItems;
282 dataList.clear();
283 // Grab the third line (Phasics)
284 nameline = in.readLine();
285 nameline = in.readLine();
286 nameline = in.readLine();
287 // Grab the successive line (Phasics)
288 dataline = in.readLine();
289 nameItems = nameline.split(sep);
290 dataItems = dataline.split(sep);
291 int namepos = 0;
292 foreach (const QString &name, nameItems) {
293 QLOG_DEBUG() << "FileParser::getFileData> parsed parameter : " << name;
294 foreach (const QString &nameRef, nameList) {
295 if ( nameRef == name ) {
296 QString data = dataItems.at(namepos);
297 data.replace(",",".");
298 QLOG_INFO() << "FileParser::getFileData> match with parameter : " << nameRef << " = " << data;
299 dataList.push_back(data.toDouble());
300 // Go back to previous line
301 if ( dataList.size() == nameList.size() )
302 success = true;
303 }
304 }
305 namepos++;
306 }
307 }
308 file.close();
309 }
310 // Check directory content deletion after
311 if ( del == "AFTER" ) {
312 foreach (const QString &filename, fileList)
313 directory.remove( filename );
314 }
315 return success;
316}
