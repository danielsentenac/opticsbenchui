173 bool
174 FileParser::getFileData()
175 {
176  bool success = false;
177  // Create QDir object to navigate into directory path
178  QDir directory(dir);
179  QDir subdirectory(dir + "/Results");
180  QStringList fileList = directory.entryList(QDir::Files, QDir::Time | QDir::Reversed);
181  QStringList subfileList = subdirectory.entryList(QDir::Files, QDir::Time | QDir::Reversed);
182  // Check directory and Results subdirectory content deletion
183  if ( del == "BEFORE" ) {
184  int cnt = 0;
185  foreach (const QString &filename, fileList) {
186  if ( cnt == fileList.count() )
187  break;
188  QLOG_DEBUG() << "FileParser::getFileData> deleting file" << filename;
189  directory.remove( filename );
190  cnt++;
191  }
192  cnt = 0;
193  foreach (const QString &filename, subfileList) {
194  if ( cnt == subfileList.count() )
195  break;
196  QLOG_DEBUG() << "FileParser::getFileData> deleting file" << filename;
197  subdirectory.remove( filename );
198  cnt++;
199  }
200  }
201 
202  // Apply a delay before new dir scan
203  usleep( (int) (delay.toFloat() * 1000000) );
204 
205  // Look for the requested file from new dir scan
206  QDir freshdir(dir + "/Results");
207  QFileInfoList freshFileInfoList = freshdir.entryInfoList(QDir::NoFilter, QDir::Time | QDir::Reversed);
208  QFile file ("");
209  uint filetime = 0;
210  parsedFileName = "";
211  foreach (const QFileInfo &fileinfo, freshFileInfoList) {
212  QString filename = fileinfo.fileName();
213  QLOG_DEBUG() << "FileParser::getFileData> treating file" << filename;
214  QDateTime date (fileinfo.lastModified());
215  uint filetimeTmp = static_cast<uint>(date.toSecsSinceEpoch());
216  if ( filename.left(prefix.length()) == prefix &&
217  filename.right(suffix.length()) == suffix ) {
218  if ( order == "FIRST" ) {
219  file.setFileName(freshdir.filePath(filename));
220  break;
221  }
222  else if ( order == "LAST" && filetime < filetimeTmp ) {
223  file.setFileName(freshdir.filePath(filename));
224  filetime = filetimeTmp;
225  }
226  }
227  }
228  // parse the file data
229  if ( file.fileName() == "" ) {
230  QLOG_WARN () << "FileParser::getFileData> No available file "
231  << " with prefix " << prefix
232  << " suffix " << suffix
233  << " in directory " << dir + "/Results";
234  return false;
235  }
236  else {
237  QLOG_INFO () << "FileParser::getFileData> parsing file " << file.fileName();
238  parsedFileName = file.fileName();
239  // Open File
240  if ( format == "ASCII" )
241  file.open(QIODevice::ReadOnly | QIODevice::Text);
242  else
243  file.open(QIODevice::ReadOnly);
244  QTextStream in(&file);
245  if ( type == "ARRAY" ) {
246  // Get array size in File
247  arrayHeight = 1;
248  arrayWidth = 0;
249  QString line = "";
250  QStringList dataItems;
251  QLOG_DEBUG() << "FileParser::getFileData> reading..." << file.fileName();
252  QLOG_INFO() << "FileParser::getFileData> separator [" << sep << "]";
253  line = in.readLine();
254  while (!in.atEnd()) {
255  QLOG_DEBUG() << "FileParser::getFileData> reading..." << line ;
256  dataItems = line.split(sep);
257  arrayHeight++;
258  arrayWidth = dataItems.count();
259  line = in.readLine();
260  }
261  QLOG_INFO() << "FileParser::getFileData> arrayHeight = " << arrayHeight
262  << " arrayWidth = " << arrayWidth;
263  if ( array ) { free(array); array = NULL;}
264  array = (double*) malloc(sizeof(double) * arrayHeight * arrayWidth);
265  // Parse the file to array
266  in.seek(0);
267  line = "";
268  int cnt = 0;
269  line = in.readLine();
270  while (!in.atEnd()) {
271  QLOG_DEBUG() << "FileParser::getFileData> grabbing..." << line ;
272  dataItems = line.split(sep);
273  foreach (const QString &data, dataItems) {
274  QLOG_DEBUG() << "FileParser::getFileData> grabbing [" << data << "]";
275  array[cnt++] = data.toDouble();
276  }
277  success = true;
278  line = in.readLine();
279  }
280  }
281  else if ( type == "LIST" ) {
282  // Extract list of data
283  QString nameline = "";
284  QString dataline = "";
285  QStringList nameItems;
286  QStringList dataItems;
287  dataList.clear();
288  // Grab the third line (Phasics)
289  nameline = in.readLine();
290  nameline = in.readLine();
291  nameline = in.readLine();
292  // Grab the successive line (Phasics)
293  dataline = in.readLine();
294  nameItems = nameline.split(sep);
295  dataItems = dataline.split(sep);
296  int namepos = 0;
297  foreach (const QString &name, nameItems) {
298  QLOG_DEBUG() << "FileParser::getFileData> parsed parameter : " << name;
299  foreach (const QString &nameRef, nameList) {
300  if ( nameRef == name ) {
301  QString data = dataItems.at(namepos);
302  data.replace(",",".");
303  QLOG_INFO() << "FileParser::getFileData> match with parameter : " << nameRef << " = " << data;
304  dataList.push_back(data.toDouble());
305  // Go back to previous line
306  if ( dataList.size() == nameList.size() )
307  success = true;
308  }
309  }
310  namepos++;
311  }
312  }
313  file.close();
314  }
315  // Check directory content deletion after
316  if ( del == "AFTER" ) {
317  foreach (const QString &filename, fileList)
318  directory.remove( filename );
319  }
320  return success;
321 }
