489void
490CameraRaspi::run() {
491 int acq_err = 0;
492 int acq_cnt = 0;
493 if (camera_err == 0 && suspend == true ) {
494 suspend = false;
495 eTimeTotal = 0;
496 while (suspend == false) {
497 QLOG_DEBUG () << "CameraRaspi::run> " << id << " : start new Acquisition";
498 double eTime = Utils::GetTimeMicroseconds();
499 acqstart->wakeAll();
500 acq_err = acquireImage();
501 QLOG_DEBUG () << "CameraRaspi::run> " << id << " : done";
502 acqend->wakeAll();
503 if (acq_err == 1)
504 eTimeTotal+= (Utils::GetTimeMicroseconds() - eTime);
505 acq_cnt++;
506 if (acq_cnt == FREQUENCY_AVERAGE_COUNT) {
507 eTimeTotal/=1e6;
508 eTimeTotal/=FREQUENCY_AVERAGE_COUNT;
509 if ( eTimeTotal > 0 )
510 frequency = 1.0 / eTimeTotal;
511 else
512 frequency = 0;
513 QLOG_DEBUG() << "CameraRaspi::run> Acquisition "
514 << "freq " << (int) frequency << " Hz";
515 eTimeTotal = 0;
516 acq_cnt = 0;
517 // Update props
518 getProps();
519 }
520 has_started = true;
521 }
522 QLOG_DEBUG() << "CameraRaspi thread exiting";
523 }

902int
903CameraRaspi::acquireImage() {
904
905 int acq_err = 1;
906 QLOG_DEBUG () << " Cycle start ";
907 acquireMutex->lock();
908 /*-----------------------------------------------------------------------
909 * acquire frame
910 *-----------------------------------------------------------------------*/
911 double eTime = Utils::GetTimeMicroseconds();
912 camera->grab();
913 camera->retrieve ( data );
914 if (data == nullptr) {
915 QLOG_ERROR() << "ACQUISITION ERROR";
916 return (-1);
917 }
918 // Lock the acquisition
919 snapshotMutex->lock();
920 //QLOG_INFO() << width << " " << height;
921 //QLOG_INFO() << "buffersize=" << camera->getImageBufferSize();
922 //for (int i = 0 ; i < 100; i++)
923 // QLOG_INFO() << "data: " << data[i];
924 // calculate min,max
925 max = 0;
926 min = 65535;
927 //QLOG_INFO() << "pixel encoding = " << pixel_encoding;
928 if ( pixel_encoding == B8 ) {
929 for (unsigned int i = 0 ; i < width * height; i++) {
930 if (data[i] > max)
931 max = data[i];
932 if (data[i] < min)
933 min = data[i];
934 }
935 }
936 //QLOG_INFO() << "min = " << min << " max = " << max;
937 // copy image to buffer (8-bit : 0 - 255 range) and to buffer32 (32-bit)
938 // Reset buffer (8-bit) and buffer32 (32-bit)
939 memset(buffer,0,height * width * sizeof(uchar));
940 memset(buffer32,0,height * width * sizeof(int));
941
942 if ( pixel_encoding == B8 ) {
943 memcpy(buffer,data,width * height * sizeof(uchar));
944 for (unsigned int i = 0 ; i < width * height; i++)
945 buffer32[i] = buffer[i];
946 }
947
948 if (vflip) {
949 buffer = reversebytes(buffer,height * width);
950 buffer = fliphorizontal(buffer,height * width, width);
951 buffer32 = reversebytes(buffer32,height * width);
952 buffer32 = fliphorizontal(buffer32,height * width, width);
953 }
954 if (!hflip) {
955 buffer = fliphorizontal(buffer,height*width, width);
956 buffer32 = fliphorizontal(buffer32,height*width, width);
957 }
958 snapshotMutex->unlock();
959 // Format video image
960 QLOG_DEBUG() << "size IMAGE = " << imageWidth << " " << imageHeight;
961 QImage imagescaled = image->scaled(imageWidth,imageHeight);
962 QImage imagergb32 = imagescaled.convertToFormat(QImage::Format_ARGB32_Premultiplied);
963 emit getImage(imagergb32);
964 emit updateMin(min);
965 emit updateMax(max);
966 eTime = Utils::GetTimeMicroseconds() - eTime;
967 QLOG_DEBUG () << " Process eTime " << QString::number(eTime);
968 QLOG_DEBUG () << " Cycle end ";
969 acquireMutex->unlock();
970 return (acq_err);
971}
