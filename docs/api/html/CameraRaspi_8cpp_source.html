494 void
495 CameraRaspi::run() {
496  int acq_err = 0;
497  int acq_cnt = 0;
498  if (camera_err == 0 && suspend == true ) {
499  suspend = false;
500  eTimeTotal = 0;
501  while (suspend == false) {
502  QLOG_DEBUG () << "CameraRaspi::run> " << id << " : start new Acquisition";
503  double eTime = Utils::GetTimeMicroseconds();
504  acqstart->wakeAll();
505  acq_err = acquireImage();
506  QLOG_DEBUG () << "CameraRaspi::run> " << id << " : done";
507  acqend->wakeAll();
508  if (acq_err == 1)
509  eTimeTotal+= (Utils::GetTimeMicroseconds() - eTime);
510  acq_cnt++;
511  if (acq_cnt == FREQUENCY_AVERAGE_COUNT) {
512  eTimeTotal/=1e6;
513  eTimeTotal/=FREQUENCY_AVERAGE_COUNT;
514  if ( eTimeTotal > 0 )
515  frequency = 1.0 / eTimeTotal;
516  else
517  frequency = 0;
518  QLOG_DEBUG() << "CameraRaspi::run> Acquisition "
519  << "freq " << (int) frequency << " Hz";
520  eTimeTotal = 0;
521  acq_cnt = 0;
522  // Update props
523  getProps();
524  }
525  has_started = true;
526  }
527  QLOG_DEBUG() << "CameraRaspi thread exiting";
528  }

531 void
532 CameraRaspi::setFeature(int feature, double value) {
533 
534  QLOG_INFO() << " FEATURE NUMBER " << feature << " value requested " << (int)value;
535  // Stop Acquisition
536  this->stop();
537  //acquireMutex->lock();
538  QLOG_INFO () << "CameraRaspi::setFeature> AcquisitionStop...";
539  switch ( feature ) {
540  case 0:
541  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[0])
542  << " value " << QString::number(value);
543  camera->setExposure((RASPICAM_EXPOSURE)value);
544  break;
545  case 1:
546  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[1])
547  << "value " << (int)value;
548  camera->setISO((int)value);
549  break;
550  case 2:
551  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[2])
552  << " value " << (int)value;
553  camera->setAWB((RASPICAM_AWB)value);
554  break;
555  case 3:
556  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[3])
557  << " value " << (int)value;
558  camera->setSharpness((int)value);
559  break;
560  case 4:
561  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[4])
562  << " value " << (int)value;
563  camera->setBrightness((int)value);
564  break;
565  case 5:
566  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[5])
567  << " value " << (int)value;
568  camera->setSaturation((int)value);
569  break;
570  case 6:
571  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[6])
572  << " value " << (int)value;
573  switch ((int)value) {
574  case 0: camera->setVideoStabilization(false);
575  videostabilization_num = 0;
576  break;
577  case 1: camera->setVideoStabilization(true);
578  videostabilization_num = 1;
579  break;
580  }
581  break;
582  case 7:
583  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[7])
584  << " value " << (int)value;
585  camera->setExposureCompensation((int)value);
586  exposure_compensation = (int) value;
587  break;
588  case 8:
589  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[8])
590  << " value " << (int)value;
591  camera->setShutterSpeed((unsigned int)(value));
592  shutterspeed = value;
593  break;
594  case 9:
595  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[9])
596  << " value " << (int)value;
597  camera->setFrameRate((int)value);
598  break;
599  case 10:
600  QLOG_INFO() << "CameraRaspi::setFeature> Update feature " << QString(raspi_features[10])

910  int acq_err = 1;
911  QLOG_DEBUG () << " Cycle start ";
912  acquireMutex->lock();
913  /*-----------------------------------------------------------------------
914  * acquire frame
915  *-----------------------------------------------------------------------*/
916  double eTime = Utils::GetTimeMicroseconds();
917  camera->grab();
918  camera->retrieve ( data );
919  if (data == nullptr) {
920  QLOG_ERROR() << "ACQUISITION ERROR";
921  return (-1);
922  }
923  // Lock the acquisition
924  snapshotMutex->lock();
925  //QLOG_INFO() << width << " " << height;
926  //QLOG_INFO() << "buffersize=" << camera->getImageBufferSize();
927  //for (int i = 0 ; i < 100; i++)
928  // QLOG_INFO() << "data: " << data[i];
929  // calculate min,max
930  max = 0;
931  min = 65535;
932  //QLOG_INFO() << "pixel encoding = " << pixel_encoding;
933  if ( pixel_encoding == B8 ) {
934  for (unsigned int i = 0 ; i < width * height; i++) {
935  if (data[i] > max)
936  max = data[i];
937  if (data[i] < min)
938  min = data[i];
939  }
940  }
941  //QLOG_INFO() << "min = " << min << " max = " << max;
942  // copy image to buffer (8-bit : 0 - 255 range) and to buffer32 (32-bit)
943  // Reset buffer (8-bit) and buffer32 (32-bit)
944  memset(buffer,0,height * width * sizeof(uchar));
945  memset(buffer32,0,height * width * sizeof(int));
946 
947  if ( pixel_encoding == B8 ) {
948  memcpy(buffer,data,width * height * sizeof(uchar));
949  for (unsigned int i = 0 ; i < width * height; i++)
950  buffer32[i] = buffer[i];
951  }
952

955  buffer = fliphorizontal(buffer,height * width, width);

957  buffer32 = fliphorizontal(buffer32,height * width, width);

960  buffer = fliphorizontal(buffer,height*width, width);
961  buffer32 = fliphorizontal(buffer32,height*width, width);
962  }
963  snapshotMutex->unlock();
964  // Format video image

968  emit getImage(imagergb32);

970  emit updateMax(max);
971  eTime = Utils::GetTimeMicroseconds() - eTime;
972  QLOG_DEBUG () << " Process eTime " << QString::number(eTime);
973  QLOG_DEBUG () << " Cycle end ";
974  acquireMutex->unlock();
