550void
551CameraNeo::run() {
552 int acq_err = 0;
553 int acq_cnt = 0;
554 if (camera_err == 0 && suspend == true ) {
555 suspend = false;
556 eTimeTotal = 0;
557 while (suspend == false) {
558 QLOG_DEBUG () << "CameraNeo::run> " << id << " : start new Acquisition";
559 double eTime = Utils::GetTimeMicroseconds();
560 acqstart->wakeAll();
561 acq_err = acquireImage();
562 QLOG_DEBUG () << "CameraNeo::run> " << id << " : done";
563 acqend->wakeAll();
564 if (acq_err == 1)
565 eTimeTotal+= (Utils::GetTimeMicroseconds() - eTime);
566 acq_cnt++;
567 if (acq_cnt == FREQUENCY_AVERAGE_COUNT) {
568 eTimeTotal/=1e6;
569 eTimeTotal/=FREQUENCY_AVERAGE_COUNT;
570 if ( eTimeTotal > 0 )
571 frequency = 1.0 / eTimeTotal;
572 else
