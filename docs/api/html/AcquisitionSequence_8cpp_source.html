476 bool
477 AcquisitionSequence::setAvg(AcquisitionSequence *sequenceLeft, AcquisitionSequence *sequenceRight) {
478 
479  uchar *imageRight = NULL;
480  uchar *imageLeft = NULL;
481  int *imageRight32 = NULL;
482  int *imageLeft32 = NULL;
483  bool success = false;
484  if (sequenceLeft != (AcquisitionSequence *)NULL && sequenceRight != (AcquisitionSequence *)NULL ) {
485  if (sequenceLeft->status == true && sequenceRight->status == true) {
486  if (sequenceLeft->instrumentType == "CAMERA" &&
487  sequenceRight->instrumentType == "CAMERA") {
488  instrumentRef = "CAMERA";
489  if ( sequenceRight->getImage() )
490  imageRight = sequenceRight->getImage();
491  else if ( sequenceRight->getImage32() )
492  imageRight32 = sequenceRight->getImage32();
493  if ( sequenceLeft->getImage() )
494  imageLeft = sequenceLeft->getImage();
495  else if ( sequenceLeft->getImage32() )
496  imageLeft32 = sequenceLeft->getImage32();
497  if (reset == true) {
498  if (data_2D_FLOAT) {
499  free(data_2D_FLOAT);
500  data_2D_FLOAT = NULL;
501  }
502  if (sequenceLeft->imageWidth <= sequenceRight->imageWidth)
503  data_2D_FLOAT_DIM_X = sequenceLeft->imageWidth;
504  else
505  data_2D_FLOAT_DIM_X = sequenceRight->imageWidth;
506  if (sequenceLeft->imageHeight <= sequenceRight->imageHeight)
507  data_2D_FLOAT_DIM_Y = sequenceLeft->imageHeight;
508  else
509  data_2D_FLOAT_DIM_Y = sequenceRight->imageHeight;
510  data_2D_FLOAT = (float*) malloc (sizeof(float) * data_2D_FLOAT_DIM_X * data_2D_FLOAT_DIM_Y);
511  memset(data_2D_FLOAT,0,sizeof(data_2D_FLOAT));
512  reset = false;
513  }
514  for (int i = 0 ; i < data_2D_FLOAT_DIM_X * data_2D_FLOAT_DIM_Y; i++) {
515  if ( imageLeft && imageRight ) {
516  if ( avgOp == "+" )
517  data_2D_FLOAT[i] = data_2D_FLOAT[i] + (imageLeft[i] + imageRight[i]);
518  else if ( avgOp == "-" )
519  data_2D_FLOAT[i] = data_2D_FLOAT[i] + (imageLeft[i] - imageRight[i]);
520  else if ( avgOp == "*" )
521  data_2D_FLOAT[i] = data_2D_FLOAT[i] + (imageLeft[i] * imageRight[i]);
522  else if ( avgOp == "/") {
523  if (imageRight[i] != 0)
524  data_2D_FLOAT[i] = data_2D_FLOAT[i] + (imageLeft[i] / imageRight[i]);
525  }
526  }
527  else if ( imageLeft32 && imageRight32 ) {
528  if ( avgOp == "+" )
529  data_2D_FLOAT[i] = data_2D_FLOAT[i] + (imageLeft32[i] + imageRight32[i]);
530  else if ( avgOp == "-" )
531  data_2D_FLOAT[i] = data_2D_FLOAT[i] + (imageLeft32[i] - imageRight32[i]);
532  else if ( avgOp == "*" )
533  data_2D_FLOAT[i] = data_2D_FLOAT[i] + (imageLeft32[i] * imageRight32[i]);
534  else if ( avgOp == "/") {
535  if (imageRight[i] != 0)
536  data_2D_FLOAT[i] = data_2D_FLOAT[i] + (imageLeft32[i] / imageRight32[i]);
537  }
538  }
539  }
540  success = true;
541  }
542  else if (sequenceLeft->instrumentType.contains("DAC") &&
543  sequenceRight->instrumentType.contains("DAC")) {
544  instrumentRef = "DAC";
545  if (reset == true) {
546  dacValue = 0;
547  }
548  if ( avgOp == "+" )
549  dacValue = dacValue + (sequenceLeft->dacValue + sequenceRight->dacValue);
550  else if ( avgOp == "-" )
551  dacValue = dacValue + (sequenceLeft->dacValue - sequenceRight->dacValue);
552  else if ( avgOp == "*" )
553  dacValue = dacValue + (sequenceLeft->dacValue * sequenceRight->dacValue);
554  else if ( avgOp == "/" )
