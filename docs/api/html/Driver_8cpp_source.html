239int
240Driver::SendGeneralCommand(char *buffer,string& rply) const
241{
242 int retStatus = 1;
243 string answer = "";
244 string command = "";
245 char mybuffer[32];
246 vector<string> mBufferVect;
247 string strbuffer = buffer;
248
249 if (!mBufferVect.empty()) mBufferVect.erase(mBufferVect.begin(),mBufferVect.end());
250 Tokenize(strbuffer,mBufferVect,",");
251 for (unsigned int i = 0 ; i < mBufferVect.size(); i++ ) {
252 sprintf(mybuffer,"%s\r",mBufferVect[i].c_str());
253 command = mybuffer;
254 QLOG_DEBUG ( ) << "Driver> Send Command " << QString(command.c_str());
256 // Apply Command
258 if (_pcommChannel->Write(command) < (int)command.length()) {
259 ReportCommError("Driver> Unable to write to port");
260 return -1;
261 }
262 usleep(100000);
263 }
265 // Read reply
267 if (!_pcommChannel->Read(answer)) {
268 ReportCommError("Driver> Unable to read from port");
269 return -2;
270 }
271 QLOG_DEBUG ( ) << "Driver> Reply to Command is :" << QString(answer.c_str());
272 rply = answer;
273 QLOG_DEBUG ( ) << "Driver> returning retStatus " << retStatus;
274 return retStatus;
275}

277void
278Driver::Tokenize(const string& str,
279 vector<string>& tokens,
280 const string& delimiters) const
281{
282 // Skip delimiters at beginning.
283 string::size_type lastPos = str.find_first_not_of(delimiters, 0);
284 // Find first "non-delimiter".
285 string::size_type pos = str.find_first_of(delimiters, lastPos);
286
287 while (string::npos != pos || string::npos != lastPos)
288 {
289 // Found a token, add it to the vector.
290 tokens.push_back(str.substr(lastPos, pos - lastPos));
291 // Skip delimiters. Note the "not_of"
292 lastPos = str.find_first_not_of(delimiters, pos);
293 // Find next "non-delimiter"
294 pos = str.find_first_of(delimiters, lastPos);
295 }
296}
