50 void AnalysisThread::run() {
51  {
52  QMutexLocker locker(mutex);
53  if (!suspend) {
54  return;
55  }
56  suspend = false;
57  }
58 
59  for (const AnalysisTask& task : tasks) {
60  {
61  QMutexLocker locker(mutex);
62  if (suspend) {
63  break;
64  }
65  }
66  if (task.codePath.trimmed().isEmpty()) {
67  emit showWarning("Analysis task has empty code path; skipping.");
68  continue;
69  }
70 
71  emit analysisStarted(task.record);
72  QLOG_INFO() << "AnalysisThread::run> Starting task " << task.record
73  << " cmd=" << task.codePath << " args=" << task.arguments;
74 
75  QString program = task.codePath.trimmed();
76  QStringList arguments = splitArguments(task.arguments);
77  if (program.contains(' ')) {
78  const QStringList commandParts = splitArguments(program);
79  if (!commandParts.isEmpty()) {
80  program = commandParts.first();
81  QStringList extraArgs = commandParts;
82  extraArgs.removeFirst();
83  if (!extraArgs.isEmpty()) {
84  arguments = extraArgs + arguments;
85  }
86  }
87  }
88 
89  QProcess localProcess;
90  localProcess.setProcessChannelMode(QProcess::MergedChannels);
91  localProcess.start(program, arguments);
92 
93  if (!localProcess.waitForStarted()) {
94  const QString output =
95  "Unable to start command.\n"
96  "Command: " + program + "\n"
97  "Arguments: " + arguments.join(' ') + "\n"
98  "Error: " + localProcess.errorString();
99  emit analysisFinished(task.record, false, output);
100  emit showWarning("Unable to start analysis command: " + task.codePath);
101  continue;
102  }
103 
104  bool stopped = false;
105  QString outputBuffer;
106  while (!localProcess.waitForFinished(100)) {
107  const QString chunk = localProcess.readAllStandardOutput();
108  if (!chunk.isEmpty()) {
109  outputBuffer += chunk;
110  emit analysisOutput(task.record, chunk);
111  }
112  QMutexLocker locker(mutex);
113  if (suspend) {
114  stopped = true;
115  break;
116  }
117  }
118 
119  if (stopped) {
120  localProcess.terminate();
121  localProcess.waitForFinished(2000);
122  if (localProcess.state() != QProcess::NotRunning) {
123  localProcess.kill();
124  localProcess.waitForFinished(2000);
125  }
126  const QString chunk = localProcess.readAllStandardOutput();
127  if (!chunk.isEmpty()) {
128  outputBuffer += chunk;
129  emit analysisOutput(task.record, chunk);
130  }
131  emit analysisFinished(task.record, false, "Stopped by user.");
132  break;
133  }
134 
135  const QString remaining = localProcess.readAllStandardOutput();
136  if (!remaining.isEmpty()) {
137  outputBuffer += remaining;
138  emit analysisOutput(task.record, remaining);
139  }
140 
141  const bool success =
142  (localProcess.exitStatus() == QProcess::NormalExit &&
143  localProcess.exitCode() == 0);
144  const QString output =
145  "Exit code: " + QString::number(localProcess.exitCode());
146  emit analysisFinished(task.record, success, output);
147  }
148 
149  QMutexLocker locker(mutex);
150  suspend = true;
151  QLOG_DEBUG() << "AnalysisThread::run> End of thread";
152 }
