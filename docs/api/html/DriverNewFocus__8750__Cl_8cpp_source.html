290 int DriverNewFocus_8750_Cl::OperationComplete(string& rstateData,
291  string actuatorSetting,
292  ADLimitSwitch& rlimitSwitch) const
293 {
294  QLOG_DEBUG() << "DriverNewFocus_8750_Cl::OperationComplete";
295 
296  const unsigned int UPPER_LIMIT_STATE = 0x40;
297  const unsigned int LOWER_LIMIT_STATE = 0x20;
298  const unsigned int MOTION_PERFORMED = 0x01;
299 
300  int retStatus = -1;
301  unsigned int driverStatus;
302  string answer;
303  string getStatusCommand;
304  string servoLoopEnableCommand;
305  istringstream streamStatus;
306 
307  getStatusCommand = "STA A" + _setting + "\n";
308  servoLoopEnableCommand = "SER A" + _setting + "\n";
309 
310  if (SendCommand(getStatusCommand,answer))
311  {
312  QLOG_DEBUG() << "Operation Complete failed for A" << QString(_setting.c_str());
313  retStatus = -1;
314  }
315  else
316  {
317  unsigned int statusIndice;
318  rstateData = answer;
319  statusIndice = answer.find(DELIMITER,0);
320  if (statusIndice != string::npos)
321  {
322  // status reading
323  statusIndice++;
324  streamStatus.str(answer.substr(statusIndice,answer.length()-statusIndice));
325  streamStatus.setf(ios_base::hex, ios_base::basefield);
326  streamStatus >> driverStatus;
327 
328  // reached limit checking
329  if (!(driverStatus & UPPER_LIMIT_STATE) &&
330  !(driverStatus & LOWER_LIMIT_STATE))
331  {
332  // end of motion status reading
333  retStatus = driverStatus & 0x01;
334  rlimitSwitch = IN_BETWEEN;
335  }
336  // driver may be in open loop mode
337  // better to switch again in the closed loop one
338  else if (SendCommand(servoLoopEnableCommand,answer) != WRITE_ERROR )
339  {
340 
341  if ((driverStatus & UPPER_LIMIT_STATE) &&
342  (driverStatus & LOWER_LIMIT_STATE))
343  {
344  rlimitSwitch = UNAVAILABLE;
345  }
346  else if (driverStatus & UPPER_LIMIT_STATE)
347  {
348  rlimitSwitch = UPPER_LIMIT;
349  retStatus = MOTION_PERFORMED;
350  }
351  else if (driverStatus & LOWER_LIMIT_STATE)
352  {
353  rlimitSwitch = LOWER_LIMIT;
354  retStatus = MOTION_PERFORMED;
355  }
356  }
357  }
358  }
359 
360  return retStatus;
361 }
