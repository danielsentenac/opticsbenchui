52

56 getStatusCommand = "STA A" + _setting + "\n";
57
58 if (SendCommand(defaultLoadCommand,answer) == WRITE_ERROR ||
59 SendCommand(enableCommand,answer) == WRITE_ERROR ||
60 SendCommand(servoLoopEnableCommand,answer) == WRITE_ERROR ||
61 SendCommand(getStatusCommand,answer) ||
62 answer.find(DELIMITER,0) == string::npos)
63
64 {
65 QLOG_DEBUG() << "Init failed for A" << QString(_setting.c_str());
66 retStatus = -1;
67 }
68 else
69 {
70 rstateData = answer;
71 }
72 return retStatus;

104 setFrequencyCommandstream << "VEL A" << _setting

106
107 setPositionCommandstream << "POS A" << _setting << " " << position << "\n";
108

110 (SendCommand(setPositionCommandstream.str(),answer) == WRITE_ERROR))
111 {
112 QLOG_DEBUG() << "Init actuator failed on A" << QString(_setting.c_str()) << ":" << motor;
113 retStatus = -1;

285int DriverNewFocus_8750_Cl::OperationComplete(string& rstateData,
286 string actuatorSetting,
287 ADLimitSwitch& rlimitSwitch) const
288{
289 QLOG_DEBUG() << "DriverNewFocus_8750_Cl::OperationComplete";
290
291 const unsigned int UPPER_LIMIT_STATE = 0x40;
292 const unsigned int LOWER_LIMIT_STATE = 0x20;
293 const unsigned int MOTION_PERFORMED = 0x01;
294
295 int retStatus = -1;
296 unsigned int driverStatus;
297 string answer;
298 string getStatusCommand;
299 string servoLoopEnableCommand;
300 istringstream streamStatus;
301
302 getStatusCommand = "STA A" + _setting + "\n";
303 servoLoopEnableCommand = "SER A" + _setting + "\n";
304
305 if (SendCommand(getStatusCommand,answer))
306 {
307 QLOG_DEBUG() << "Operation Complete failed for A" << QString(_setting.c_str());
308 retStatus = -1;
309 }
310 else
311 {
312 unsigned int statusIndice;
313 rstateData = answer;
314 statusIndice = answer.find(DELIMITER,0);
315 if (statusIndice != string::npos)
316 {
317 // status reading
318 statusIndice++;
319 streamStatus.str(answer.substr(statusIndice,answer.length()-statusIndice));
320 streamStatus.setf(ios_base::hex, ios_base::basefield);
321 streamStatus >> driverStatus;
322
323 // reached limit checking
324 if (!(driverStatus & UPPER_LIMIT_STATE) &&
325 !(driverStatus & LOWER_LIMIT_STATE))
326 {
327 // end of motion status reading
328 retStatus = driverStatus & 0x01;
329 rlimitSwitch = IN_BETWEEN;
330 }
331 // driver may be in open loop mode
332 // better to switch again in the closed loop one
333 else if (SendCommand(servoLoopEnableCommand,answer) != WRITE_ERROR )
334 {
335
336 if ((driverStatus & UPPER_LIMIT_STATE) &&
337 (driverStatus & LOWER_LIMIT_STATE))
338 {
339 rlimitSwitch = UNAVAILABLE;
340 }
341 else if (driverStatus & UPPER_LIMIT_STATE)
342 {
343 rlimitSwitch = UPPER_LIMIT;
344 retStatus = MOTION_PERFORMED;
345 }
346 else if (driverStatus & LOWER_LIMIT_STATE)
347 {
348 rlimitSwitch = LOWER_LIMIT;
349 retStatus = MOTION_PERFORMED;
350 }
351 }
352 }
353 }
354
355 return retStatus;
356}

389{
390 QLOG_DEBUG() << "DriverNewFocus_8750_Cl::SendCommand";
391
