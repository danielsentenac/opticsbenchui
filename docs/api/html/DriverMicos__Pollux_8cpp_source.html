53//
54// InitActuator implementation
55//
56// use company delivered optimized settings
57//
58int DriverMicos_Pollux::InitActuator(string actuatorSetting,
59 float position) const
60{
61 QLOG_DEBUG ( ) << "DriverMicos_Pollux::InitActuator";
62
63 int retStatus = 0;
64 string command = "";
65 double vel = 0;
66 char buffer[BUFFER_SIZE];
67 // parse configuration parameters
68 // vel=xxx
69 if (sscanf(actuatorSetting.c_str(),
70 "vel=%lf",
71 &vel) != NB_ITEM_INIT_SETTING) {
72 ReportSettingError("Actuator setting string format incorrect");
73 retStatus = -1;
74 return retStatus;
75 }
76
77 // first send the address code
78 if (SendAddressCode() != 0) {
79 QLOG_DEBUG ( ) << "Could not send address code";
80 retStatus = -1;
81 return retStatus;
82 }
83
84 // send SVxxx<CR> to set the velocity
85 sprintf(buffer, "%lf %d setnvel\r", vel,_axisNumber);
86 // send command
87 command = buffer;
88 if (_pcommChannel->Write(command,NULL) < (int)command.length()) {
89 ReportCommError(QString("Unable to write to port, command %1")
90 .arg(QString(command.c_str())));
91 retStatus = -1;
92 return retStatus;
93 }
94
95 // computation of the offset to apply to get a position
96 _originOffset = 0.;
97 return retStatus;
98}

100//
101// GetPos implementation
102//
103// 1. send "npos<CR>" (tell position) to get the current position
104// 2. read reply
105// 3. parse position from reply
106//
107int DriverMicos_Pollux::GetPos(string actuatorSetting,
108 float& position) const
109{
110 QLOG_DEBUG ( ) << "DriverMicos_Pollux::GetPos";
111
112 int retStatus = 0;
113
114 string command = "";
115 string answer = "";
116 char buffer[BUFFER_SIZE];
117 // init position to zero
118 position = 0;
119
120 // first send the address code
121 if (SendAddressCode() != 0) {
122 QLOG_DEBUG ( ) << "Could not send address code";
123 retStatus = -1;
124 return retStatus;
125 }
126
127 // create buffer with command np<CR>
128 sprintf(buffer, "%d npos\r", _axisNumber);
129
130 // send command
131 command = buffer;
132 if (_pcommChannel->Write(command,NULL) < (int)command.length()) {
133 ReportCommError("Unable to write to port");
134 retStatus = -1;
135 return retStatus;
136 }
137
138 // read reply
139 if (!_pcommChannel->Read(answer,NULL)) {
140 ReportCommError("Unable to read from port");
141 retStatus = -1;
142 return retStatus;
143 }
144
145 // parse and extract position
146 const char* realAnswer = answer.c_str();
147 sscanf (realAnswer, "%f", &position);
148 QLOG_DEBUG ( ) << "GetPos gives : " << position << " origineoffset : " << _originOffset;
149 return retStatus;
150}

152//
153// Move implementation
154//
155// 1. send "nrmove<CR>" (move relative) to move xxxx steps
156//
157int DriverMicos_Pollux::Move(string actuatorSetting,
158 float nbSteps,
159 int unit) const
160{
161 QLOG_DEBUG ( ) << "DriverMicos_Pollux::Move";
162
163 int retStatus = 0;
164 float range;
165 float motion;
166
167 string command = "";
168 char buffer[BUFFER_SIZE];
169
170 if ( ConvertUnit(unit,nbSteps,motion,range) ||
171 (fabs(motion) > range))
172 {
173 retStatus = DISPLACEMENT_EXCEEDS_RANGE;
174 _lastError = OUT_OF_RANGE_ERROR;
175 }
176 else
177 {
178 // first send the address code
179 if (SendAddressCode() != 0) {
180 QLOG_DEBUG ( ) << "Could not send address code";
181 retStatus = -1;
182 return retStatus;
183 }
184
185 // create buffer with command MRxxxx<CR>
186 sprintf(buffer, "%f %d nrmove\r", motion, _axisNumber);
187 QLOG_DEBUG ( ) << "Send MoveRel Command " << QString( buffer);
188 // send command
189 command = buffer;
190 if (_pcommChannel->Write(command,NULL) < (int)command.length()) {
191 ReportCommError("Unable to write to port");
192 retStatus = -1;
193 }
194 }
195
196 return retStatus;
197}

275//
276// OperationComplete implementation
277//
278// 1. send "nst<CR>" to retrieve the status
279// 2. get the reply
280// 3. parse the reply
281// 4. compare the bits
282//
283//
284int DriverMicos_Pollux::OperationComplete(string& rstateData,
285 string actuatorSetting,
286 ADLimitSwitch& rlimitSwitch) const
287{
288 QLOG_DEBUG ( ) << "DriverMicos_Pollux::OperationComplete";
289
290 int retStatus = -1;
291
292 string command = "";
293 string answer = "";
294 char buffer[BUFFER_SIZE];
295 int status = 0;
296
297 // first send the address code
298 if (SendAddressCode() != 0) {
299 QLOG_DEBUG ( ) <<"Could not send address code";
300 retStatus = -1;
301 return retStatus;
302 }
303
304 // create buffer with command nst<CR>
305 sprintf(buffer, "%d nst\r", _axisNumber);
306 QLOG_DEBUG ( ) <<"send message : " <<QString( buffer);
307
308
309 // send command
310 command = buffer;
311 if (_pcommChannel->Write(command,NULL) < (int)command.length()) {
312 ReportCommError("Unable to write to port");
313 retStatus = -1;
314 return retStatus;
315 }
316
317 // read reply
318 if (!_pcommChannel->Read(answer,NULL)) {
319 ReportCommError("Unable to read from port");
320 retStatus = -1;
321 QLOG_DEBUG ( ) << "Status reply expected, but got reply " << QString(answer.c_str());
322 return retStatus;
323 }
324 QLOG_DEBUG ( ) << "Status reply = " << QString( answer.c_str());
325 rstateData = answer;
326 // No limits !
327 rlimitSwitch = IN_BETWEEN;
328
329 // Evaluate status of motion
330 const char* realAnswer = answer.c_str();
331 sscanf (realAnswer, "%d", &status);
332 if (status == 1) // motion not completed
333 retStatus = 0;
334 if (status == 0) // motion completed
335 retStatus = 1;
336
337 return retStatus;
338}

406// Use indifferently default units or custom units configuration (STEPS == MM)
407int DriverMicos_Pollux::ConvertUnit(int unit,
408 float valueToConvert,
409 float& rconvertedValue,
410 float& rrange) const
411{
412 QLOG_DEBUG ( ) << "DriverMicos_Pollux::ConvertUnit";
413
414 int retStatus = 0;
415 const float STEPS_FOR_ONE_MM = 1;
416
417 if (unit == CUSTOM_UNIT)
418 {
419 rrange = MICOS_POLLUX_FEATURE.customRange * STEPS_FOR_ONE_MM;
420 rconvertedValue = valueToConvert * STEPS_FOR_ONE_MM;
421 }
422 else if (unit == DEFAULT_UNIT)
423 {
424 rrange = MICOS_POLLUX_FEATURE.range;
425 rconvertedValue = valueToConvert;
426 }
427 else if (unit == FROM_DEFAULT_TO_CUSTOM)
428 {
429 rrange = MICOS_POLLUX_FEATURE.customRange;
430 rconvertedValue = valueToConvert / STEPS_FOR_ONE_MM;
431 }
432 else
433 {
434 retStatus = -1;
435 }
436
437 return retStatus;
438}
