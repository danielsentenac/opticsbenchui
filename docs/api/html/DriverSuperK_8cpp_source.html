307 DriverSuperK::readInterbus_Stream(unsigned char deviceId, unsigned char registerId,
308  unsigned char *data, int dataSize)
309 {
310  if (!data || dataSize <= 0) {
311  return false;
312  }
313  unsigned char tempData[255] = {};
314  if (comm->GetStatus() == ACCom::OPEN) {
315  sendInterbusMessage(deviceId, registerId, 0x04, NULL);
316  if (ReceiveMessage(deviceId, registerId, 0x08, tempData) == MessageReady) {
317  int copySize = dataSize;
318  if (copySize > static_cast<int>(sizeof(tempData))) {
319  copySize = static_cast<int>(sizeof(tempData));
320  }
321  memcpy(data, tempData, copySize);
322  if (copySize < dataSize) {
323  memset(data + copySize, 0, dataSize - copySize);
324  }
325  return true;
326  }
327  }
328  return false;

341 unsigned short
342 DriverSuperK::calcCRC16(unsigned char data, unsigned short OldCRC)
343 {
344  OldCRC = (unsigned short)((OldCRC >> 8) | (OldCRC << 8));
345  OldCRC ^= data;
346  OldCRC ^= (unsigned char)((OldCRC & 0xff) >> 4);
347  OldCRC ^= (unsigned short)((OldCRC << 8) << 4);
348  OldCRC ^= (unsigned short)(((OldCRC & 0xff) << 4) << 1);
349  return OldCRC;
350 }

358 DriverSuperK::addToTxMsgData(unsigned char data, bool escParse, bool updCRC)
359 {
360  if (updCRC)
361  txCRC = calcCRC16(data, txCRC);
362  if (escParse) {
363  if (data == 0x0D || data == 0x0A || data == 0x5E) {
364  txBuffer[txBufferSize] = 0x5E;
365  if (txBufferSize < sizeof(txBuffer))
366  txBufferSize++;
367  data += 0x40;
368  }
369  }
370  txBuffer[txBufferSize] = data;
371  //printf("txBuffer[%d]=0x%x\n",txBufferSize,txBuffer[txBufferSize]);
372  if (txBufferSize < sizeof(txBuffer)) // Assure we are not overloading the txBuffer
373  txBufferSize++;

375 void
376 DriverSuperK::sendInterbusMessage(unsigned char deviceId, unsigned char registerId, unsigned char msgType, unsigned char *data)
377 {
378  txCRC = 0;
379  txBufferSize = 0;
380  addToTxMsgData(0x0D, false, false); // Start of Telegram
381  addToTxMsgData(deviceId, true, true);
382  addToTxMsgData(masterId, true, true);
383  addToTxMsgData(msgType, true, true);
384  addToTxMsgData(registerId, true, true);
385  if (msgType == 0x05 && data != NULL) // 0x05 = Write messsage - 0x04 = Read message
386  addToTxMsgData(data, sizeof(data), true, true); // Add write data to message
387  addToTxMsgData((unsigned char)(txCRC >> 8), true, false);
388  addToTxMsgData((unsigned char)txCRC, true, false);
389  addToTxMsgData(0x0A, false, false); // End of Telegram
390 
391  string message;
392  for (int i = 0 ; i < txBufferSize; i++)
393  message[i] = txBuffer[i];
394  //printf("0x%x,0x%x,0x%x,0x%x,0x%x\n",message[0],message[1],message[2],message[3],message[4]);
395  QString buffersize = "SIZE="+QString::number(txBufferSize);
396  comm->Write(message, buffersize.toStdString().c_str());
397 }

401 int
402 DriverSuperK::ReceiveMessage(unsigned char deviceId, unsigned char registerId, unsigned char msgType, unsigned char *payload)
403 {
404  int rxIntData;
405  unsigned char rxCh;
406  rxState = Hunting_SOT;
407  rxTimeout = timeout_mS;
408  string message;
409  while (rxState == Hunting_SOT || rxState == Hunting_EOT) {
410  rxIntData = comm->Read(message,"BYTE");
411  if (rxIntData == -1) { // No RX data ready - Go and wait for 1mS
412  if (rxTimeout > 0) {
413  rxTimeout--;
414  usleep(1000);
415  }
416  else
417  rxState = Timeout_Error;
418  }
419  else {
420  // We have RX data ready
421  rxCh = (unsigned char) message[0];
422  if (rxState == Hunting_SOT) { // Hunting Start Of Telegram
423  if (rxCh == 0x0D) { // Got start of telegram
424  rxBufferSize = 0;
425  for (int i = 0 ; i < sizeof(rxBuffer); i++)
426  rxBuffer[i] = 0;
427  rxEscape = false;
428  rxCRC = 0;
429  rxState = Hunting_EOT;
430  }
431  }
432  else if (rxState == Hunting_EOT) { // Hunting End Of Telegram while collecting telegram
433  if (rxCh == 0x0A) { // Got end of telegram
434  if (rxBufferSize >= 5) { // Dest + Src + Type + msbCRC + lsbCRC - Minimum telegram length
435  if (rxCRC == 0) { // We have collected a message with valid CRC - Check the contents
436  if (rxBuffer[0] == masterId && rxBuffer[1] == deviceId && rxBuffer[2] == msgType && rxBuffer[3] == registerId){
437  // We have a complete telegram - Copy the payload to destination array
438  rxBufferSize--; // Remove CRC
439  rxBuffer[rxBufferSize] = 0;
440  rxBufferSize--;
441  rxBuffer[rxBufferSize] = 0;
442  QLOG_DEBUG() << "rxBufferSize = " << rxBufferSize;
443  for (int i = 4 ; i < rxBufferSize; i++) {
444  if ( rxBuffer[i] == '\0' && i < rxBufferSize - 1)
445  payload[i-4] = ' ';
446  else
447  payload[i-4] = rxBuffer[i];
448  }
449  rxState = MessageReady;
450  }
451  else
452  rxState = Content_Error;
453  }
454  else
455  rxState = CRC_Error;
456  }
457  else
458  rxState = Garbage_Error;
459  }
460  else { // Collecting telegram
461  if (rxCh == 0x5E)
462  rxEscape = true; // Got escape sequence
463  else { // Got normal telegram contents
464  if (rxEscape) {
465  rxBuffer[rxBufferSize] = (unsigned char)(rxCh - 0x40);
466  rxEscape = false;
467  }
468  else
469  rxBuffer[rxBufferSize] = rxCh;
470  rxCRC = calcCRC16(rxBuffer[rxBufferSize], rxCRC); // Update crc
471  if (rxBufferSize < sizeof(rxBuffer))
472  rxBufferSize++;
473  else
474  rxState = Overrun_Error;
475  }
476  }
477  }
478  }
479  }
480  return rxState;
481 }
