35{
36 QLOG_DEBUG () << "DriverNewPort_AGUC2::Init";
37 int retStatus = 0;
38 string command;
39 string answer;
40 int cnt = 0;

42
43 // Send initialization message to AG-UC2 device (USB port)
44 if (_pcommChannel->Write(command,"CONTROL",0xC0,0x00,0x00,0x00,0x00,1,0) < 1)
45 {
46 ReportCommError("DriverNewPort_AGUC2::Init> Unable to write to port");
47 _lastError = COMMUNICATION_ERROR;
48 retStatus = -1;
49 }
50 if (_pcommChannel->Write(command,"CONTROL",0xC0,0x04,0x08,0x00,0x00,1,0) < 1)
51 {
52 ReportCommError("DriverNewPort_AGUC2::Init> Unable to write to port");
53 _lastError = COMMUNICATION_ERROR;
54 retStatus = -1;
55 }
56 if (_pcommChannel->Write(command,"CONTROL",0xC0,0x03,0x8003,0x00,0x00,1,0) < 1)
57 {
58 ReportCommError("DriverNewPort_AGUC2::Init> Unable to write to port");
59 _lastError = COMMUNICATION_ERROR;
60 retStatus = -1;
61 }
62 if (_pcommChannel->Write(command,"CONTROL",0xC0,0x02,0x00,0x00,0x00,1,0) < 1)
63 {
64 ReportCommError("DriverNewPort_AGUC2::Init> Unable to write to port");
65 _lastError = COMMUNICATION_ERROR;
66 retStatus = -1;
67 }
68 if (retStatus == -1) return retStatus;
69 // Ask device name
70 command = "VE\r\n";

74 retStatus = 0;

79 _lastError = COMMUNICATION_ERROR;
80 retStatus = -1;

83 usleep(10000);

85 {
86 ReportCommError("DriverNewPort_AGUC2::Init> Unable to read from port");

159 retStatus = -1;
