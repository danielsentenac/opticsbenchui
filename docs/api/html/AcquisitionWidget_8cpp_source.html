294 void AcquisitionWidget::run() {
295  this->stop();
296  QSqlQuery query(QSqlDatabase::database(path));
297  query.exec("update acquisition_sequence set status = ''");
298  query.exec("update acquisition_sequence set acquiring = ''");
299  query.exec("select record,type,instrument,settings,scanplan from acquisition_sequence "
300  "order by record");
301 
302  // Store acquisition sequence
303  sequenceList.clear();
304  cur_record = 0;
305  while (query.next()) {
306  int seq_record = query.value(0).toInt();
307  if (seq_record < 0) {
308  continue;
309  }
310  AcquisitionSequence *sequence = new AcquisitionSequence();
311  sequence->record = cur_record++;
312  sequence->seq_record = seq_record;
313  sequence->instrumentType = query.value(1).toString();
314  sequence->instrumentName = query.value(2).toString();
315  sequence->settings = query.value(3).toString();
316  sequence->scanplan = query.value(4).toString();
317  sequenceList.append(sequence);
318  }
319 
320  // Prepare sequence
321  for (int i = 0; i < sequenceList.size(); i++) {
322  AcquisitionSequence *sequence = sequenceList.at(i);
323  sequence->prepare();
324  }
325  if (motor) {
326  QSet<QString> motorNames;
327  for (int i = 0; i < sequenceList.size(); i++) {
328  const AcquisitionSequence *sequence = sequenceList.at(i);
329  if (sequence->instrumentType == "MOTOR") {
330  motorNames.insert(sequence->instrumentName);
331  }
332  }
333  for (const QString& name : motorNames) {
334  if (!name.isEmpty()) {
335  motor->connectMotor(name);
336  }
337  }
338  }
339  // Set File name and number from widget
340  QString filepath = acqfile + "_" + filenumber + ".h5";
341  acquisition->setFile(filepath, filenumber.toInt());
342  acquisition->setCamera(cameraList);
343  acquisition->setMotor(motor);
344  acquisition->setSuperK(superk);
345  acquisition->setDac(dac);
346  acquisition->setComediCounter(comedicounter);
347  acquisition->setComediDac(comedidac);
348  acquisition->setRaspiDac(raspidac);
349  acquisition->setSequenceList(sequenceList);
350  acquisition->start();
351  QLOG_DEBUG() << "AcquisitionWidget::run started";
352 }

415 void AcquisitionWidget::getAcquiring(int record) {
416  QSqlQuery query(QSqlDatabase::database(path));
417 
418  query.exec("update acquisition_sequence set acquiring = ''");
419  query.exec("select record from acquisition_sequence order by record");
420  QLOG_DEBUG() << query.lastError().text();
421  int seq_record = 0;
422  int count = 0;
423  while (query.next()) {
424  seq_record = query.value(0).toInt();
425  if (seq_record < 0) {
426  continue;
427  }
428  if (count == record) {
429  break;
430  }
431  count++;
432  }
433  QLOG_DEBUG() << seq_record;
434  query.prepare("update acquisition_sequence set acquiring = '*' where record = ?");
435  query.addBindValue(seq_record);
436  query.exec();
437  InitConfig();
438  cur_record = seq_record;
439 }

441 void AcquisitionWidget::setAcquiringToLastRecord() {
442  QSqlQuery query(QSqlDatabase::database(path));
443  query.exec("update acquisition_sequence set acquiring = ''");
444  if (!query.exec("select record from acquisition_sequence "
445  "where record >= 0 order by record desc limit 1")) {
446  QLOG_WARN() << "AcquisitionWidget::setAcquiringToLastRecord"
447  << query.lastError().text();
448  InitConfig();
449  return;
450  }
451  if (query.next()) {
452  const int lastRecord = query.value(0).toInt();
453  QSqlQuery update(QSqlDatabase::database(path));
454  update.prepare("update acquisition_sequence set acquiring = '*' where record = ?");
455  update.addBindValue(lastRecord);
456  update.exec();
457  }
458  InitConfig();
459 }
