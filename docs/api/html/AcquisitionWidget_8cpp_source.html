173void AcquisitionWidget::setDelegates() {
174 // Populate types
175 QStringList *typeList = new QStringList();
176 typeList->append("");
177 typeList->append("MOTOR");
178 typeList->append("DAC");
179 typeList->append("COMEDIDAC");
180 typeList->append("COMEDICOUNTER");
181 typeList->append("RASPIDAC");
182 typeList->append("SUPERK");
183 typeList->append("CAMERA");
184 typeList->append("SLM");
185 typeList->append("FILE");
186 typeList->append("TREATMENT");
187 ComboBoxDelegate *typeCombo = new ComboBoxDelegate(this, typeList);
188
189 // Populate instrument list from motor database
190 QString motordb = Utils::BuildDbPath(appDirPath, "motor.db3");
191 QSqlQuery querymotor(QSqlDatabase::database(motordb));
192 QStringList *instrumentList = new QStringList();
193 instrumentList->append("");
194 querymotor.exec("select * from motor_actuator");
195 while (querymotor.next()) {
196 instrumentList->append(querymotor.value(0).toString());
197 }
198
199 // Populate instrument list from dac database
200 QString dacdb = Utils::BuildDbPath(appDirPath, "dac.db3");
201 QSqlQuery querydac(QSqlDatabase::database(dacdb));
202 querydac.exec("select name from dac_settings");
203 while (querydac.next()) {
204 instrumentList->append(querydac.value(0).toString());
205 }
206
207 // Populate instrument list from comedi counter database
208 QString comedicounterdb = Utils::BuildDbPath(appDirPath, "comedicounter.db3");
209 QSqlQuery querycomedicounter(QSqlDatabase::database(comedicounterdb));
210 querycomedicounter.exec("select name from comedi_settings");
211 while (querycomedicounter.next()) {
212 instrumentList->append(querycomedicounter.value(0).toString());
213 }
214
215 // Populate instrument list from comedi dac database
216 QString comedidacdb = Utils::BuildDbPath(appDirPath, "comedidac.db3");
217 QSqlQuery querycomedidac(QSqlDatabase::database(comedidacdb));
218 querycomedidac.exec("select name from comedi_settings");
219 while (querycomedidac.next()) {
220 instrumentList->append(querycomedidac.value(0).toString());
221 }
222
223 // Populate instrument list from superk database
224 QString superkdb = Utils::BuildDbPath(appDirPath, "superk.db3");
225 QSqlQuery querysuperk(QSqlDatabase::database(superkdb));
226 querysuperk.exec("select name from superk_driver");
227 while (querysuperk.next()) {
228 instrumentList->append(querysuperk.value(0).toString());
229 }
230
231 // Populate instrument list from cameraList
232 for (int i = 0; i < cameraList.size(); i++) {
233 Camera* camera = cameraList.at(i);
234 instrumentList->append(camera->model);
235 }
236 ComboBoxDelegate *instrumentCombo = new ComboBoxDelegate(this, instrumentList);
237
238 acquisitionview->setItemDelegateForColumn(1, typeCombo);
239 acquisitionview->setItemDelegateForColumn(2, instrumentCombo);
240
241}

285void AcquisitionWidget::run() {
286 this->stop();
287 QSqlQuery query(QSqlDatabase::database(path));
288 query.exec("update acquisition_sequence set status = ''");
289 query.exec("update acquisition_sequence set acquiring = ''");
290 query.exec("select record,type,instrument,settings,scanplan from acquisition_sequence "
291 "order by record");
292
293 // Store acquisition sequence
294 sequenceList.clear();
295 cur_record = 0;
296 while (query.next()) {
297 int seq_record = query.value(0).toInt();
298 if (seq_record < 0) {
299 continue;
300 }
301 AcquisitionSequence *sequence = new AcquisitionSequence();
302 sequence->record = cur_record++;
303 sequence->seq_record = seq_record;
304 sequence->instrumentType = query.value(1).toString();
305 sequence->instrumentName = query.value(2).toString();
306 sequence->settings = query.value(3).toString();
307 sequence->scanplan = query.value(4).toString();
308 sequenceList.append(sequence);
309 }
310
311 // Prepare sequence
312 for (int i = 0; i < sequenceList.size(); i++) {
313 AcquisitionSequence *sequence = sequenceList.at(i);
314 sequence->prepare();
315 }
316 // Set File name and number from widget
317 QString filepath = acqfile + "_" + filenumber + ".h5";
318 acquisition->setFile(filepath, filenumber.toInt());
319 acquisition->setCamera(cameraList);
320 acquisition->setMotor(motor);
321 acquisition->setSuperK(superk);
322 acquisition->setDac(dac);
323 acquisition->setComediCounter(comedicounter);
324 acquisition->setComediDac(comedidac);
325 acquisition->setRaspiDac(raspidac);
326 acquisition->setSequenceList(sequenceList);
327 acquisition->start();
328 QLOG_DEBUG() << "AcquisitionWidget::run started";
329}

391void AcquisitionWidget::getAcquiring(int record) {
392 QSqlQuery query(QSqlDatabase::database(path));
393
394 query.exec("update acquisition_sequence set acquiring = ''");
395 query.exec("select record from acquisition_sequence order by record");
396 QLOG_DEBUG() << query.lastError().text();
397 int seq_record = 0;
398 int count = 0;
399 while (query.next()) {
400 seq_record = query.value(0).toInt();
401 if (seq_record < 0) {
402 continue;
403 }
404 if (count == record) {
405 break;
406 }
407 count++;
408 }
409 QLOG_DEBUG() << seq_record;
410 query.prepare("update acquisition_sequence set acquiring = '*' where record = ?");
411 query.addBindValue(seq_record);
412 query.exec();
413 InitConfig();
414 cur_record = seq_record;
415}

424void AcquisitionWidget::dbConnexion() {
425 QLOG_DEBUG() << "AcquisitionWidget::dbConnexion";
426 path = Utils::BuildDbPath(appDirPath, "acquisition.db3");
427 QSqlDatabase db = Utils::ConnectSqliteDb(path,
428 "AcquisitionWidget::dbConnexion>");
429 if (!db.isOpen()) {
430 Utils::EmitWarning(this, __FUNCTION__,
431 db.lastError().text());
432 }
433 // Create acquisition tables
434 QSqlQuery query(QSqlDatabase::database(path));
435 Utils::ExecSql(
436 query,
437 "create table if not exists acquisition_sequence "
438 "(record int primary key, "
439 "type varchar(128), "
440 "instrument varchar(128), "
441 "settings varchar(255), "
442 "scanplan varchar(128), "
443 "status varchar(128), "
444 "acquiring varchar(64))",
445 "AcquisitionWidget::dbConnexion>");
446 Utils::ExecSql(query, "update acquisition_sequence set status = ''",
447 "AcquisitionWidget::dbConnexion>");
448 Utils::ExecSql(query, "update acquisition_sequence set acquiring = ''",
449 "AcquisitionWidget::dbConnexion>");
450}
