102 Motor::connectMotor(QString newactuator) {
103 
104  bool driverexist = false;
105  bool comexist = false;
106  QSqlDatabase db = connectDb(path);
107  QSqlQuery query(db);
108 
109  if (newactuator == "") return;
110 
111  //
112  // Get actuator data
113  //
114  QString actutype, actusettings, actudescription;
115  float actuposition = 0;
116  query.prepare("select settings,description,position from motor_actuator where name = ?");
117  query.addBindValue(newactuator);
118  query.exec();
119  while (query.next()) {
120  actutype = query.value(0).toString();
121  actusettings = query.value(0).toString();
122  actudescription = query.value(1).toString();
123  actuposition = query.value(2).toString().toFloat();
124  }
125  //
126  // Check if actuator already configured
127  //
128  for (int i = 0 ; i < actuator.size(); i++) {
129  if ( actuator.at(i) == newactuator ) {
130  QLOG_INFO () << newactuator << " already connected";
131  if ( connectSuccess.at(i) == true) {
132  emit getPosition(actuposition);
133  emit getDescription(actudescription);
134  return;
135  }
136  else {
137  QLOG_DEBUG () << newactuator << " try to reconnect...";
138  //
139  // Free Actuator related structures
140  //
141  actuatorCom.remove(i);
142  actuatorDriver.remove(i);
143  actuatorSettings.remove(i);
144  description.remove(i);
145  com.remove(i);
146  driver.remove(i);
147  actuator.remove(i);
148  connectSuccess.remove(i);
149  operationcomplete.remove(i);
150  position.remove(i);
151  break;
152  }
153  }
154  }
155  //
156  // First time actuator initialization
157  //
158 
159  //
160  // Check if driver already initialized for the new actuator
161  //
162  query.prepare("select type,settings,name from motor_driver where name = "
163  "(select driver from motor_actuator where name = ?)");
164  query.addBindValue(newactuator);
165  query.exec();
166  QString drvtype, drvsettings, newdriver;
167  while (query.next()) {
168  drvtype = query.value(0).toString();
169  drvsettings = query.value(1).toString();
170  newdriver = query.value(2).toString();
171  }
172  for (int i = 0 ; i < driver.size(); i++) {
173  if ( driver.at(i) == newdriver ) {
174  QLOG_INFO () << " Assign previously created driver " << driver.at(i)
175  << "to actuator " << newactuator;
176  driver.push_back(driver.at(i));
177  actuatorDriver.push_back(actuatorDriver.at(i));
178  driverexist = true;
179  break;
180  }
181  }
182  if ( driverexist == false ) {
183  //
184  // Check if communication already exist for actuator
185  //
186  query.prepare("select type,address,settings,name from motor_com where name = "
187  "(select com from motor_driver where name = "
188  "(select driver from motor_actuator where name = ?))");
189  query.addBindValue(newactuator);
190  query.exec();
191  QString comtype, comaddress, comsettings, newcom;
192  while (query.next()) {
193  comtype = query.value(0).toString();
194  comaddress = query.value(1).toString();
195  comsettings = query.value(2).toString();
196  newcom = query.value(3).toString();
197  }
198  for (int j = 0 ; j < com.size(); j++) {
199  if (com.at(j) == newcom ) {
200  QLOG_INFO () << " Assign previously created communication " << com.at(j)
201  << "to driver ";
202  com.push_back(com.at(j));
203  actuatorCom.push_back(actuatorCom.at(j));
204  comexist = true;
205  break;
206  }
207  }
208  if ( comexist == false ) {
209  //
210  // Create new com
211  //
212  QLOG_INFO() << " Create new Communication channel at " << comaddress;
213 
214  ACCom *accom = ACCom::Create(comtype.toStdString(),
215  comaddress.toStdString(),
216  comsettings.toStdString());
217  if (!accom) {
218  Utils::EmitWarning(this, __FUNCTION__, "Could not create Communication channel");
219  return;
220  }
221  if (accom == (ACCom*)(-1))
222  QLOG_INFO() << "No Communication channel used";
223  else if (accom->Open()) {
224  Utils::EmitWarning(this, __FUNCTION__, "Could not open communication with device");
225  delete accom;
226  return;
227  }
228  com.push_back(newcom);
229  actuatorCom.push_back(accom);
230  }
231  //
232  // Create new driver
233  //
234  QLOG_INFO() << " Create new Driver with settings " << drvsettings;
235  Driver *drv = Driver::Create(drvtype.toStdString(),
236  drvsettings.toStdString(),
237  actuatorCom.at(actuatorCom.size() -1 ));
238  if (!drv) {
239  Utils::EmitWarning(this, __FUNCTION__, "Could not create Driver");
240  return;
241  }
242  else {
243  string driverState;
244  if (drv->Init(driverState)) {
245  Utils::EmitWarning(this, __FUNCTION__, "Could not init driver");
246  delete drv;
247  return;
248  }
249  }
250  actuatorDriver.push_back(drv);
251  driver.push_back(newdriver);
252  }
253  //
254  // Create new actuator
255  //
256  actuator.push_back(newactuator);
257  connectSuccess.push_back(true);
258  actuatorSettings.push_back(actusettings);
259  position.push_back(actuposition);
260  description.push_back(actudescription);
261  operationcomplete.push_back(0);
262  QLOG_INFO() << " Create new Actuator with settings " << actusettings;
263  if (actuatorDriver.at(actuator.size() - 1)->InitActuator(actusettings.toStdString(),actuposition)) {
264  Utils::EmitWarning(this, __FUNCTION__, "Could not init actuator");
265  connectSuccess.replace(actuator.size() - 1, false);
266  return;
267  }
268  emit getPosition(actuposition);
269  emit getDescription(actudescription);
270  return;

340 Motor::operationComplete() {
341 
342  for (int i = 0 ; i < actuator.size(); i++) {
343  if (actuator.at(i) == runningActuator ) {
344  string stateData;
345  DriverDefinition::ADLimitSwitch limitSwitch;
346  iscompleting = false;
347  if (connectSuccess.at(i) == true && iscompleting == false) {
348  int success = 0;
349  QString positionQString;
350  float curpos;
351  iscompleting = true;
352  success = actuatorDriver.at(i)->OperationComplete(stateData,
353  actuatorSettings.at(i).toStdString(),
354  limitSwitch);
355  actuatorDriver.at(i)->GetPos(actuatorSettings.at(i).toStdString(),curpos);
356  position.replace(i, curpos);
357  positionQString.setNum (position.at(i), 'f',3);
358  QLOG_DEBUG ( ) << "OperationComplete " << success << " position " << position.at(i);
359  emit getPosition(position.at(i));
360  if (success > 0 ) {
361  // Save last position in Db
362  actuatorDriver.at(i)->GetPos(actuatorSettings.at(i).toStdString(),curpos);
363  position.replace(i, curpos);
364  positionQString.setNum (position.at(i), 'f',3);
365  QLOG_DEBUG ( ) << "OperationComplete " << success << " position " << position.at(i);
366  emit getPosition(position.at(i));
367  QSqlDatabase db = connectDb(path);
368  QSqlQuery query(db);
369  query.prepare("update motor_actuator set position = ? "
370  "where name = ?");
371  query.addBindValue(positionQString);
372  query.addBindValue(actuator.at(i));
373  query.exec();
374  operationcomplete.replace(i, success);
375  QLOG_DEBUG ( ) << "OperationComplete " << success << " position " << position.at(i);
376  emit stopTimer();
377  }
378  iscompleting = false;
379  }
380  break;
381  }
