48bool Assistant::startAssistant() {
49 if (!proc) {
50 proc = new QProcess();
51 }
52
53 if (proc->state() != QProcess::Running) {
54 QString app = QLibraryInfo::location(QLibraryInfo::BinariesPath) + QDir::separator();
55#if !defined(Q_OS_MAC)
56 app += QLatin1String("assistant");
57#else
58 app += QLatin1String("Assistant.app/Contents/MacOS/Assistant");
59#endif
60 if (!QFile::exists(app)) {
61 const QString fallback = QStandardPaths::findExecutable("assistant");
62 const QString fallbackQt5 = QStandardPaths::findExecutable("assistant-qt5");
63 const QString fallbackQt = QStandardPaths::findExecutable("qtassistant");
64 if (!fallback.isEmpty()) {
65 app = fallback;
66 } else if (!fallbackQt5.isEmpty()) {
67 app = fallbackQt5;
68 } else if (!fallbackQt.isEmpty()) {
69 app = fallbackQt;
70 }
71 }
72 QStringList args;
73 args << QLatin1String("-collectionFile")
74 << appDirPath + QLatin1String("/docs/help/OpticsBenchUIColl.qhc")
75 << QLatin1String("-enableRemoteControl");
76 proc->start(app, args);
77 if (!proc->waitForStarted()) {
78 QMessageBox::critical(0, QObject::tr("OpticsBenchUI"),
79 QObject::tr("Unable to launch Qt Assistant (%1)").arg(app));
80 return false;
81 }
82 }
83 return true;
84}
