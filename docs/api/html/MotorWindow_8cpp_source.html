211void
212MotorWindow::remove(){
213 QModelIndexList selected;
214 QItemSelectionModel *selmodel;
215 selmodel = comview->selectionModel();
216 selected = selmodel->selectedIndexes();
217 for (int i = selected.size() - 1 ; i >= 0; i--)
218 comtable->removeRow(selected.at(i).row());
219 comtable->submitAll();
220 selmodel = driverview->selectionModel();
221 selected = selmodel->selectedIndexes();
222 for (int i = selected.size() - 1 ; i >= 0; i--)
223 drivertable->removeRow(selected.at(i).row());
224 drivertable->submitAll();
225 selmodel = actuatorview->selectionModel();
226 selected = selmodel->selectedIndexes();
227 for (int i = selected.size() - 1 ; i >= 0; i--)
228 actuatortable->removeRow(selected.at(i).row());
229 actuatortable->submitAll();
230 InitConfig();
231}
232void
233MotorWindow::InitRun(){
234 actuatorList->clear();
235 QSqlQuery query(QSqlDatabase::database(dbPath));
236 query.exec("select name from motor_actuator");
237 while (query.next()) {
238 QString name = query.value(0).toString();
239 actuatorList->append(name);
240 }
241 motorWidget->setActuatorList(actuatorList);
242}
