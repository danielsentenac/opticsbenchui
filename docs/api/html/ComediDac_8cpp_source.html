99 ComediDac::connectComedi(QString newcomedi) {
100 
101  int index;
102  QLOG_INFO () << "NEW COMEDI:" << newcomedi ;
103  if (newcomedi == "") return false;
104 
105  QLOG_INFO() << "ComediDac::connectComedi> connecting DAC " << newcomedi;
106  QLOG_INFO() << "PATH:" << path;
107  QSqlDatabase db = connectDb(path);
108  QSqlQuery query(db);
109  QString description;
110  QString comedivaluesString = "";
111  QString comediSettings = "";
112 
113  query.prepare("select settings,description,comedivalues from comedi_settings where name = ?");
114  query.addBindValue(newcomedi);
115  QLOG_INFO() << query.lastQuery();
116  query.exec();
117  QLOG_INFO ( ) << query.lastError().text();
118 
119  while (query.next()) {
120  comediSettings = query.value(0).toString();
121  description = query.value(1).toString();
122  comedivaluesString = query.value(2).toString();
123  }
124  QLOG_INFO () << "comediSettings:" << comediSettings;
125 
126  if (comediSettings == "") {
127  Utils::EmitWarning(this, __FUNCTION__, tr("%1 does not exist").arg(newcomedi));
128  return false;
129  }
130 
131  //
132  // Check if comedi already configured
133  //
134  for (int i = 0 ; i < comedi.size(); i++) {
135  if ( comedi.at(i) == newcomedi ) {
136  QLOG_INFO () << newcomedi << " already connected";
137  if ( connectSuccess.at(i) == true )
138  return true;
139  else {
140  QLOG_INFO () << newcomedi << " try to reconnect...";
141  if (comedivalues.at(i)) delete comedivalues.at(i);
142  comedivalues.remove(i);
143  comediSettings.remove(i);
144  comedi.remove(i);
145  subdev.remove(i);
146  mode.remove(i);
147  min.remove(i);
148  max.remove(i);
149  outputs.remove(i);
150  fname.remove(i);
151  connectSuccess.remove(i);
152  break;
153  }
154  }
155  }
156  // First time ComediDac connection
157 
158  comedi.push_back(newcomedi);
159  // Assign index
160  index = comedi.size() - 1;
161  QStringList settings;
162  QStringList subsettings;
163  settings = comediSettings.split(" ");
164  subsettings = settings.at(0).split("=");
165  fname.push_back(subsettings.at(1));
166  subsettings = settings.at(1).split("=");
167  mode.push_back(subsettings.at(1));
168  subsettings = settings.at(2).split("=");
169  subdev.push_back(subsettings.at(1).toInt());
170  subsettings = settings.at(3).split("=");
171  outputs.push_back(subsettings.at(1).toInt());
172  subsettings = settings.at(4).split("=");
173  min.push_back(subsettings.at(1).toFloat());
174  subsettings = settings.at(5).split("=");
175  max.push_back(subsettings.at(1).toFloat());
176  connectSuccess.push_back(false);
177  comedivalues.push_back(new QVector<double>());
178 
179  QLOG_INFO() << "ComediDac::connectComedi> mode : " << mode.at(index);
180  QLOG_INFO() << "ComediDac::connectComedi> subdev : " << subdev.at(index);
181  QLOG_INFO() << "ComediDac::connectComedi> outputs : " << outputs.at(index);
182  QLOG_INFO() << "ComediDac::connectComedi> max : " << max.at(index);
183  QLOG_INFO() << "DaComedi::connectComedi> min : " << min.at(index);
184  QLOG_INFO() << "DaComedi::connectComedi> values : " << comedivaluesString;
185  // Open device
186  QLOG_INFO() << "ComediDac::connectComedi> open device "
187  << fname.at(index) << "(index = " << index << ")";
188  comedi_t *newdevice;
189  newdevice = comedi_open(fname.at(index).toStdString().c_str());
190  if (newdevice == NULL) {
191  Utils::EmitWarning(this, __FUNCTION__, tr("comedi_open:%1 (%2)").arg(fname.at(index).toStdString().c_str(),
192  comedi_strerror(comedi_errno())));
193  connectSuccess.replace(index, false);
194  return false;
195  }
196  device.push_back(newdevice);
197  comedi_set_global_oor_behavior(COMEDI_OOR_NAN);
198  for (int output = 0 ; output < outputs.at(index); output++) {
199  comedi_range *rng = comedi_get_range(device.at(index), subdev.at(index), output, 0);
200  rng->min = min.at(index);
201  rng->max = max.at(index);
202  }
203  QLOG_INFO ( ) << "ComediDac configuration success";
204  connectSuccess.replace(index, true);
205  // Update control widget
206  emit getOutputs(outputs.at(index),mode.at(index));
207  emit getDescription(description);
208  // Get registered values from Db
209  QStringList comedivaluesStringList;
210  comedivaluesStringList = comedivaluesString.split(" ",QString::SkipEmptyParts);
211  for (int i = 0 ; i < comedivaluesStringList.size(); i++ )
212  comedivalues.at(index)->push_back(comedivaluesStringList.at(i).toFloat());
213  emit getOutputValues((void*)comedivalues.at(index));
214 
215  // Reset and configure comedi device
216  connectSuccess.replace(index,resetComedi(newcomedi));
217 
218  return connectSuccess.at(index);
219 }

248 ComediDac::setComediValue(QString newcomedi, int output, void *value) {
249 
250  lsampl_t data;
251  int ret;
252  comedi_range *rng;
253  int maxdata;
254 
255  // Convert value to expected type
256  double *dvalue;
257  dvalue = (double*) value;
258 
259  for (int index = 0 ; index < comedi.size(); index++) {
260  if ( comedi.at(index) == newcomedi && connectSuccess.at(index) == true) {
261  if ( *dvalue < min.at(index) || *dvalue > max.at(index)) {
262  Utils::EmitWarning(this, __FUNCTION__, tr("%1: channel %2:value out of range").arg(newcomedi,
263  QString::number(output)));
264  return false;
265  }
266  maxdata = comedi_get_maxdata(device.at(index), subdev.at(index), output);
267  rng=comedi_get_range(device.at(index), subdev.at(index), output, 0);
268  data = comedi_from_phys(*dvalue,rng,maxdata);
269  ret = comedi_data_write(device.at(index), subdev.at(index), output, 0, AREF_GROUND, data);
270  if (ret < 0) {
271  comedi_perror(comedi.at(index).toStdString().c_str());
272  return false;
273  }
274  QLOG_INFO() << "ComediDac::setComediValue> replace at " << output
275  << " old value " << QString::number(comedivalues.at(index)->at(output))
276  << " new value " << QString::number(*dvalue);
277  double real_value;
278  getComediValue(newcomedi, output, real_value);
279  comedivalues.at(index)->replace(output,real_value);
280  return true;
281  }
282  }
283  Utils::EmitWarning(this, __FUNCTION__, tr("Check connection to %1").arg(newcomedi));
284  return false;
