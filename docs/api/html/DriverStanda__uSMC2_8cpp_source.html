31//
32// Init implementation
33//
34int DriverStanda_uSMC2::Init(string& rstateData) const
35{
36 QLOG_DEBUG () << "DriverStanda_uSMC2::Init";
37 int retStatus = 0;
38 int cnt = 0;
39 char ximc_version_str[32];
40 char device_name[256];
41 int probe_flags = 0;
42
43 ximc_version( ximc_version_str );
44 QLOG_INFO () << ximc_version_str ;
45
46// Device enumeration function. Returns an opaque pointer to device enumeration data.
47 _devenum = enumerate_devices( probe_flags, 0 );
48// Gets device count from device enumeration data
49 cnt = get_device_count( _devenum );
50 QLOG_INFO () << "Found " << cnt << " devices";
51// Terminate if there are no connected devices
52 if (cnt <= 0) {
53 QLOG_WARN() << "No devices found";
54// Free memory used by device enumeration data
55 free_enumerate_devices( _devenum );
56 return -1;
57 }
58// get devices name into a string
59 for (int i = 0; i < cnt; i++) {
60 strcpy( device_name, get_device_name( _devenum, i ) );
61
62 QLOG_INFO () << device_name;
63 }
64
65 return retStatus;
66}

77 int axisNumber;

80
81 if (sscanf(actuatorSetting.c_str(),"axisNumber=%d speed=%d",&axisNumber,&speed) != NB_ITEM_INIT_SETTING)
82 {
83 ReportSettingError("Bad actuator setting");
84 }
85// Open device
86 device_t dev = open_device(get_device_name(_devenum,axisNumber-1));
87 if (dev != -1){
88 QLOG_INFO() << "Open device " << get_device_name(_devenum,axisNumber-1);
89 _device.insert(std::pair<int,device_t>(axisNumber,dev));
90 }
91 else {
92 QLOG_ERROR() << "Error opening device with axis number " << axisNumber;
93 return -1;
94 }
95// Set speed settings
96 retStatus = get_move_settings(_device.find(axisNumber)->second,&_move_settings);
97 _move_settings.Speed = speed;
98 retStatus = set_move_settings(_device.find(axisNumber)->second,&_move_settings);
99
100 return retStatus;
