24CameraWindow::CameraWindow( QMainWindow* parent, Qt::WindowFlags fl , Camera *_camera, int _cameraNumber)
25 : QMainWindow( parent, fl )
26{
27 camera = _camera;
28 cameraNumber = _cameraNumber;
29 parentWindow = parent;
30 OpticsBenchUIMain *OpticsBenchui = (OpticsBenchUIMain*) parentWindow;
31 OpticsBenchui->setOpenCameraWindow(true,cameraNumber);
32#ifndef NO_MULTIMEDIA
33 player = new VideoPlayer(this,camera);
34#else
35 videoPlaceholder = new QLabel(tr("Video disabled (NO_MULTIMEDIA)"));
36 videoPlaceholder->setAlignment(Qt::AlignCenter);
37 videoPlaceholder->setMinimumSize(320, 240);
38#endif
39 cameraPropWidget = nullptr;
40
41 if (camera->camera_err == 0) {
42#ifndef NO_MULTIMEDIA
43 connect(this,SIGNAL(setVideoPlayerResolution(int,int)),player,
44 SLOT(setVideoPlayerResolution(int,int)));
45
46 // Creation of menuBar
47 QMenuBar* menuBar = new QMenuBar(this);
48 menuBar->resize(15,15);
49 // Creating a menu "File"
50 QMenu* menuResolution = new QMenu("Resolution");
51
52 // And adding the menu "menuResolution" in the menu bar
53 menuBar->addMenu(menuResolution);
54
55 // Add a separator
56 menuResolution->addSeparator();
57
58 menuResolution->addAction("480x320", this, SLOT(set480x320()) );
59 menuResolution->addAction("640x480", this, SLOT(set640x480()) );
60 menuResolution->addAction("1280x960", this, SLOT(set1280x960()) );
61#endif
62
63 cameraWidget = new CameraControlWidget(camera);
64 QDockWidget *dockWidget = new QDockWidget(tr("Controls"), this);
65 dockWidget->setAllowedAreas(Qt::BottomDockWidgetArea);
66 QScrollArea *controlArea = new QScrollArea(this);
67 controlArea->setWidget(cameraWidget);
68 controlArea->setWidgetResizable(true);
69 dockWidget->setWidget(controlArea);
70 this->addDockWidget(Qt::BottomDockWidgetArea, dockWidget);
71
72 if ( camera->propList.size() > 0 ) {
73 cameraPropWidget = new CameraPropWidget(camera);
74 QDockWidget *dockWidget = new QDockWidget(tr("Properties"), this);
75 dockWidget->setAllowedAreas(Qt::RightDockWidgetArea);
76 QScrollArea *propArea = new QScrollArea(this);
77 propArea->setWidget(cameraPropWidget);
78 propArea->setWidgetResizable(true);
79 dockWidget->setWidget(propArea);
80 this->addDockWidget(Qt::RightDockWidgetArea, dockWidget);
81 }
82 this->setWindowTitle(QString(camera->vendor) + " / " +
83 QString(camera->model));
84#ifndef NO_MULTIMEDIA
85 this->setCentralWidget(player);
86 this->setSizePolicy(QSizePolicy(QSizePolicy::Expanding,QSizePolicy::Expanding));
87 this->set480x320();
88#else
89 this->setCentralWidget(videoPlaceholder);
90 const int width = (cameraPropWidget != nullptr) ? 480 + OFFSET_X : 480;
91 resize(QSize(width, 320 + OFFSET_Y));
92#endif
93 }
94 else
95 this->setWindowTitle(tr("No Camera found"));
96}

106}
