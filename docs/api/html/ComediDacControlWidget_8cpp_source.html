100 void ComediDacControlWidget::getOutputs(int outputs, QString mode) {

102 
103  delete signalMapper;

121  button->setFixedWidth(150);

129  shiftButtonList->push_back(shiftbutton);
130 
131  layout->addWidget(outputLabel, i + 2, 0, 1, 1);
132  layout->addWidget(comedivalue, i + 2, 1, 1, 1);
133  layout->addWidget(unitsLabel, i + 2, 2, 1, 1);
134  layout->addWidget(button, i + 2, 3, 1, 1);
135  layout->addWidget(comedirvalue, i + 2, 4, 1, 1);
136  layout->addWidget(shiftbutton, i + 2, 5, 1, 1);
137  }
