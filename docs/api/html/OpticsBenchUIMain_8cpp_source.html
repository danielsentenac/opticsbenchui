25template <typename CameraType>
26void setupCameraManager(Camera *&manager,
27 const QString &label,
28 QVector<Camera *> &cameraList,
29 QVector<bool> &isOpenCameraWindow,
30 QVector<CameraWindow *> &cameraWindowList)
31{
32 manager = new CameraType();
33 manager->findCamera();
34 QLOG_INFO() << "Found " << manager->num << " " << label << " camera";
35
36 for (int i = 0; i < manager->num; i++) {
37 Camera *camera = new CameraType();
38 camera->setCamera(manager->cameralist.at(i), i);
39 cameraList.push_back(camera);
40 isOpenCameraWindow.push_back(false);
41 cameraWindowList.push_back(NULL);
42 }
43}

45void addCameraMenuActions(Camera *manager,
46 QMenu *menu,
47 QSignalMapper *signalMapper,
48 int &cameraNumber)
49{
50 if (!manager) {
51 return;
52 }
53 for (int i = 0; i < manager->num; i++) {
54 const QString selectedCamera =
55 QString(manager->vendorlist.at(i)) + " / " +
56 QString(manager->modelist.at(i));
57 QAction *action = new QAction(selectedCamera, menu);
58 menu->addAction(action);
59 QObject::connect(action, SIGNAL(triggered()), signalMapper, SLOT(map()));
60 signalMapper->setMapping(action, cameraNumber++);
61 }
62}

451OpticsBenchUIMain::~OpticsBenchUIMain()
452{
453 delete acquisitionwidget;
454 delete assistant;
455 if (dacwindow) delete dacwindow;
456 if (comedidacwindow) delete comedidacwindow;
457 if (comedicounterwindow) delete comedicounterwindow;
458 if (motorwindow) delete motorwindow;
459 if (superkwindow) delete superkwindow;
460 for (int i = 0 ; i < cameraList.size() ; i++) {
461 if (isopencamerawindow.at(i) == true) {
462 QLOG_INFO() << " Exiting Camera Window " << i;
463 delete camerawindowList.at(i);
464 }
465 }
466 for (int i = 0 ; i < cameraList.size() ; i++) {
467 QLOG_INFO() << " Exiting Camera " << i;
468 delete cameraList.at(i);
469 }
470 QLOG_INFO() << "OpticsBenchUI ended";
471}
