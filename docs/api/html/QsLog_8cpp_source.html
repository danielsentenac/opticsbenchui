51static const char* LevelToText(Level theLevel)
52{
53 switch( theLevel )
54 {
55 case TraceLevel:
56 return TraceString;
57 case DebugLevel:
58 return DebugString;
59 case InfoLevel:
60 return InfoString;
61 case WarnLevel:
62 return WarnString;
63 case ErrorLevel:
64 return ErrorString;
65 case FatalLevel:
66 return FatalString;
67 default:
68 {
69 assert(!"bad log level");
70 return InfoString;
71 }
72 }
73}

120void Logger::Helper::writeToLog()
121{
122 const char* const levelName = LevelToText(level);
123 const QString completeMessage(QString("%1 %2 %3")
124 .arg(levelName, 5)
125 .arg(QDateTime::currentDateTime().toString(fmtDateTime))
126 .arg(buffer)
127 );
128
129 Logger& logger = Logger::instance();
130 QMutexLocker lock(&logger.d->logMutex);
131 logger.write(completeMessage);
132}

150void Logger::write(const QString& message)
151{
152 for(DestinationList::iterator it = d->destList.begin(),
153 endIt = d->destList.end();it != endIt;++it)
154 {
155 if( !(*it) )
156 {
157 assert(!"null log destination");
158 continue;
159 }
160 (*it)->write(message);
161 }
162}
