142 void
143 CameraIEEE1394::setCamera(void* _camera, int _id)
144 {
145  camera = (dc1394camera_t*)_camera;
146  QLOG_DEBUG() << "CameraIEEE1394::setCamera> camera pointer " << camera;
147  vendor = camera->vendor;
148  model = camera->model;
149 
150  /* Init camera*/
151  err = dc1394_reset_bus (camera);
152  /*DC1394_ERR_CLN_RTN(err,cleanup_and_exit(),"reset bus reported error");
153 */
154  err = dc1394_feature_get_all(camera,&features);
155  if (err != DC1394_SUCCESS) {
156  showWarning("Could not get feature set");
157  return;
158  }
159  else {
160  QLOG_DEBUG() << dc1394_feature_print_all(&features, stdout);
161  }
162  int featureCnt = 0;
163  for (int i = 0 ; i < DC1394_FEATURE_NUM; i++) {
164  if (features.feature[i].available == DC1394_TRUE &&
165  features.feature[i].id != DC1394_FEATURE_TRIGGER) {
166  featureNameList.push_back(iidc_features[i]);
167  featureIdList.push_back(i);
168  featureMinList.push_back(features.feature[i].min);
169  featureMaxList.push_back(features.feature[i].max);
170  featureValueList.push_back(features.feature[i].value);
171  featureAbsCapableList.push_back(features.feature[i].absolute_capable);
172  featureAbsValueList.push_back(features.feature[i].abs_value);
173  if (features.feature[i].current_mode == DC1394_FEATURE_MODE_AUTO)
174  featureModeAutoList.push_back(true);
175  else
176  featureModeAutoList.push_back(false);
177  QLOG_INFO() << "CameraIEEE1394::setCamera Found feature " <<
178  featureNameList.at(featureCnt++);
179  }
180  }
181  // Video mode feature
182  featureNameList.push_back("VIDEO MODE");
183  featureIdList.push_back(DC1394_FEATURE_NUM + 1);
184  video_mode_feature = featureCnt++;
185  // get video modes:
186  err = dc1394_video_get_supported_modes(camera,&video_modes);
187  if (err != DC1394_SUCCESS) {
188  showWarning("Can't get video modes");
189  return;
190  }
191  featureMinList.push_back(0);
192  featureMaxList.push_back(video_modes.num - 1);
193  featureValueList.push_back(0);
194  featureAbsCapableList.push_back(false);
195  featureAbsValueList.push_back(0);
196  featureModeAutoList.push_back(false);
197 
198  /* select highest video mode*/
199  unsigned int tmpwidth = 0, tmpheight = 0;
200  dc1394video_mode_t tmpvideo_mode;
201  dc1394color_coding_t tmpencoding_num;
202  for ( int i = 0 ; i < video_modes.num ; i++) {
203  tmpvideo_mode = video_modes.modes[i];
204  dc1394_get_image_size_from_video_mode(camera, tmpvideo_mode,
205  &tmpwidth, &tmpheight);
206  dc1394_get_color_coding_from_video_mode(camera,
207  tmpvideo_mode,
208  &tmpencoding_num);
209  // check for highest resolution
210  QLOG_INFO () << "CameraIEEE1394::setCamera> Available IIDC video mode "
211  << tmpvideo_mode
212  << " color coding " << tmpencoding_num
213  << " width " << tmpwidth << " height " << tmpheight;
214  if ( width * height < tmpwidth * tmpheight ) {
215  width = tmpwidth;
216  height = tmpheight;
217  encoding_num = tmpencoding_num;
218  video_mode = tmpvideo_mode;
219  featureValueList.replace(video_mode_feature,i);
220  }
221  }
222  QLOG_INFO() << "CameraIEEE1394::setCamera> Found feature "
223  << featureNameList.at(video_mode_feature) << " set to "
224  << video_modes.modes[(int)featureValueList.at(video_mode_feature)];
225  uint32_t data_depth;
226  dc1394_get_color_coding_data_depth(encoding_num,&data_depth);
227  QLOG_INFO() << "CameraIEEE1394::setCamera> Pixel data depth " << data_depth;
228  /* select highest framerate */
229  if ( !dc1394_is_video_mode_scalable(video_mode) ) {
230  err = dc1394_get_color_coding_from_video_mode(camera,
231  video_mode,
232  &encoding_num);
233  dc1394framerates_t framerates;
234  err = dc1394_video_get_supported_framerates(camera,
235  video_mode,
236  &framerates);
237  framerate = (dc1394framerate_t)0;
238  for (int i = 0 ; i < DC1394_FRAMERATE_NUM; i++) {
239  QLOG_INFO () << "CameraIEEE1394::setCamera>> supported framerate "
240  << framerates.framerates[i];
241  if (framerate < framerates.framerates[i] && framerates.framerates[i] < DC1394_FRAMERATE_MAX) {
242  QLOG_INFO () << "CameraIEEE1394::setCamera>> selected highest framerate "
243  << framerate;
244  framerate = framerates.framerates[i];
245  }
246  }
247  QLOG_INFO () << "CameraIEEE1394::setCamera>> selected highest framerate "
248  << framerate;
249  err = dc1394_video_set_framerate(camera, framerate);
250  }
251  else {
252  framerate = (dc1394framerate_t)40;
253  dc1394_format7_get_color_coding(camera, video_mode, &encoding_num);
254  }
255  dc1394_get_color_coding_data_depth(encoding_num,&data_depth);
256  QLOG_INFO() << "CameraIEEE1394::setCamera> Pixel data depth " << data_depth;
257 
258  switch (encoding_num) {
259  case DC1394_COLOR_CODING_MONO8:
260  pixel_encoding = B8;
261  break;
262  case DC1394_COLOR_CODING_MONO16:
263  pixel_encoding = B16;
264  break;
265  default:
266  QLOG_INFO() << "ENCODING NUMBER " << encoding_num;
267  pixel_encoding = UNKNOWN;
268  break;
269  }
270 
271  // Color Coding feature
272  featureNameList.push_back("COLOR CODING");
273  featureIdList.push_back(DC1394_FEATURE_NUM + 2);
274  color_coding_feature = featureCnt++;
275 
276  featureMinList.push_back(0);
277  featureMaxList.push_back(1);
278 
279  switch (encoding_num) {
280  case DC1394_COLOR_CODING_MONO8:
281  featureValueList.push_back(0);
282  break;
283  case DC1394_COLOR_CODING_MONO16:
284  featureValueList.push_back(1);
285  break;
286  default:
287  featureValueList.push_back(0);
288  break;
289  }
290  featureAbsCapableList.push_back(false);
291  featureAbsValueList.push_back(0);
292  featureModeAutoList.push_back(false);
293 
294  // Properties
295  int propCnt = 0;
296 
297  // Frame Rate prop
298  QString frateStr = iidc_props[propCnt];
299  QLOG_INFO () << "CameraIEEE1394::setCamera> Frame Rate property added ";
300  propList.push_back(frateStr);
301 
302  // PC Rate prop
303  QString pcrateStr = iidc_props[++propCnt];
304  QLOG_INFO () << "CameraIEEE1394::setCamera> PC Rate property added ";
305  propList.push_back(pcrateStr);
306 
307  // Image Size prop
308  QString imgsizeStr = iidc_props[++propCnt];
309  QLOG_INFO () << "CameraIEEE1394::setCamera> Image Size property added ";
310  propList.push_back(imgsizeStr);
311 
312  // Pixel Encoding prop
313  QString encodingStr = iidc_props[++propCnt];
314  QLOG_INFO () << "CameraIEEE1394::setCamera> Pixel Encoding property added ";
315  propList.push_back(encodingStr);
316 
317  // Video Mode prop
318  QString videomodeStr = iidc_props[++propCnt];
319  QLOG_INFO () << "CameraIEEE1394::setCamera> Video Mode property added ";
320  propList.push_back(videomodeStr);
321 
322  id = _id;
323 
324  camera_err = connectCamera();
325 
326  QLOG_INFO() << "CameraIEEE1394::setCamera " << vendor << " model : "
327  << model << " - Err : " << camera_err;
328 }

366 void
367 CameraIEEE1394::run() {
368  int acq_err = 0;
369  int acq_cnt = 0;
370  if (camera_err == 0 && suspend == true ) {
371  suspend = false;
372  eTimeTotal = 0;
373  while (suspend == false) {
374  QLOG_DEBUG () << "CameraIEEE1394::run> " << id << " : start new Acquisition";
375  double eTime = Utils::GetTimeMicroseconds();
376  acqstart->wakeAll();
377  acq_err = acquireImage();
378  QLOG_DEBUG () << "CameraIEEE1394::run> " << id << " : done";
379  acqend->wakeAll();
380  if (acq_err == 1)
381  eTimeTotal+= (Utils::GetTimeMicroseconds() - eTime);
382  acq_cnt++;
383  if (acq_cnt == FREQUENCY_AVERAGE_COUNT) {
384  eTimeTotal/=1e6;
385  eTimeTotal/=FREQUENCY_AVERAGE_COUNT;
386  if ( eTimeTotal > 0 )
387  frequency = 1.0 / eTimeTotal;
388  else
389  frequency = 0;
390  QLOG_DEBUG() << "CameraIEEE1394::run> Acquisition "
391  << "freq " << (int) frequency << " Hz";
392  eTimeTotal = 0;
393  acq_cnt = 0;
394  // Update props
395  getProps();
396  }
397  has_started = true;
398  }
399  QLOG_DEBUG() << "CameraNeo thread exiting";
400  }

579 void
580 CameraIEEE1394::getFeatures() {
581  /*-----------------------------------------------------------------------
582  * report camera's features
583  *-----------------------------------------------------------------------*/
584  err = dc1394_feature_get_all(camera,&features);
585  if (err != DC1394_SUCCESS) {
586  showWarning("Could not get feature set");
587  }
588  else {
589  QLOG_DEBUG() << dc1394_feature_print_all(&features, stdout);
590  }
591  for (int i = 0 ; i < DC1394_FEATURE_NUM; i++) {
592  if (features.feature[i].available == DC1394_TRUE &&
593  features.feature[i].id != DC1394_FEATURE_TRIGGER) {

792 int
793 CameraIEEE1394::acquireImage() {
794 
795  acquireMutex->lock();
796  /*Init frame to nullptr*/
797  frame = nullptr;
798  /*-----------------------------------------------------------------------
799  * capture one frame
800  *-----------------------------------------------------------------------*/
801  err = dc1394_capture_dequeue(camera, DC1394_CAPTURE_POLICY_WAIT,
802  &frame);
803  DC1394_ERR_CLN_RTN(err,cleanup_and_exit(),
804  "acquireImage> Could not capture a frame");
805  QLOG_DEBUG() << "CameraIEEE1394::acquireImage> Frame image size " << frame->image_bytes;
806  /* copy captured image */
807  snapshotMutex->lock();
808  ushort *tmpBuf;
809  // calculate min,max
810  max = 0;
811  min = 65535;
812  if ( pixel_encoding == B8 ) {
813  for (unsigned int i = width ; i < width * height - width; i++) { // avoid first/last line to calculate min/max
814  if (frame->image[i] > max)
815  max = frame->image[i];
816  if (frame->image[i] < min)
817  min = frame->image[i];
818  }
819  }
820  if ( pixel_encoding == B16 ) {
821  tmpBuf = reinterpret_cast<ushort*>(frame->image);
822  for (unsigned int i = width ; i < width * height - width; i++) { // avoid first line to calculate min/max
823  ushort ui_current = *(tmpBuf++);
824  if (ui_current > max)
825  max = ui_current;
826  if (ui_current < min)
827  min = ui_current;
828  }
829  }
830  // copy image to buffer (8-bit : 0 - 255 range) and to buffer32 (32-bit)
831  // Reset buffer (8-bit) and buffer32 (32-bit)
832  memset(buffer,0,height * width * sizeof(uchar));
833  memset(buffer32,0,height * width * sizeof(int));
834  if ( pixel_encoding == B8 ) {
835  memcpy(buffer,frame->image,width * height * sizeof(uchar));
836  for (unsigned int i = 0 ; i < width * height; i++)
837  buffer32[i] = buffer[i];
838  }
839  if ( pixel_encoding == B16 ) {
840  tmpBuf = reinterpret_cast<ushort*>(frame->image);
841  for (unsigned int i = 0 ; i < width * height; i++) {
842  ushort ui_current = *(tmpBuf++);
843  buffer32[i] = ui_current;
844  if ( (max - min) != 0 )
845  buffer[i] = (uchar) (( 255 * (ui_current - min) ) / (max - min));
846  else
847  buffer[i] = 255;
848  }
849  }
850  if (vflip) {
851  buffer = reversebytes(buffer,height * width);
852  buffer = fliphorizontal(buffer,height * width, width);
853  buffer32 = reversebytes(buffer32,height * width);
854  buffer32 = fliphorizontal(buffer32,height * width, width);
855  }
856  if (hflip) {
857  buffer = fliphorizontal(buffer,height*width, width);
858  buffer32 = fliphorizontal(buffer32,height*width, width);
859  }
860  snapshotMutex->unlock();
861  // emit visualisation image
862  QImage imagescaled = image->scaled(imageWidth,imageHeight);
863  QImage imagergb32 = imagescaled.convertToFormat(QImage::Format_ARGB32_Premultiplied);
864 
865  emit getImage(imagergb32);
866  emit updateMin(min);
867  emit updateMax(max);
868  /*-----------------------------------------------------------------------
869  * release frame
870  *-----------------------------------------------------------------------*/
871  err = dc1394_capture_enqueue(camera, frame);
872  DC1394_ERR_CLN_RTN(err,cleanup_and_exit(),
873  "acquireImage> Could not enqueue a frame");
874  acquireMutex->unlock();
875  return (1);
876 }
