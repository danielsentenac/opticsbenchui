58 }
59 else
60 {
61 QLOG_INFO() << "Controller :" << answer.c_str();
62 }
63 return retStatus;

77int DriverPI_E725::InitActuator(string actuatorSetting,
78 float position) const
79{
80 QLOG_DEBUG() << "DriverPI_E725::InitActuator";
81
82 int retStatus = 0;
83 string commandLine, answer;
84 char axisNumber[8];
85
86 if (sscanf(actuatorSetting.c_str(),"axisNumber=%s",(char*)&axisNumber) != NB_ITEM_INIT_SETTING)
87 {
88 ReportSettingError("DriverPI_E725::InitActuator> Bad actuator setting");
89 }
90 usleep(_delay);
91 commandLine = "\nSVO " + string(axisNumber) + " 1\n";
92
93 if (_pcommChannel->Write(commandLine,NULL) < (int)commandLine.length())
94 {
95 ReportCommError(QString("DriverPI_E725::InitActuator> Cannot put %1 in ServoMode").arg(axisNumber));
96 retStatus = -1;
97 }
98 GetPos(actuatorSetting,position);
99 QLOG_INFO () << " Actuator " << axisNumber << " position " << QString::number(position);
100
101 return retStatus;
102}

113int DriverPI_E725::GetPos(string actuatorSetting,
114 float& position) const
115{
116 QLOG_DEBUG() << "DriverPI_E725::GetPos";
117
118 int retStatus = 0;
119 string answer;
120 string commandLine;
121 char axisNumber[8];
122 int _delay;
123
124 if (sscanf(actuatorSetting.c_str(),"axisNumber=%s",(char*)&axisNumber) != NB_ITEM_INIT_SETTING)
125 {
126 ReportSettingError("DriverPI_E725::GetPos> Bad actuator setting");
127 }
128 usleep(_delay);
129 commandLine = "\nPOS? " + string(axisNumber) + "\n";
130
131 if (_pcommChannel->Write(commandLine,NULL) < (int)commandLine.length())
132 {
133 ReportCommError("DriverPI_E725::GetPos> Unable to write to port");
134 retStatus = -1;
135 }
136 if (!_pcommChannel->Read(answer,NULL))
137 {
138 ReportCommError("DriverPI_E725::GetPos> Unable to read from port");
139 retStatus = -1;
140 }
141 else
142 {
143 QLOG_DEBUG() << "Position reply " << answer.c_str();
144 vector<string> tokens;
145 Tokenize(answer,tokens,"=");
146 if (tokens.size() > 0 && tokens.at(0) != axisNumber) {
147 ReportWarning(OTHER_ERROR,
148 QString("DriverPI_E725::GetPos> Position reply expected, but got reply = %1")
149 .arg(answer.c_str()));
150 retStatus = -1;
151 return retStatus;
152 }
153 QString positionStr = QString::fromStdString(tokens.at(1));
154 QLOG_DEBUG () << " token Position" << positionStr;
155 position = positionStr.toFloat();
156 QLOG_DEBUG () << " Position " << QString::number(position);
157 }
158 return retStatus;
159}

314 retStatus = -1;
315 }
316 else

318 QLOG_DEBUG () << "OperationComplete> " << answer.c_str();
319 if (answer.find("0") == string::npos) // motion not completed
320 retStatus = 0;
321 else // motion completed
322 retStatus = 1;
323 }
