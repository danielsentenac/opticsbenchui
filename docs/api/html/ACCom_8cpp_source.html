56ACCom *ACCom::Create (string channelType, string device, string settings)
57{
58 ACCom *channel = NULL;
59
60 if (channelType == RS_COM)
61 {
62 channel = new ACRSCom (device, settings);
63 }
64 else if (channelType == ETH_COM)
65 {
66 channel = new ACEthCom (device, settings);
67 }
68 else if (channelType == USB_COM)
69 {
70 channel = new ACUsbCom (device, settings);
71 }
72 else if (channelType == NULL_COM)
73 {
74 channel = (ACCom*)(-1);
75 }
76 else
77 {
78 Utils::ReportWarning(NULL, QString("ACCom::Create> Unknown channel type %1")
79 .arg(QString(channelType.c_str())));
80 }
81
82 return channel;
83}
