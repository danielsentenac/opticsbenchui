154 void AnalysisThread::run() {
155  {
156  QMutexLocker locker(mutex);
157  if (!suspend) {
158  return;
159  }
160  suspend = false;
161  }
162 
163  for (const AnalysisTask& task : tasks) {
164  {
165  QMutexLocker locker(mutex);
166  if (suspend) {
167  break;
168  }
169  }
170  if (task.codePath.trimmed().isEmpty()) {
171  emit showWarning("Analysis task has empty code path; skipping.");
172  continue;
173  }
174 
175  emit analysisStarted(task.record);
176  QLOG_INFO() << "AnalysisThread::run> Starting task " << task.record
177  << " cmd=" << task.codePath << " args=" << task.arguments;
178 
179  QString program = task.codePath.trimmed();
180  QStringList arguments = splitArguments(task.arguments);
181  bool runInLoginShell = false;
182  if (program.contains(' ')) {
183  const QStringList commandParts = splitArguments(program);
184  if (!commandParts.isEmpty()) {
185  program = commandParts.first();
186  QStringList extraArgs = commandParts;
187  extraArgs.removeFirst();
188  if (!extraArgs.isEmpty()) {
189  arguments = extraArgs + arguments;
190  }
191  }
192  }
193 
194  // Run Python scripts through an explicit interpreter so dependencies
195  // resolve from the expected Python environment.
196  if (IsPythonScriptPath(program)) {
197  runInLoginShell = true;
198  const QString pythonInterpreter = ResolvePythonInterpreter();
199  if (!pythonInterpreter.isEmpty()) {
200  arguments.prepend(program);
201 #ifdef Q_OS_WIN
202  const QString pyName =
203  QFileInfo(pythonInterpreter).fileName().toLower();
204  if (pyName == "py" || pyName == "py.exe") {
205  arguments.prepend("-3");
206  }
207 #endif
208  program = pythonInterpreter;
209  } else {
210  emit showWarning(
211  "No Python interpreter found in PATH. Trying direct script run.");
212  }
213  }
214 
215  QString launchProgram = program;
216  QStringList launchArguments = arguments;
217 #ifdef Q_OS_UNIX
218  if (runInLoginShell) {
219  const QString bash = ResolveBashExecutable();
220  if (!bash.isEmpty()) {
221  const QString command = BuildShellCommand(program, arguments);
222  launchProgram = bash;
223  launchArguments =
224  QStringList() << "-lc"
225  << QString("if [ -f \"$HOME/.bashrc\" ]; then "
226  ". \"$HOME/.bashrc\" >/dev/null 2>&1; "
227  "fi; %1")
228  .arg(command);
229  } else {
230  emit showWarning("bash not found; running analysis without shell init.");
231  }
232  }
233 #endif
234 
235  QProcess localProcess;
236  localProcess.setProcessChannelMode(QProcess::MergedChannels);
237  localProcess.setProcessEnvironment(QProcessEnvironment::systemEnvironment());
238  localProcess.start(launchProgram, launchArguments);
239 
240  if (!localProcess.waitForStarted()) {
241  const QString output =
242  "Unable to start command.\n"
243  "Command: " + launchProgram + "\n"
244  "Arguments: " + launchArguments.join(' ') + "\n"
245  "Error: " + localProcess.errorString();
246  emit analysisFinished(task.record, false, output);
247  emit showWarning("Unable to start analysis command: " + task.codePath);
248  if (currentPid != 0) {
249  currentPid = 0;
250  emit pidChanged(0);
251  }
252  continue;
253  }
254  currentPid = static_cast<qint64>(localProcess.processId());
255 #ifdef Q_OS_UNIX
256  if (currentPid > 0) {
257  setpgid(static_cast<pid_t>(currentPid),
258  static_cast<pid_t>(currentPid));
259  }
260 #endif
261  emit pidChanged(currentPid);
262 
263  bool stopped = false;
264  QString outputBuffer;
265  while (!localProcess.waitForFinished(100)) {
266  const QString chunk = localProcess.readAllStandardOutput();
267  if (!chunk.isEmpty()) {
268  outputBuffer += chunk;
269  emit analysisOutput(task.record, chunk);
270  }
271  QMutexLocker locker(mutex);
272  if (suspend) {
273  stopped = true;
274  break;
275  }
276  }
277 
278  if (stopped) {
279  localProcess.terminate();
280  localProcess.waitForFinished(2000);
281  if (localProcess.state() != QProcess::NotRunning) {
282  localProcess.kill();
283  localProcess.waitForFinished(2000);
284  }
285 #ifdef Q_OS_UNIX
286  if (IsProcessAlive(currentPid)) {
287  KillProcessTree(currentPid, SIGKILL);
288  }
289 #endif
290  const QString chunk = localProcess.readAllStandardOutput();
291  if (!chunk.isEmpty()) {
292  outputBuffer += chunk;
293  emit analysisOutput(task.record, chunk);
294  }
295  emit analysisFinished(task.record, false, "Stopped by user.");
296  currentPid = 0;
297  emit pidChanged(0);
298  break;
299  }
300 
301  const QString remaining = localProcess.readAllStandardOutput();
302  if (!remaining.isEmpty()) {
303  outputBuffer += remaining;
304  emit analysisOutput(task.record, remaining);
305  }
306 
307  const bool success =
308  (localProcess.exitStatus() == QProcess::NormalExit &&
309  localProcess.exitCode() == 0);
310  const QString output =
311  "Exit code: " + QString::number(localProcess.exitCode());
312  emit analysisFinished(task.record, success, output);
313  currentPid = 0;
314  emit pidChanged(0);
315  }
316 
317  QMutexLocker locker(mutex);
318  suspend = true;
319  if (currentPid != 0) {
320  currentPid = 0;
321  emit pidChanged(0);
322  }
323  QLOG_DEBUG() << "AnalysisThread::run> End of thread";
324 }
