97 void DacControlWidget::getOutputs(int outputs, QString mode) {
98  clearOutputs();
99 
100  // Create a signal mapper for buttons
101  delete signalMapper;
102  delete shiftsignalMapper;
103  shiftsignalMapper = new QSignalMapper(this);
104  signalMapper = new QSignalMapper(this);
105  for (int i = 0; i < outputs; i++) {
106  QLabel *outputLabel = new QLabel(tr("Output %1").arg(i));
107  outputLabel->setFixedWidth(100);
108  outputsList->push_back(outputLabel);
109  QLineEdit *dacvalue = new QLineEdit("");
110  dacvalue->setFixedWidth(100);
111  dacvalueList->push_back(dacvalue);
112  QLineEdit *dacrvalue = new QLineEdit("");
113  dacrvalue->setFixedWidth(100);
114  dacrvalueList->push_back(dacrvalue);
115  QPushButton *button = new QPushButton(tr("SET %1").arg(mode));
116  button->setFixedWidth(150);
117  connect(button, SIGNAL(clicked()), signalMapper, SLOT(map()));
118  signalMapper->setMapping(button, i);
119  setButtonList->push_back(button);
120  QPushButton *shiftbutton = new QPushButton(tr("SHIFT %1").arg(mode));
121  shiftbutton->setFixedWidth(150);
122  connect(shiftbutton, SIGNAL(clicked()), shiftsignalMapper, SLOT(map()));
123  shiftsignalMapper->setMapping(shiftbutton, i);
124  shiftButtonList->push_back(shiftbutton);
125 
126  layout->addWidget(outputLabel, i + 2, 0, 1, 1);
127  layout->addWidget(dacvalue, i + 2, 1, 1, 1);
128  layout->addWidget(button, i + 2, 2, 1, 1);
129  layout->addWidget(dacrvalue, i + 2, 3, 1, 1);
130  layout->addWidget(shiftbutton, i + 2, 4, 1, 1);
131  }
132  connect(signalMapper, SIGNAL(mapped(int)), this, SLOT(setDacValue(int)));
133  connect(shiftsignalMapper, SIGNAL(mapped(int)), this, SLOT(setDacRValue(int)));
134 }
