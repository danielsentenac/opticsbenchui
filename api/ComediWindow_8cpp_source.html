150 void ComediWindow::InitRun() {
151  comediList->clear();
152  QSqlDatabase db = QSqlDatabase::database(dbPath);
153  QSqlQuery query(db);
154  query.exec("select name from comedi_settings");
155  while (query.next()) {
156  QString name = query.value(0).toString();
157  comediList->append(name);
158  }
159  if (comedi->comeditype == "COMEDICOUNTER") {
160  comediWidget->setComediList(comediList);
161  }
162  else if (comedi->comeditype == "COMEDIDAC") {
163  daccomediWidget->setComediList(comediList);
164  }
165 }
