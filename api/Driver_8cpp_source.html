244 int
245 Driver::SendGeneralCommand(char *buffer,string& rply) const
246 {
247  int retStatus = 1;
248  string answer = "";
249  string command = "";
250  char mybuffer[32];
251  vector<string> mBufferVect;
252  string strbuffer = buffer;
253 
254  if (!mBufferVect.empty()) mBufferVect.erase(mBufferVect.begin(),mBufferVect.end());
255  Tokenize(strbuffer,mBufferVect,",");
256  for (unsigned int i = 0 ; i < mBufferVect.size(); i++ ) {
257  sprintf(mybuffer,"%s\r",mBufferVect[i].c_str());
258  command = mybuffer;
259  QLOG_DEBUG ( ) << "Driver> Send Command " << QString(command.c_str());
261  // Apply Command
263  if (_pcommChannel->Write(command) < (int)command.length()) {
264  ReportCommError("Driver> Unable to write to port");
265  return -1;
266  }
267  usleep(100000);
268  }
270  // Read reply
272  if (!_pcommChannel->Read(answer)) {
273  ReportCommError("Driver> Unable to read from port");
274  return -2;
275  }
276  QLOG_DEBUG ( ) << "Driver> Reply to Command is :" << QString(answer.c_str());
277  rply = answer;
278  QLOG_DEBUG ( ) << "Driver> returning retStatus " << retStatus;
279  return retStatus;
280 }

282 void
283 Driver::Tokenize(const string& str,
284  vector<string>& tokens,
285  const string& delimiters) const
286 {
287  // Skip delimiters at beginning.
288  string::size_type lastPos = str.find_first_not_of(delimiters, 0);
289  // Find first "non-delimiter".
290  string::size_type pos = str.find_first_of(delimiters, lastPos);
291 
292  while (string::npos != pos || string::npos != lastPos)
293  {
294  // Found a token, add it to the vector.
295  tokens.push_back(str.substr(lastPos, pos - lastPos));
296  // Skip delimiters. Note the "not_of"
297  lastPos = str.find_first_not_of(delimiters, pos);
298  // Find next "non-delimiter"
299  pos = str.find_first_of(delimiters, lastPos);
300  }
301 }
