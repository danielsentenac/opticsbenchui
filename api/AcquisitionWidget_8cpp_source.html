33 QString FormatElapsed(qint64 elapsedMs) {
34  const qint64 totalSeconds = elapsedMs / 1000;
35  const qint64 hours = totalSeconds / 3600;
36  const qint64 minutes = (totalSeconds % 3600) / 60;
37  const qint64 seconds = totalSeconds % 60;
38  return QString("Elapsed: %1:%2:%3")
39  .arg(hours, 2, 10, QLatin1Char('0'))
40  .arg(minutes, 2, 10, QLatin1Char('0'))
41  .arg(seconds, 2, 10, QLatin1Char('0'));

378 void AcquisitionWidget::run() {
379  stopAcquisition(false);
380  QSqlQuery query(QSqlDatabase::database(path));
381  query.exec("update acquisition_sequence set status = ''");
382  query.exec("update acquisition_sequence set acquiring = ''");
383  query.exec("select record,type,instrument,settings,scanplan from acquisition_sequence "
384  "order by record");
385 
386  // Store acquisition sequence
387  sequenceList.clear();
388  totalAcqRecords = 0;
389  acqProgressValue = 0;
390  progressTickRecords.clear();
391  cur_record = 0;
392  while (query.next()) {
393  int seq_record = query.value(0).toInt();
394  if (seq_record < 0) {
395  continue;
396  }
397  AcquisitionSequence *sequence = new AcquisitionSequence();
398  sequence->record = cur_record++;
399  sequence->seq_record = seq_record;
400  sequence->instrumentType = query.value(1).toString();
401  sequence->instrumentName = query.value(2).toString();
402  sequence->settings = query.value(3).toString();
403  sequence->scanplan = query.value(4).toString();
404  sequenceList.append(sequence);
405  }
406  // Prepare sequence
407  for (int i = 0; i < sequenceList.size(); i++) {
408  AcquisitionSequence *sequence = sequenceList.at(i);
409  sequence->prepare();
410  }
411  // Compute progress ticks based on deepest loop level.
412  int maxDepth = 0;
413  int maxDepthProduct = 1;
414  QVector<int> loopStack;
415  for (int i = 0; i < sequenceList.size(); ++i) {
416  const AcquisitionSequence* sequence = sequenceList.at(i);
417  if (sequence->loop == AcquisitionSequence::IS_LOOP &&
418  sequence->loopends == AcquisitionSequence::LOOP_START &&
419  sequence->loopNumber > 0) {
420  loopStack.append(sequence->loopNumber);
421  int depth = loopStack.size();
422  int product = 1;
423  for (int v : loopStack) {
424  product *= v;
425  }
426  if (depth > maxDepth) {
427  maxDepth = depth;
428  maxDepthProduct = product;
429  progressTickRecords.clear();
430  progressTickRecords.insert(sequence->record);
431  } else if (depth == maxDepth) {
432  if (product > maxDepthProduct) {
433  maxDepthProduct = product;
434  }
435  progressTickRecords.insert(sequence->record);
436  }
437  } else if (sequence->loop == AcquisitionSequence::IS_LOOP &&
438  sequence->loopends == AcquisitionSequence::LOOP_END) {
439  if (!loopStack.isEmpty()) {
440  loopStack.removeLast();
441  }
442  }
443  }
444  if (maxDepth > 0) {
445  totalAcqRecords = maxDepthProduct;
446  } else {
447  totalAcqRecords = sequenceList.size();
448  }
449  if (motor) {
450  QSet<QString> motorNames;
451  for (int i = 0; i < sequenceList.size(); i++) {
452  const AcquisitionSequence *sequence = sequenceList.at(i);
453  if (sequence->instrumentType == "MOTOR") {
454  motorNames.insert(sequence->instrumentName);
455  }
456  }
457  for (const QString& name : motorNames) {
458  if (!name.isEmpty()) {
459  motor->connectMotor(name);
460  }
461  }
462  }
463  // Set File name and number from widget
464  QString filepath = acqfile + "_" + filenumber + ".h5";
465  acquisition->setFile(filepath, filenumber.toInt());
466  acquisition->setCamera(cameraList);
467  acquisition->setMotor(motor);
468  acquisition->setSuperK(superk);
469  acquisition->setDac(dac);
470  acquisition->setComediCounter(comedicounter);
471  acquisition->setComediDac(comedidac);
472  acquisition->setRaspiDac(raspidac);
473  acquisition->setSequenceList(sequenceList);
474  elapsedTimer.start();
475  elapsedLabel->setText("Elapsed: 00:00:00");
476  stopNoteLabel->setText("");
477  currentAcquiringRecordValue = -1;
478  if (totalAcqRecords > 0) {
479  acquisitionProgress->setRange(0, totalAcqRecords);
480  acquisitionProgress->setValue(0);
481  } else {
482  acquisitionProgress->setRange(0, 1);
483  acquisitionProgress->setValue(0);
484  }
485  elapsedTimerTick->start();
486  emit runningChanged(true);
487  acquisition->start();
488  QLOG_DEBUG() << "AcquisitionWidget::run started";
489 }

564 void AcquisitionWidget::getAcquiring(int record) {
565  QSqlQuery query(QSqlDatabase::database(path));
566 
567  if (totalAcqRecords > 0) {
568  if (!progressTickRecords.isEmpty()) {
569  if (progressTickRecords.contains(record)) {
570  acqProgressValue = qMin(acqProgressValue + 1, totalAcqRecords);
571  acquisitionProgress->setValue(acqProgressValue);
572  }
573  } else {
574  acquisitionProgress->setValue(qMin(record + 1, totalAcqRecords));
575  }
576  }
577  query.exec("update acquisition_sequence set acquiring = ''");
578 
579  int seq_record = -1;
580  if (record >= 0 && record < sequenceList.size()) {
581  const AcquisitionSequence* sequence = sequenceList.at(record);
582  if (sequence) {
583  seq_record = sequence->seq_record;
584  }
585  }
586  if (seq_record >= 0) {
587  const QVector<QPair<int, int>> loopIterations =
588  currentLoopIterationsForRecord(record);
589  QSet<int> loopRecords;
590  for (const QPair<int, int>& loopInfo : loopIterations) {
591  if (loopInfo.first < 0) {
592  continue;
593  }
594  loopRecords.insert(loopInfo.first);
595  query.prepare("update acquisition_sequence set acquiring = ? where record = ?");
596  query.addBindValue(QString("RUNNING LOOP=%1").arg(loopInfo.second));
597  query.addBindValue(loopInfo.first);
598  query.exec();
599  }
600  if (!loopRecords.contains(seq_record)) {
601  query.prepare("update acquisition_sequence set acquiring = ? where record = ?");
602  query.addBindValue(QString("RUNNING"));
603  query.addBindValue(seq_record);
604  query.exec();
605  }
606  currentAcquiringRecordValue = seq_record;
607  cur_record = seq_record;
608  } else {
609  currentAcquiringRecordValue = -1;
610  }
611  InitConfig(false);
612  acquisitionview->viewport()->update();
613 }

615 QVector<QPair<int, int>> AcquisitionWidget::currentLoopIterationsForRecord(
616  int recordIndex) const {
617  QVector<QPair<int, int>> iterations;
618  if (recordIndex < 0 || recordIndex >= sequenceList.size()) {
619  return iterations;
620  }
621 
622  int loopDepth = 0;
623  for (int i = recordIndex; i >= 0; --i) {
624  const AcquisitionSequence* sequence = sequenceList.at(i);
625  if (!sequence || sequence->loop != AcquisitionSequence::IS_LOOP) {
626  continue;
627  }
628  if (sequence->loopends == AcquisitionSequence::LOOP_END) {
629  loopDepth++;
630  continue;
631  }
632  if (sequence->loopends == AcquisitionSequence::LOOP_START) {
633  if (loopDepth > 0) {
634  loopDepth--;
635  continue;
636  }
637  if (sequence->loopNumber > 0 && sequence->seq_record >= 0) {
638  const int rawIteration =
639  sequence->loopNumber - sequence->remainingLoops + 1;
640  iterations.prepend(
641  qMakePair(sequence->seq_record,
642  qBound(1, rawIteration, sequence->loopNumber)));
643  }
644  }
645  }
646 
647  return iterations;
648 }
649 void AcquisitionWidget::setAcquiringToLastRecord() {

651  query.exec("update acquisition_sequence set acquiring = ''");
