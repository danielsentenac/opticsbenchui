41 //
42 // Init implementation
43 //
44 int DriverNewPort_NSC200::Init(string& rstateData) const
45 {
46  QLOG_DEBUG() << "DriverNewPort_NSC200::Init";
47  int retStatus = 0;
48  char buffer[BUFFER_SIZE];
49  string command;
50  string answerTE;
51  int tries = 0;
52  //
53  // Driver initialization
54  //
55  // parse driverSetting string for axis number
56  if (sscanf(_setting.c_str(),"axisNumber=%d delay=%d",
57  &_axisNumber, &_delay) != NB_ITEM_DRV_SETTING) {
58  ReportSettingError("DriverNewPort_NSC200> driverSetting string incorrect");
59  retStatus = -1;
60  return retStatus;
61  }
62  //
63  // check for axisNumber validity
64  //
65  QLOG_DEBUG() << "axis number = " << _axisNumber;
66 
67  if ((_axisNumber < 1) || (_axisNumber > MAX_DEVICES)) {
68  ReportRangeError("DriverNewPort_NSC200> axis number out of range");
69  retStatus = -1;
70  return retStatus;
71  }
72 
74  // Check error of the controller
76  while (answerTE == "") {
77  usleep(_delay);
78  memset(buffer,0,BUFFER_SIZE);
79  sprintf(buffer, "%dTE?\r", _axisNumber);
80  QLOG_DEBUG() <<"DriverNewPort_NSC200> INIT : send message : " << QString(buffer);
81  // send command
82  command = buffer;
83  if (_pcommChannel->Write(command,NULL) < (int)command.length()) {
84  QLOG_INFO() <<"DriverNewPort_NSC200> axisNumber " << _axisNumber << ": unable to write to port";
85  }
86  usleep(_delay);
87  // read reply
88  if (!_pcommChannel->Read(answerTE,NULL)) {
89  QLOG_INFO() <<"DriverNewPort_NSC200> axisNumber " << _axisNumber << ": unable to read from port";
90  QLOG_INFO() << "DriverNewPort_NSC200> Error code reply expected, but got reply "
91  << QString(answerTE.c_str());
92  }
93  usleep(_delay);
94  QLOG_DEBUG() << "Error code = " << QString(answerTE.c_str());
95  if (answerTE == "") {
96  QLOG_DEBUG() << "Trying again...";
97  }
98  tries++;
99  if (tries == MAX_TRIES) {
100  QLOG_INFO() << "Giving Up !";
101  retStatus = -1;
102  return retStatus;
103  }
104  }
105  return retStatus;
106 }

182

187  else
188  sprintf(buffer, "%dTP?\r", _axisNumber);

197  // read reply
198  if (!_pcommChannel->Read(answer,NULL)) {
199  QLOG_DEBUG() <<"DriverNewPort_NSC200> axisNumber " << _axisNumber << ": unable to read from port";
200  return (-1);
201  }
202  QLOG_DEBUG() << "DriverNewPort_NSC200> Getting answer = " << QString(answer.c_str());
203  // parse and extract position

212  sscanf (realAnswer, "%f",&position);

346 //
347 // OperationComplete implementation
348 //
349 // 1. send "xxTS?" and "xxTE?nn" to retrieve the status and error code of the controller
350 // 2. get the reply
351 // 3. parse the reply
352 // 4. compare the bits
353 //
354 //
355 int DriverNewPort_NSC200::OperationComplete(string& rstateData,
356  string actuatorSetting,
357  ADLimitSwitch& rlimitSwitch) const
358 {
359  QLOG_DEBUG() << "DriverNewPort_NSC200::OperationComplete";
360 
361  const int PARAMETER_OUT_OF_RANGE = 7;
362  const int UPPER_LIMIT_STATE = 26;
363  const int LOWER_LIMIT_STATE = 27;
364  const int MOTOR_NOT_ENABLE_STATE = 213;
365  const int PARAMETER_CHANGE_STATE = 226;
366  const int COMMAND_PARAMETER_MISSING_STATE = 38;
367  const int NO_MOTOR_CONNECTED_STATE = 8;
368  const int PARAMETER_OUT_OF_RANGE_STATE = 7;
369  const int INVALID_COMMAND_STATE = 6;
370  const int DRIVER_FAULT_OPEN_LOAD_STATE = 1;
371  const int DRIVER_FAULT_THERMAL_STATE = 2;
372  const int DRIVER_FAULT_SHORT_STATE = 3;
373  const int NO_ERRORS_STATE = 0;
374  const int MOTOR_ON_RUNNING_STATE = 80;
375  const int MOTOR_ON_STOP_STATE = 81;
376  const int MOTOR_OFF_STOP_STATE = 65;
377 
378  int retStatus = -1;
379  string command = "";
380  string answerTS = "";
381  string answerTE = "";
382  char buffer[BUFFER_SIZE];
383  int status;
384  int error;
385  stringstream axisStr;
386  stringstream channelStr;
387  string subTS = "";
388  string subTE = "";
389  const char* realAnswerTE;
390  const char* realAnswerTS;
391  // parse driverSetting string for axis number
392  if (sscanf(_setting.c_str(),"axisNumber=%d delay=%d",
393  &_axisNumber, &_delay) != NB_ITEM_DRV_SETTING) {
394  ReportSettingError("DriverNewPort_NSC200> driverSetting string incorrect");
395  retStatus = -1;
396  return retStatus;
397  }
398  // parse actuator configuration parameters
399  if (sscanf(actuatorSetting.c_str(),
400  "channel=%d vel=%f",
401  &_channel,&_vel) != NB_ITEM_INIT_SETTING) {
402  _channel = -1 ;/*init case*/
403  }
405  // create buffer with command xxTEnn? for controller Error Code query
407  memset(buffer,0,BUFFER_SIZE);
408  sprintf(buffer, "%dTE%d?\r", _axisNumber,_channel);
409  QLOG_DEBUG() << "DriverNewPort_NSC200> axisNumber " << _axisNumber << " channel " << _channel
410  << " buffer = " << QString(buffer);
411 
412  QLOG_DEBUG() <<"DriverNewPort_NSC200> send message : " << QString(buffer);
413  // send command
414  command = buffer;
415  if (_pcommChannel->Write(command,NULL) < (int)command.length()) {
416  QLOG_DEBUG() <<"DriverNewPort_NSC200> axisNumber " << _axisNumber << ": unable to write to port";
417  return (-1);
418  }
419  usleep(_delay);
420  // read reply
421  if (!_pcommChannel->Read(answerTE,NULL)) {
422  QLOG_DEBUG() <<"DriverNewPort_NSC200> axisNumber " << _axisNumber << ": unable to read from port";
423  QLOG_DEBUG() << "DriverNewPort_NSC200> Status reply expected, but got reply = " <<
424  QString(answerTE.c_str());
425  return (-1);
426  }
427  QLOG_DEBUG() << "Status reply = " << QString(answerTE.c_str());
428 
429  // Evaluate error code
430 
431  axisStr << _axisNumber;
432  channelStr << _channel;
433  if (answerTE.length() > 5) {
434  subTE = answerTE.substr(5);
435  realAnswerTE = subTE.c_str();
436  QLOG_DEBUG() << "Reply to TE? = " << QString(realAnswerTE);
437  sscanf (realAnswerTE, "%d",&error);
438  }
439  else
440  error = -1;
441  QLOG_DEBUG() << "axisNumber " << _axisNumber << " channel " << _channel << " has error code "
442  << error;
443 
444  if ( error == PARAMETER_OUT_OF_RANGE ) {
445  ReportRangeError(QString("DriverNewPort_NSC200> axisNumber %1 channel %2: Parameter out of range!")
446  .arg(QString(axisStr.str().c_str()))
447  .arg(QString(channelStr.str().c_str())));
448 
449  rstateData = "axisNumber=" + axisStr.str() + " channel=" + channelStr.str() +
450  ": Parameter out of range";
451  retStatus = 1;
452  }
453  else if ( error == UPPER_LIMIT_STATE ) {
454  QLOG_DEBUG() << "DriverNewPort_NSC200> axisNumber " << QString(axisStr.str().c_str())
455  << " channel " << QString(channelStr.str().c_str())
456  << ": Upper limit reached!";
457 
458  rstateData = "axisNumber=" + axisStr.str() + " channel=" + channelStr.str() +
459  ": Upper limit reached!";
460  retStatus = 1;
461  }
462  else if ( status == LOWER_LIMIT_STATE ) {
463  QLOG_DEBUG() << "DriverNewPort_NSC200> axisNumber " << QString(axisStr.str().c_str())
464  << " channel " << QString(channelStr.str().c_str())
465  << ": Lower limit reached!";
466 
467  rstateData = "axisNumber=" + axisStr.str() + " channel=" + channelStr.str() +
468  ": Lower limit reached!";
469  retStatus = 1;
470  }
471 
473  // create buffer with command xxTS? for controller Status query
475  memset(buffer,0,BUFFER_SIZE);
476  sprintf(buffer, "%dTS?\r", _axisNumber);
477  QLOG_DEBUG() << "DriverNewPort_NSC200> axisNumber " << _axisNumber << " channel " << _channel
478  << " buffer = " << QString(buffer);
479 
480  QLOG_DEBUG() <<"DriverNewPort_NSC200> send message : " << QString(buffer);
481  // send command
482  command = buffer;
483  if (_pcommChannel->Write(command,NULL) < (int)command.length()) {
484  QLOG_DEBUG() <<"DriverNewPort_NSC200> axisNumber " << _axisNumber << ": unable to write to port";
485  return (-1);
486  }
487  usleep(_delay);
488  // read reply
489  if (!_pcommChannel->Read(answerTS,NULL)) {
490  QLOG_DEBUG() <<"DriverNewPort_NSC200> axisNumber " << _axisNumber << ": unable to read from port";
491  QLOG_DEBUG() << "DriverNewPort_NSC200> Status reply expected, but got reply = " <<
492  QString(answerTS.c_str());
493  return (-1);
494  }
495  QLOG_DEBUG() << "Status reply = " << QString(answerTS.c_str());
496 
497  // Evaluate status of motion
498 
499  axisStr << _axisNumber;
500  if (answerTS.length() > 5) {
501  subTS = answerTS.substr(5);
502  realAnswerTS = subTS.c_str();
503  QLOG_DEBUG() << "Reply to TS? = " << QString(realAnswerTS);
504  sscanf (realAnswerTS, "%d",&status);
505  }
506  else
507  status = -1;
508  QLOG_DEBUG() << "axisNumber " << _axisNumber << " channel " << _channel << " has status "
509  << status;
510  if ( status == MOTOR_ON_STOP_STATE ) {
511  QLOG_DEBUG() << "DriverNewPort_NSC200> axisNumber " << QString(axisStr.str().c_str())
512  << ": Motor On, Motion not in progress";
513  retStatus = 1;
514  rstateData = "axisNumber=" + axisStr.str() + ": Motor On, Motion not in progress";
515  }
516  else if ( status == MOTOR_ON_RUNNING_STATE ) {
517  QLOG_DEBUG() << "DriverNewPort_NSC200> axisNumber " << QString(axisStr.str().c_str())
518  << ": Motor On, Motion in progress";
519  rstateData = "axisNumber=" + axisStr.str() + ": Motor On, Motion in progress";
520  retStatus = 0;
521  }
522  else if ( status == MOTOR_OFF_STOP_STATE ) {
523  QLOG_DEBUG() << "DriverNewPort_NSC200> axisNumber " << QString(axisStr.str().c_str())
524  << ": Motor Off, Motion not in progress";
525 
526  rstateData = "axisNumber=" + axisStr.str() + ": Motor Off, Motion not in progress";
527  retStatus = 1;
528  }
529  return retStatus;
530 }
