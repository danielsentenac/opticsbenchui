32 QString FormatElapsed(qint64 elapsedMs) {
33  const qint64 totalSeconds = elapsedMs / 1000;
34  const qint64 hours = totalSeconds / 3600;
35  const qint64 minutes = (totalSeconds % 3600) / 60;
36  const qint64 seconds = totalSeconds % 60;
37  return QString("Elapsed: %1:%2:%3")
38  .arg(hours, 2, 10, QLatin1Char('0'))
39  .arg(minutes, 2, 10, QLatin1Char('0'))
40  .arg(seconds, 2, 10, QLatin1Char('0'));

242 QVector<AnalysisTask> AnalysisWidget::buildTaskList() const {
243  QVector<AnalysisTask> tasks;
244  QSqlQuery query(QSqlDatabase::database(path));
245  query.exec("select record, codepath, arguments from analysis_sequence "
246  "order by record");
247  while (query.next()) {
248  const int record = query.value(0).toInt();
249  if (record < 0) {
250  continue;
251  }
252  AnalysisTask task;
253  task.record = record;
254  task.codePath = query.value(1).toString();
255  task.arguments = query.value(2).toString();
256  tasks.append(task);
257  }
258  return tasks;
259 }

314 void AnalysisWidget::analysisFinished(int record, bool success,
315  const QString& output) {
316  QLOG_INFO() << "AnalysisWidget::analysisFinished record=" << record
317  << " success=" << success;
318  if (!output.trimmed().isEmpty()) {
319  QLOG_INFO() << "Analysis output: " << output;
320  outputView->append(output);
321  }
322  if (analysisWasStopped && !success) {
323  statusLabel->setText("Stopped");
324  outputView->append(">> Stopped by user");
325  outputView->append(QString(">> PID %1 killed")
326  .arg(lastActivePid > 0 ? lastActivePid : 0));
327  } else {
328  statusLabel->setText(success ? "Done" : "Failed");
329  outputView->append(success ? ">> Done" : ">> Failed");
330  }
331  if (expectedTasks > 0) {
332  finishedTasks++;
333  if (finishedTasks >= expectedTasks && elapsedTimer.isValid()) {
334  elapsedLabel->setText(FormatElapsed(elapsedTimer.elapsed()));
335  }
336  }
