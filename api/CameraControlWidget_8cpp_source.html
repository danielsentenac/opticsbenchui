71 static int AutoEnumFromText(const QString &text, bool *ok) {
72  const QString trimmed = text.trimmed();
73  const QString lower = trimmed.toLower();
74  if (lower == "off") {
75  if (ok) {
76  *ok = true;
77  }
78  return 0;
79  }
80  if (lower == "once") {
81  if (ok) {
82  *ok = true;
83  }
84  return 1;
85  }
86  if (lower == "continuous" || lower == "on") {
87  if (ok) {
88  *ok = true;
89  }
90  return 2;
91  }
92  bool localOk = false;
93  const int numeric = trimmed.toInt(&localOk);
94  if (ok) {
95  *ok = localOk;
96  }
97  return numeric;
98 }

275 CameraControlWidget::snapShot() {
276 #ifdef NO_HDF5
277  Utils::EmitWarning(this, __FUNCTION__,
278  "HDF5 support is disabled. Snapshot in HDF5 format is unavailable.");
279  return;
280 #else
281 
282  uchar *img8 = nullptr;
283  int *img32 = nullptr;
284 
285  // Take snapshot
286  if ( camera->pixel_encoding == B8 ) {
287  QLOG_INFO() << " Take SNAPSHOT...";
288  img8 = camera->getSnapshot();
289  }
290  else
291  img32 = camera->getSnapshot32();
292  // Save File
293  QString filename = QFileDialog::getSaveFileName(
294  this, tr("Take snapshot in HDF5 format"),
295  Utils::DefaultHdf5Path("untitled"),
296  Utils::Hdf5FileDialogFilter());
297  if (filename == "") return;
298  filename = Utils::EnsureHdf5Extension(filename);
299  hid_t file_id = H5Fcreate(filename.toStdString().c_str(),
300  H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT);
301  if ( file_id < 0 ) {
302  QLOG_WARN () << "Unable to open file - " << filename ;
303  Utils::EmitWarning(this, __FUNCTION__, "Unable to open file - " + filename );
304  return;
305  }
306  // Save data to File
307  if ( camera->pixel_encoding == B8 ) {
308  QLOG_INFO () << "Saving image to file - " << filename ;
309  QLOG_INFO () << "width - " << camera->width ;
310  QLOG_INFO () << "height - " << camera->height ;
311  QLOG_INFO () << "min - " << camera->snapShotMin ;
312  QLOG_INFO () << "max - " << camera->snapShotMax ;
313 
314  H5IMmake_image_8bit(file_id,"SNAPSHOT",
315  camera->width,
316  camera->height,
317  img8);
318 
319  H5LTset_attribute_int(file_id, "SNAPSHOT", "min", &camera->snapShotMin,1);
320  H5LTset_attribute_int(file_id, "SNAPSHOT", "max", &camera->snapShotMax,1);
321  }
322  else {
323  QLOG_INFO() << "CameraControlWidget::snapShot> Take snapshot pixel encoding "
324  << camera->pixel_encoding
325  << " width " << camera->width
326  << " height " << camera->height;
327  hsize_t dset_dims[2];
328  dset_dims[0] = camera->height;
329  dset_dims[1] = camera->width;
330  H5LTmake_dataset_int(file_id,"SNAPSHOT32", 2,dset_dims,img32);
331  H5LTset_attribute_int(file_id, "SNAPSHOT32", "min", &camera->snapShotMin,1);
332  H5LTset_attribute_int(file_id, "SNAPSHOT32", "max", &camera->snapShotMax,1);
333  }
334  // Close file
335  H5Fclose(file_id);
336 #endif
337 }
